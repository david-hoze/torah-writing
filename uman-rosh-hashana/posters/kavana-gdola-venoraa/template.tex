\documentclass[12pt]{article}

% פונטים ו־RTL
\usepackage{fontspec}
\usepackage{polyglossia}
\setmainlanguage{hebrew}
\setotherlanguage{english}
\setmainfont{David}           % שנה לפונט העדיף אצלך
\newfontfamily\headingfont{David}
\newfontfamily\quotefont{David}

\usepackage{geometry}         % שוליים
\geometry{margin=2.5cm}

% כדי שפנדוק לא יקרוס ברשימות

  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% עיצוב ציטוטים
\usepackage{etoolbox}
\AtBeginEnvironment{quote}{\quotefont\itshape\small}

\fancyhf{}
\fancyhead[L]{\headingfont בס"ד בר"ה בה"ב} % ימין עליון: שם העלון
\fancyfoot[C]{\thepage}

\setlength{\parskip}{0.7em}

% --- דחיסת פריסת העמוד ---------------------------------------
\geometry{top=1.5cm,bottom=1.5cm,left=2cm,right=2cm} % שוליים

\linespread{0.94}                                     % צופף שורות קלות
