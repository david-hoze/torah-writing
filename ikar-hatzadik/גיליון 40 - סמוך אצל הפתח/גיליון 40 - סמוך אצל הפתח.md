## סָמוּךְ אֵצֶל הַפֶּתַח (סִפּוּר לְשַבָּת)

## ליקו"מ תנינא תורה מח

... וְדַע, שֶׁיֵּשׁ אֶחָד שֶׁכְּבָר הוּא אֵצֶל הַפֶּתַח שֶׁל הַקְּדֻשָּׁה, וְהוּא חוֹזֵר לַאֲחוֹרָיו מֵחֲמַת הַבִּלְבּוּלִים הַנַּ"ל, אוֹ שֶׁאֲזַי כְּשֶׁהוּא סָמוּךְ אֵצֶל הַפֶּתַח, אָז מִתְגַּבֵּר עָלָיו הַסִּטְרָא אָחֳרָא וְהַבַּעַל דָּבָר מְאֹד מְאֹד, רַחֲמָנָא לִצְלָן, בְּהִתְגַּבְּרוּת גָּדוֹל וְנוֹרָא מְאֹד מְאֹד, רַחֲמָנָא לִצְלָן, וְאֵין מַנִּיחִין אוֹתוֹ לִכָּנֵס לְתוֹךְ הַפֶּתַח, וּמַחֲמַת זֶה הוּא חוֹזֵר לְאָחוֹר, חַס וְשָׁלוֹם. כִּי כֵן דֶּרֶךְ הַבַּעַל דָּבָר וְהַסִּטְרָא אָחֳרָא, כְּשֶׁרוֹאֶה שֶׁהָאָדָם סָמוּךְ סָמוּךְ מַמָּשׁ לְשַׁעֲרֵי הַקְּדֻשָּׁה, וְכִמְעַט שֶׁיִּכְנֹס, אֲזַי הוּא מִתְפַּשֵּׁט עָלָיו בְּהִתְגַּבְּרוּת גָּדוֹל מְאֹד מְאֹד, רַחֲמָנָא לִצְלָן. עַל כֵּן צָרִיךְ אָז הִתְחַזְּקוּת גָּדוֹל נֶגְדּוֹ ... וְדַע, שֶׁכָּל הַתְּנוּעוֹת וְהַהַעְתָּקוֹת, שֶׁאַתָּה נִתָּק וְנֶעְתָּק בְּכָל פַּעַם אֵיזֶה מְעַט מִן גַּשְׁמִיּוּת לַעֲבוֹדָתוֹ יִתְבָּרַךְ, כֻּלָּם מִתְקַבְּצִים וּמִתְחַבְּרִים וּמִתְקַשְּׁרִים וּבָאִים לְעֶזְרָתְךָ בְּעֵת הַצֹּרֶךְ, דְּהַיְנוּ כְּשֶׁיֵּשׁ, חַס וְשָׁלוֹם, אֵיזֶה דֹּחַק וְעֵת צָרָה, חַס וְשָׁלוֹם. וְדַע, שֶׁהָאָדָם צָרִיךְ לַעֲבֹר עַל גֶּשֶׁר צַר מְאֹד מְאֹד, וְהַכְּלָל וְהָעִקָּר שֶׁלֹּא יִתְפַּחֵד כְּלָל ... וְהָעִקָּר לִהְיוֹת בְּשִֹמְחָה תָּמִיד. וִישַֹמַּח עַצְמוֹ בְּכָל מַה שֶּׁיּוּכַל, וַאֲפִלּוּ עַל יְדֵי מִלֵּי דִּשְׁטוּתָא, לַעֲשֹוֹת עַצְמוֹ כְּשׁוֹטֶה וְלַעֲשֹוֹת עִנְיְנֵי שְׁטוּת וּצְחוֹק אוֹ קְפִיצוֹת וְרִקּוּדִים, כְּדֵי לָבוֹא לְשִֹמְחָה, שֶׁהוּא דָּבָר גָּדוֹל מְאֹד:‏

## הסוד של החלל הפנוי

חַגַּי הִגִּיעַ לַמַּאֲפִיָּה בַּבֹּקֶר, וְאָסַף, שֻׁתָּפוֹ וַחֲבֵרוֹ הַקָּרוֹב, הִבְחִין שֶׁפָּנָיו אֵינָן כִּתְמוֹל שִׁלְשׁוֹם.. "מַדּוּעַ נָפְלוּ פָּנֶיךָ?" שָׁאַל אֶת חֲבֵרוֹ. "הֲשֹׁמֵר אָחִי אָנֹכִי?" עָנָה לוֹ חַגַּי כְּדַרְכּוֹ וְהִמְשִׁיךְ לַשֻּׁלְחָן שֶׁלּוֹ בַּמִּשְׂרָד.. אָסַף חִיֵּךְ, אַךְ מִיָּד לְאַחַר מִכֵּן הִמְשִׁיךְ לִשְׁאוֹל "לֹא בֶּאֱמֶת חַגַּי, אֲנִי מַכִּיר אוֹתְךָ לֹא מֵהַיּוֹם, אֲנִי יוֹדֵעַ שֶׁמַּשֶּׁהוּ קָרָה.. אַל תִּדְאַג אֲנִי לֹא אֲחַזֵּק אוֹתְךָ אִם לֹא תִּרְצֶה..". חַגַּי חִיֵּךְ. "שְׁנֵינוּ יוֹדְעִים שֶׁבַּסּוֹף אַתָּה כֵּן תְּחַזֵּק אוֹתִי, גַּם אֲנִי אֶרְצֶה וְגַם אִם לֹא.. אֲבָל בִּשְׁבִיל זֶה אֲנִי חָבֵר שֶׁלְּךָ.." אָסַף חִיֵּךְ וְהִקְשִׁיב לַחֲבֵרוֹ בִּמְלֹא תְּשׂוּמֶת הַלֵּב. "תִּשְׁמַע," הִתְחִיל חַגַּי "אַתָּה זוֹכֵר שֶׁסִּפַּרְתִּי לְךָ שֶׁיֵּשׁ לִי אֶת הָעִנְיָן הַזֶּה עִם הֶעָרוֹת בַּבַּיִת.. אַתָּה יוֹדֵעַ, אַתָּה טִפּוּס יוֹתֵר זוֹרֵם, וַאֲנִי קְצָת יוֹתֵר קַפְּדָן וְאוֹהֵב סֵדֶר וּשְׁלִיטָה.. מַה אֲנִי אַגִּיד לְךָ, בִּגְלַל זֶה אֲנַחְנוּ מַשְׁלִימִים אֶחָד אֶת הַשֵּׁנִי בַּמַּאֲפִיָּה הַזֹּאת, מִי יוֹדֵעַ מַה הָיָה קוֹרֶה אִם הָיִיתִי נוֹתֵן לְךָ לְבַלְגֶן כָּאן אֶת כָּל הַמָּקוֹם עִם הָאוֹרוֹת שֶׁלְּךָ...". אָסַף חִיֵּךְ וְהִנְהֵן בְּהַסְכָּמָה.. "אֲבָל אַתָּה יוֹדֵעַ, בַּבַּיִת.. אִשְׁתִּי וְהַיְלָדִים מְאֹד סוֹבְלִים מִמֶּנִּי.. אֲנִי יוֹדֵעַ שֶׁהָרַב אָרוּשׁ אוֹמֵר לֹא לְהָעִיר בִּכְלָל, אֲבָל אֲנִי פָּשׁוּט לֹא מְסֻגָּל לְהַפְסִיק... אֲנִי אוֹמֵר לְעַצְמִי, אִם אֲנִי רַק אַגִּיד אֶת זֶה וְאֶת זֶה וְאֶת זֶה אָז הַכֹּל יִסְתַּדֵּר וְהַכֹּל יִהְיֶה בִּשְׁלִיטָה, לָמָּה לֹא, רַק הֶעָרָה קְטַנָּה וְזֶהוּ, מַה הַבְּעָיָה... וְאָז אֲנִי שׁוּב שׁוֹאֵל אֶת עַצְמִי לָמָּה הֵם נִפְגָּעִים, וְלָמָּה הֵם מִתְרַחֲקִים מִמֶּנִּי... אֲנִי פָּשׁוּט לֹא שׁוֹלֵט בָּזֶה וּכְבָר מְיֹאָשׁ מֵעַצְמִי..". "אֲבָל אַתָּה זוֹכֵר מַה לָמַדְנוּ מֵרַבֵּינוּ.." מִהֵר אָסַף לוֹמַר, "כָּל תְּנוּעָה וּתְנוּעָה נֶחְשֶׁבֶת, וּבַסּוֹף זֶה עוֹזֵר שֶׁה' לוֹקֵחַ מִמְּךָ אֶת זֶה בְּעַצְמוֹ..". "אֲנִי יוֹדֵעַ, אֲנִי יוֹדֵעַ.." הִנְהֵן חַגַּי, "בַּהַתְחָלָה הִתְחַזַּקְתִּי, וְעָשִׂיתִי אֶת זֶה, חָמֵשׁ דַּקּוֹת לְהִתְאַפֵּק פֹּה, חָמֵשׁ דַּקּוֹת לְהִתְאַפֵּק שָׁם, מַה שֶׁיָּכֹלְתִּי.. אֲבָל הַמַּצָּב נִהְיָה הָזוּי.. אֲנִי נִכְשָׁל וְנִכְשַׁל וְנִכְשַׁל, וְכָל פַּעַם רִיב, וְעָגְמַת נֶפֶשׁ, וְעַד שֶׁמִּתְפַּיְּסִים.. אֲנִי מַרְגִּישׁ שֶׁאֲנִי פָּשׁוּט לֹא רָאוּי לְנַהֵל מִשְׁפָּחָה, וַאֲנִי לֹא מֵבִין לָמָּה ה' שָׂם אוֹתִי בְּמָקוֹם כָּזֶה, עִם יֵצֶר כָּזֶה..". אָסַף הִרְגִּישׁ אֶת הַכְּאֵב שֶׁל חֲבֵרוֹ.. "מְחִילָה,", הוֹסִיף חַגַּי, "אֲנִי מַאֲמִין בְּרַבֵּינוּ, אֲנִי מַאֲמִין בַּה'.. אֲבָל אֲנִי פָּשׁוּט מַרְגִּישׁ שֶׁאֲנִי בַּמָּקוֹם הֲלֹא נָכוֹן.. אוּלַי אָסוּר לְהַגִּיד אֶת הַדְּבָרִים הָאֵלֶּה.. אֲבָל זֶה לֹא סוּג שֶׁל הִתְעַלְּלוּת לָשִׂים בֶּן אָדָם כָּמוֹנִי בְּמַצַּב כָּזֶה וְלָתֵת לוֹ לְהִכָּשֵׁל שׁוּב וְשׁוּב וְשׁוּב? אֲנִי לֹא מַצְלִיחַ לְהָבִין..". אָסַף נֶאֱנָח וְהִנְהֵן בְּהִזְדַּהוּת. "תִּשְׁמַע חַגַּי, אֲנִי מֵבִין לְגַמְרֵי אֵיפֹה אַתָּה נִמְצָא.. גַּם לִי יֵשׁ אֶת הַהִתְמוֹדְדוּיוֹת שֶׁלִּי בַּבַּיִת, בְּתוֹר בֶּן אָדָם מְבוּלְגַן וּבְאוֹרוֹת יֶתֶר.. אֵין לִי תְּשׁוּבָה טוֹבָה לָמָּה ה' שָׂם אוֹתָנוּ, כָּל אֶחָד אֵיפֹה שֶׁהוּא נִמְצָא, בְּמָקוֹם בּוֹ אֲנַחְנוּ נוֹפְלִים שׁוּב וְשׁוּב וְשׁוּב.. אֲבָל אִם אֲנַחְנוּ רוֹצִים בֶּאֱמֶת לְהִתְחַזֵּק וּלְהַמְשִׁיךְ לְנַסּוֹת לְהִשְׁתַּפֵּר בָּעִנְיָנִים הָאֵלּוּ, אֲנַחְנוּ חַיָּבִים לִהְיוֹת בְּשִׂמְחָה.. זֶה מַה שֶׁרַבֵּינוּ מְלַמֵּד אוֹתָנוּ, הִנֵּה תִּרְאֶה מַה כָּתוּב כָּאן":

### ליקוטי עצות – ערך התחזקות אות יט

גם עצבות מזיק מאד ונותן כח להיצר הרע. על כן צריך להתגבר מאד לשמח את נפשו בכל דרכי העצות המבארים במקומם, כי עקר התחזקות הוא על ידי שמחה וחדוה כמו שכתוב: כִּי חֶדְוַת ה' הִיא מָעֻזְּכֶם:‏

"זֶהוּ.. נָכוֹן, אֲבָל אֵיךְ אֲנִי אֶשְׂמַח? אֲנִי מַרְגִּישׁ שֶׁהַמִּדּוֹת שֶׁלִּי הֵן כָּאֵלּוּ לֹא טוֹבוֹת, וְשֶׁאֲנִי לֹא שָׁוֶה כְּלוּם, וְשֶׁה' בְּדִיעֲבַד נוֹתֵן לְבֵן אָדָם גָּרוּעַ כָּמוֹנִי לַעֲבֹד עַל הַמִּדּוֹת הַגְּרוּעוֹת שֶׁלּוֹ, אֲבָל רַק בִּתְנַאי שֶׁאֲנִי אֶעֱבוֹר כָּאֵלּוּ קְשָׁיִים וּבִזְיוֹנוֹת, כִּי זֶה מַה שֶׁמַּגִּיעַ לְבֵן אָדָם כָּמוֹנִי.. אֵיךְ אֶפְשָׁר לִשְׂמֹחַ בְּחַיִּים כָּאֵלֶּה?...". "אֲנִי לְגַמְרֵי מֵבִין אוֹתְךָ" אָמַר אָסַף, "אֲבָל זֹאת לֹא הָאֱמֶת בֶּאֱמֶת.. זוֹכֵר אֶת הַחוֹבֶרֶת שֶׁהֶרְאֵיתִי לְךָ שָׁבוּעַ שֶׁעָבַר מֵאַחַד מֵרַבָּנֵי בְּרֶסְלֶב בְּיָמֵינוּ? הוּא כּוֹתֵב גַּם עַל זֶה":

### אתחלתא דגאולה – ירח האיתנים התשפ"ג

... כי רבינו אומר בתורה ס"ו, שיש 'קמץ דקדושה' שזה 'נקודת הקמץ', ותכלית הכיסופים של האדם זה להגיע לקמץ הזה. והקמץ הזה נקרא 'כתר', ולמה? – כתוב בתיקוני זוהר בגלל שהניקוד שלו מורכב מ'קו' שמתחתיו יש 'נקודה'. ומה הפשט של זה, למה זה כל כך גבוה? – אז מסביר רבינו בתורה ס"ו, שה'נקודה' זה 'נקודת הכסף', שאדם 'כוסף' לה' יתברך – בחינת 'לב איש הישראלי בוער עד אי"ס'. וה'פתח' (הקו של נקודת הקמץ), זה נקרא שיש דבר שמונע [וחוסם] אותו. כלומר, ברגע שאדם כוסף עד אין סוף, אז, דוקא הקב"ה עושה שיהיה 'רקיע', שזה נקרא שמעכבים אותו, וזה דווקא נקודת הקמץ!! ... שפה רבינו מגלה לנו סוד הכי גדול בעולם, וזה הסוד שלו בעצמו, הסוד של החלל הפנוי. והוא, שדווקא על ידי ה'פתח' וה'מניעה', על ידי ה'רקיע' וה'משטח' ששוטחים לו 'על הנקודה' – דווקא מזה נהיית לו ה'נקודה'!

"הֵבַנְתִּי, נָכוֹן, סִפַּרְתָּ לִי עַל זֶה.. שֶׁעַל יְדֵי הַמְּנִיעָה נִהְיָה הָרָצוֹן.. אֲבָל עַד כְּדֵי כָּךְ? אֲנִי לֹא מֵבִין אֶת זֶה, יֵשׁ לִי רָצוֹן לִהְיוֹת בַּעַל יוֹתֵר טוֹב, אַבָּא יוֹתֵר טוֹב.. זֶה לֹא מַסְפִּיק? לָמָּה לְהָבִיא אוֹתִי עַד כְּדֵי יֵאוּשׁ, עַד כְּדֵי שֶׁהַמְּרִירוּת הַזֹּאת מַמָּשׁ מוֹצִיאָה לִי אֶת הַטַּעַם מֵהַחַיִּים?". אָסַף הִנְהֵן בַּהֲבָנָה.. "נָכוֹן, אַתָּה צוֹדֵק.. אֲבָל זֶה בְּדִיּוּק הָעִנְיָן שֶׁל הֶחָלָל הַפָּנוּי.. רַבֵּינוּ כּוֹתֵב שֶׁה' בֶּאֱמֶת נִמְצָא גַּם בַּחֲלַל הַפָּנוּי.. גַּם בְּצִמְצוּם, גַּם בִּמְנִיעָה נִמְצָא ה', גַּם בָּ'הִתְעַלְּלוּת' הַבִּלְתִּי מוּבֶנֶת שֶׁאַתָּה עוֹבֵר, ה' נִמְצָא, וְהוּא עוֹשֶׂה אֶת הַכֹּל מִתּוֹךְ רַחֲמִים אֵין סוֹפִיִּים..". חַגַּי הֵרִים אֶת גַּבּוֹתָיו.. "בֶּאֱמֶת?" שָׁאַל בִּפְלִיאָה. "כֵּן, פָּשׁוּט הָעִנְיָן בַּחֲלַל הַפָּנוּי הוּא שֶׁאֲנַחְנוּ לֹא יְכוֹלִים לְהָבִין אֶת זֶה בָּעוֹלָם הַזֶּה.. רַק בָּעוֹלָם הַבָּא יִהְיוּ לָנוּ כֵּלִים לְהָבִין שֶׁבְּעֶצֶם הַצִּמְצוּם וְהַדִּינִים שֶׁאֲנַחְנוּ עוֹבְרִים כְּדֵי לִבְחֹן אוֹתָנוּ, הֵם רַחֲמִים אֵין סוֹפִיִּים שֶׁל ה' יִתְבָּרַךְ עָלֵינוּ.. אֲנַחְנוּ מְבִינִים בְּאוֹפֶן כְּלָלִי שֶׁזֶּה כְּדֵי לִיצוֹר כִּסּוּפִים, שֶׁהֵם בְּעֶצֶם מַתָּנָה מֵה' יִתְבָּרַךְ, וְגַם עֲלֵיהֶם נְקַבֵּל שָׂכָר בָּעוֹלָם הַבָּא, אֲבָל לְהָבִין עַד הַסּוֹף לָמָּה זֶה צָרִיךְ לִהְיוֹת כָּל כָּךְ קָשֶׁה, כָּל כָּךְ מַר, כָּל כָּךְ מְיָיאֵשׁ.. וּלְמַה אֲנַחְנוּ זוֹכִים כְּשֶׁאֲנַחְנוּ עוֹבְרִים אֶת הַ'מִּבְחָנִים' הַקָּשִׁים הָאֵלּוּ, אֶת זֶה נָבִין בִּשְׁלֵמוּת רַק בָּעוֹלָם הַבָּא.. בָּעוֹלָם הַזֶּה אֲנַחְנוּ יְכוֹלִים רַק לְהַאֲמִין בָּזֶה...". חַגַּי הִתְחִיל לְהַנְהֵן בַּהֲבָנָה, כְּשֶׁהַדְּבָרִים הָעֲמֻקִּים מְחַלְחְלִים אַט אַט אֶל לִבּוֹ. "אֲנִי מֵבִין.. אֲנִי אִישִׁית רוֹצֶה לְהַצְלִיחַ בַּחַיִּים בְּצוּרָה יַחֲסִית פְּשׁוּטָה, לִהְיוֹת בַּעַל טוֹב, אַבָּא טוֹב, יְהוּדִי טוֹב.. אֲבָל ה' כָּל כָּךְ מְרַחֵם עָלַי, וְכָל כָּךְ אוֹהֵב אוֹתִי, שֶׁהוּא רוֹצֶה מִמֶּנִּי דְּבָרִים הַרְבֵּה יוֹתֵר נִשְׂגָּבִים מִמַּה שֶׁאֲנִי מְאַחֵל לְעַצְמִי.. הוּא רוֹצֶה לְהַשְׁפִּיעַ לִי כָּזֶה שֶׁפַע עָצוּם.. עֶצֶם זֶה שֶׁאֲנִי לֹא מֵבִין מַה קוֹרֶה אִתִּי וְלָמָּה, זֶה רַק סִימָן לְזֶה שֶׁזֶּה שֶׁפַע הַרְבֵּה יוֹתֵר גָּדוֹל מִמַּה שֶׁאֲנִי מְסֻגָּל לְהַשִּׂיג בְּשִׂכְלִי..". אָסַף חִיֵּךְ לְנֹכַח דְּבָרָיו שֶׁל חֲבֵרוֹ וְהוֹסִיף, "נָכוֹן, וְעֶצֶם זֶה שֶׁאֲנַחְנוּ לֹא מְבִינִים בִּשְׁלֵמוּת לָמָּה ה' בָּרָא מְנִיעוֹת, לָמָּה ה' בָּרָא מַלְאָכִים שֶׁמְּקַטְרְגִים עָלֵינוּ, לָמָּה כָּל דָּבָר בַּקְּדֻשָּׁה בָּא בְּכָזֶה קוֹשִׁי, בְּכָזֶה עָמָל, בְּכָזֹאת מְרִירוּת בַּהַתְחָלָה, לָמָּה ה' הָרַחֲמָן לֹא פָּשׁוּט שָׂם אוֹתָנוּ בְּגַן עֵדֶן וְזֶהוּ.. זֶה חֵלֶק מֵהַסּוֹד שֶׁל הֶחָלָל הַפָּנוּי.. מֵהַצִּמְצוּם הַזֶּה שֶׁה' צִמְצֵם אֶת רַחֲמָיו הָאֵין סוֹפִיִּים כְּדֵי לָתֵת לָנוּ מָקוֹם.. מָקוֹם לְהוֹכִיחַ אֶת עַצְמֵנוּ בֶּאֱמֶת, הַרְבֵּה יוֹתֵר מִמַּה שֶׁאֲנַחְנוּ חוֹשְׁבִים שֶׁצָּרִיךְ..". "הַרְבֵּה יוֹתֵר מִמַּה שֶׁאֲנַחְנוּ מְסֻגָּלִים לְהָבִין שֶׁצָּרִיךְ" הוֹסִיף חַגַּי. אָסַף חִיֵּךְ בְּשִׂמְחָה וּבְסִפּוּק.. "נוּ, נוּ, רוֹאֶה, בַּסּוֹף אַתָּה מְחַזֵּק אוֹתִי..". חַגַּי חִיֵּךְ, "תּוֹדָה אָסַף, הָיִיתִי צָרִיךְ אֶת זֶה.. נָתַתָּ לִי כֹּחַ לְהַמְשִׁיךְ.. אֲנִי עַכְשָׁיו סוֹף סוֹף מֵבִין מַה רַבֵּינוּ אָמַר בְּסוֹף הַתּוֹרָה.. וְהָעִקָּר לִהְיוֹת בְּשִׂמְחָה.. הֵבַנְתִּי דֵּי בַּהַתְחָלָה אֶת כָּל הַחִזּוּקִים, שֶׁכָּל תְּנוּעָה שֶׁאֲנִי עוֹשֶׂה חֲשׁוּבָה וִיקָרָה בְּעֵינֵי ה' יִתְבָּרַךְ, שֶׁבַּסּוֹף הַכֹּל יִצְטַבֵּר לְכָךְ שֶׁאוּכַל בֶּאֱמֶת לָצֵאת מֵאֵיפֹה שֶׁאֲנִי נִמְצָא לְמָקוֹם יוֹתֵר טוֹב מִבְּחִינָה רוּחָנִית.. אֲבָל הָעִקָּר שֶׁל כָּל הַחִזּוּקִים זֶה לְהָבִין שֶׁה' אוֹהֵב אוֹתִי, שֶׁכָּל מַה שֶּׁקּוֹרֶה לִי זֶה לְטוֹבָתִי הָאִישִׁית, וְה' פָּשׁוּט לֹא מְוַתֵּר עָלַי.. הוּא יוֹדֵעַ לְמַה אֲנִי בֶּאֱמֶת מְסֻגָּל, וּלְמַה אֲנִי אֶזְכֶּה אִם אֲנִי בֶּאֱמֶת אֶעֱמֹד בְּמַה שֶׁעוֹבֵר עָלַי, וְהוּא לֹא נוֹתֵן לִי לִחְיוֹת אֶת הַחַיִּים הַשְּׁלֵוִים וְהַבֵּינוֹנִים שֶׁאֲנִי מְאַחֵל לְעַצְמִי.. אֵין שִׂמְחָה יוֹתֵר גְּדוֹלָה מִזֶּה, שֶׁאֲנִי יוֹדֵעַ שֶׁה' אוֹהֵב אוֹתִי, שֶׁהוּא אִתִּי..". אָסַף אָחַז בִּכְתֵפוֹ שֶׁל חֲבֵרוֹ וְחִיֵּךְ.. "נָכוֹן מְאֹד.. אָדָם שָׂמֵחַ מָתַי שֶׁהוֹלֵךְ לוֹ בַּחַיִּים, וְרַבֵּינוּ מְסַפֵּר לָנוּ שֶׁלַּמְרוֹת שֶׁאֲנַחְנוּ חוֹשְׁבִים שֶׁלֹּא הוֹלֵךְ לָנוּ כְּלוּם בַּחַיִּים, בְּעֶצֶם זֶה בְּדִיּוּק הָפוּךְ... אֲנִי שָׂמֵחַ שֶׁיָּכֹלְתִּי לַעֲזֹר חַגַּי, אֲבָל מַה אַתָּה אוֹמֵר, אוּלַי נַתְחִיל לֶאֱפוֹת קְצָת? בְּכָל זֹאת, זוֹ מַאֲפִיָּה עַד כַּמָּה שֶׁאֲנִי זוֹכֵר..". חַגַּי חִיֵּךְ וְהִנְהֵן בְּהַסְכָּמָה, וּשְׁנֵיהֶם פָּנוּ אֶל עֵבֶר הַמִּטְבָּח, לְהַתְחִיל יוֹם חָדָשׁ בְּמַאֲפִיַּת 'טָרִי וּבָרִיא'.

## הַחִזּוּקִים הִפְסִיקוּ לַעֲבֹד

שָׁבוּעַ עָבַר מֵאָז אוֹתוֹ יוֹם בַּמַּאֲפִיָּה, וְהָעִנְיָנִים דֵּי חָזְרוּ לְעַצְמָם, אַךְ אָסַף שָׂם לֵב שֶׁחַגַּי מְאַחֵר קְצָת הַיּוֹם.. הוּא הִתְחִיל אֶת הָעֲבוֹדָה לְבַד, כְּשֶׁהוּא מִתְקַשֵּׁר מִדֵּי פַּעַם לַחֲבֵרוֹ הַטּוֹב, אַךְ לְלֹא מַעֲנֵה. לְמַזָּלוֹ בְּאוֹתוֹ הַיּוֹם לֹא הָיְתָה כָּל כָּךְ הַרְבֵּה עֲבוֹדָה, וְהוּא הִצְלִיחַ לְהַעֲבִיר אֶת הַיּוֹם לְלֹא עֶזְרָתוֹ שֶׁל חַגַּי. אַךְ עֲדַיִן, הָעֻבְדָּה שֶׁחַגַּי לֹא הִגִּיעַ כָּל אוֹתוֹ הַיּוֹם, לֹא הִתְקַשֵּׁר וְלֹא עָנָה לְפֶלֶאפוֹנִים, הִדְאִיגָה אוֹתוֹ.. הוּא הִשְׁתַּדֵּל לְהַמְשִׁיךְ אֶת הַיּוֹם בְּשִׂמְחָה וֶאֱמוּנָה, כְּשֶׁלְּפֶתַע, מַמָּשׁ לִקְרַאת הַסְּגִירָה, הוֹפִיעַ חַגַּי בְּפֶתַח הַמַּאֲפִיָּה. "אוֹ, הִנֵּה אַתָּה.. מַה קוֹרֶה? הַכֹּל בְּסֵדֶר?" שָׁאַל אָסַף בַּחֲמִימוּת. חַגַּי הִסְתַּכֵּל עָלָיו בְּמַבַּט עָיֵף. "זֶהוּ, נִגְמְרוּ הַכֹּחוֹת" פָּלַט בְּשָׂפָה רָפָה.. "שׁוּב פַּעַם נְפִילָה.. שׁוּב פַּעַם רִיב בַּבַּיִת.. אֲנִי לֹא יָכוֹל לְהַמְשִׁיךְ יוֹתֵר.. אֲנִי הַבַּעַל הֲכִי גָּרוּעַ בָּעוֹלָם... אִשְׁתִּי בְּקֹשִׁי מְדַבֶּרֶת אִתִּי.. אֵין לִי מֻשָּׂג מַה לַעֲשׂוֹת, הַחַיִּים שֶׁלִּי אֲבוּדִים לְגַמְרֵי.. אֲנִי כְּבָר לֹא יָכוֹל לְהַמְשִׁיךְ כָּכָה יוֹתֵר..". אָסַף הִסְתַּכֵּל עַל חֲבֵרוֹ הַטּוֹב וְהִנְהֵן בְּהַקְשָׁבָה.. "תַּקְשִׁיב אָסַף," אָמַר חַגַּי בְּעוֹדוֹ מְחַוֶּה בְּיָדָיו תְּנוּעָה שֶׁל הִסְתַּיְּגוּת, "אֲנִי אוֹמֵר לְךָ, אֲנִי מֵבִין אֶת כָּל הַחִזּוּקִים, אֶת כָּל העומקים שֶׁל רַבֵּינוּ, אֲנִי אֲפִילוּ מְנַסֶּה בְּעַצְמִי לְהִתְחַזֵּק בָּזֶה, לַחְזוֹר עַל כָּל הַדְּבָרִים שֶׁהֵבַנּוּ בְּיַחַד.. אֲבָל זֶהוּ, אֲנִי בְּמַצַּב שֶׁהַדְּבָרִים הָאֵלּוּ כְּבָר לֹא עוֹבְדִים.. אֲנִי יוֹדֵעַ שֶׁבְּדֶרֶךְ כְּלָל יֵשׁ לְךָ דְּבָרִים מְאֹד חֲכָמִים לְהַגִּיד אָסַף, וְאוּלַי אַתָּה אֲפִילוּ תְּחַדֵּשׁ לִי דְּבָרִים שֶׁאֲנִי לֹא יוֹדֵעַ, אֲבָל אֲנִי בְּמַצַּב שֶׁאֲנִי כְּבָר לֹא יָכוֹל לִשְׁמֹעַ וּלְהָכִיל שׁוּם דָּבָר... אַתָּה לֹא מֵבִין מַה הָיָה לִי בַּבַּיִת.. הִסְתּוֹבַבְתִּי כָּל הַיּוֹם בְּגִנּוֹת צִבּוּרִיּוֹת כְּשֶׁהַמֹּחַ שֶׁלִּי מָלֵא בְּמַחְשָׁבוֹת שֶׁל אֵיךְ מְסַיְּמִים אֶת כָּל הַסִּפּוּר הַזֶּה, עַד שֶׁנִּמְאַס לִי וּבָאתִי לְכָאן..". אָסַף נָשַׁם עֲמֻקּוֹת. "תַּקְשִׁיב, אֲנִי דֵּי סָגַרְתִּי הַכֹּל לְבַד כָּאן.. לְמַזָּלֵנוּ הַיּוֹם הָיָה יוֹם רָגוּעַ יַחֲסִית.. יֵשׁ לִי רַעְיוֹן.. מַה דַעְתָּךְ עַל לִנְסֹעַ לְאַנְשֶׁהוּ?". חַגַּי כִּוֵוץ גַּבּוֹתָיו. הוּא הָיָה נִרְאֶה עָיֵף וּמֻתָּשׁ.. "שָׂדֶה? הִתְבּוֹדְדוּת? תַּקְשִׁיב אָסַף, אַתָּה לֹא מֵבִין מַה הַמַּצָּב שֶׁלִּי, אגרתי אֶת כָּל הַכֹּחוֹת שֶׁלִּי כְּדֵי לָבוֹא לְכָאן וְלֹא לִקְפֹּץ מֵאֵיזֶשֶׁהוּ גַּג..". אָסַף הֵנִיד רֹאשׁוֹ בְּבִטּוּל, "לֹא, לֹא, מַמָּשׁ לֹא.. זֶה מַשֶּׁהוּ אַחֵר, בֹּא תִּרְאֶה, סְמֹךְ עָלַי..". "טוֹב, נוּ, מַה אֲנִי כְּבָר יָכוֹל לְהַפְסִיד, זֶה בֶּטַח יוֹתֵר טוֹב מִכָּל רַעְיוֹן שֶׁאֲנִי יָכוֹל לַחְשֹׁב עָלָיו עַכְשָׁיו.." אָמַר חַגַּי וְגָרַר אֶת עַצְמוֹ מֵהַכִּסֵּא הַמִּשְׂרַדִי.

## מִנְהָרָה צָרָה מְאֹד

"מַה?! לְמַה הֵבֵאתָ אוֹתִי לְכָאן?" הִזְדַּעֵק חַגַּי כְּשֶׁרָאָה לְאֵיפֹה אָסַף לָקַח אוֹתוֹ. הֵם הִגִּיעוּ לְגַן שַׁעֲשׁוּעִים שֶׁנִּמְצָא בִּקְצֵה הָעִיר, שֶׁכָּרֶגַע, עֶקֶב שְׁעַת הַלַּיְלָה הַמְּאֻחֶרֶת, עָמַד רֵיק וְשׁוֹמֵם. "זֶה מָקוֹם שֶׁאֲנִי הוֹלֵךְ אֵלָיו כְּשֶׁאֲנִי בַּמַּצָּב שֶׁלְּךָ.." אָמַר אָסַף. "אַל תִּדְאַג, זֶה מְאֹד עוֹזֵר.. בֹּא נַתְחִיל" אָמַר אָסַף וְהִתְחִיל לָלֶכֶת בְּהֶחְלֵטִיּוּת אֶל עֵבֶר הַנַּדְנֵדָה. חַגַּי הַמֻּפְתָּע הִתְעַכֵּב לְרֶגַע מֵאֲחוֹרָיו, אֲבָל מִיָּד הִתְחִיל לָלֶכֶת אַחֲרָיו. אָסַף הִתְיַשֵּׁב עַל הַצָּד הָרָחוֹק שֶׁל הַנַּדְנֵדָה. "בְּבַקָּשָׁה אֲדֹנִי" הִצְבִּיעַ עַל הַצָּד הַשֵּׁנִי, "אַתָּה מֻזְמָן לָשֶׁבֶת". חַגַּי עָצַר לְכַמָּה רְגָעִים, אַךְ אִלֵּץ אֶת עַצְמוֹ לָשֶׁבֶת בְּסוֹפוֹ שֶׁל דָּבָר. אָסַף מִיָּד חִיֵּךְ וְהִתְחִיל לָשִׁיר בְּהִתְלַהֲבוּת.. "נָד נֵד, נָד נֵד, רֵד עָלֶה עָלֶה וָרֵד, נָד נֵד, נָד נֵד, רֵד עָלֶה עָלֶה וָרֵד". חִיּוּךְ קַל עָלָה עַל שְׂפָתָיו שֶׁל חַגַּי.. "אַתָּה נִשְׁמַע כְּמוֹ הַגַּנֶּנֶת שֶׁל הַבַּת שֶׁלִּי, אַתָּה יוֹדֵעַ". "מִלְמַעְלָה עַד לְמַטָּה, רַק אֲנִי, אֲנִי וְאַתָּה, נָד נֵד, נָד נֵד, רֵד עָלֶה עָלֶה וָרֵד". "אַתָּה יוֹדֵעַ שֶׁזֶּה שִׁיר שֶׁל צִיּוֹנִים, כֵּן?". "אֲנַחְנוּ מַעֲלִים אֶת הַנִּצּוֹצוֹת, אַל תִּדְאַג.. נָד נֵד, נָד נֵד, רֵד עָלֶה עָלֶה וָרֵד..". הַבָּעַת הַיֵּאוּשׁ קְצָת הִסְתַּלְּקָה מִפָּנָיו שֶׁל חַגַּי וּפִנְּתָה אֶת מְקוֹמָהּ לְחִיּוּךְ מְאֻפָּק.. "אוֹ, אֲנַחְנוּ מִתְקַדְּמִים" אָמַר אָסַף בְּהִתְלַהֲבוּת, "לַתַּחֲנָה הַבָּאָה!" אָסַף יָרַד מֵהַנַּדְנֵדָה וְהִתְחִיל לָלֶכֶת בְּהֶחְלֵטִיּוּת אֶל עֵבֶר הַמַּגְלֵשָׁה.. הוּא הִפְנָה מַבָּט רְצִינִי לְעֵבֶר חַגַּי שֶׁהָלַךְ מֵאֲחוֹרָיו.. "זֶה מַמָּשׁ מִתְקַן מַפְחִיד, שֶּׁתֵּדַע לְךָ..". חַגַּי הִסְתַּכֵּל עַל הַמַּגְלֵשָׁה.. לְהַעֲרָכָתוֹ הִיא הָיְתָה מְיֹעֶדֶת לְגִילָאִים 6-8. "אֲנִי מֵבִין..". "חַכֵּה" אָמַר אָסַף בְּהִתְרַגְּשׁוּת, "עוֹד לֹא עָלִיתָ, זֶה פַּחַד פְּחָדִים!" אָמַר וְהִתְחִיל לַעֲלוֹת עַל הַסֻּלָּם שֶׁל הַמְּתַקֵּן. "ווווווהווהוהווהוווהווו!" הֵרִים אָסַף אֶת יָדָיו לְמַעְלָה בְּעוֹדוֹ מִתְגָּלֵשׁ לְמַטָּה בְּמוֹרָד הַמַּגְלֵשָׁה. חַגַּי הִבְלִיעַ חִיּוּךְ וְעָלָה אַחֲרָיו. "וווהוהווהו" הוֹסִיף חַגַּי בְּקוֹל נָמוּךְ וְהִתְגַּלֵּשׁ אַחֲרָיו.. "נוּ, אַתָּה רוֹאֶה שֶׁזֶּה מְטֹרָף?! לֹא קָפַץ לְךָ הַלֵּב כְּשֶׁהִתְגַלַשְׁתָּ?!". צְחוֹק קַל עָלָה מִכִּוּוּנוֹ שֶׁל חַגַּי.. "יוֹפִי!" אָמַר אָסַף בְּסִפּוּק, "אֲנַחְנוּ מִתְקַדְּמִים לַשָּׁלָב הַבָּא!" אָמַר וְנִכְנַס בְּתוֹךְ הַמִּנְהָרָה שֶׁל הַמִּתְקַן שֶׁמּוֹבִילָה לְמַגְלְשָׁה אַחֶרֶת, יוֹתֵר גְבֹהָה. "אַשְׁרֵינוּ, מַה טוֹב חֶלְקֵנוּ שֶׁזָּכִינוּ לְהִתְקָרֵב לְרַבֵּינוּ" הִתְחִיל אָסַף לָשִׁיר וְלִרְקֹד בְּהִתְלַהֲבוּת וּבִדְבֵקוּת בְּעוֹדוֹ הוֹלֵךְ עַל אַרְבַּע בְּתוֹךְ הַמִּנְהָרָה שֶׁבְּקֹשִׁי הֵכִילָה אוֹתוֹ. "אָסַף, זֶה כּוֹאֵב בַּבִּרְכַּיִים, לְאֵיפֹה הֵבֵאתָ אוֹתִי?!" אָמַר חַגַּי בְּעוֹדוֹ מְנַסֶּה לַעֲקֹב אַחֲרָיו. "אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה, אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה" הִמְשִׁיךְ אָסַף בִּדְבֵקוּת. חַגַּי, שֶׁנִּסָּה בְּאִטִּיּוּת לְהִתְקַדֵּם עַל אַרְבַּע בְּתוֹךְ הַמִּנְהָרָה הַצָּרָה כְּבָר לֹא יָכַל לַעֲמֹד בַּמַּצָּב הַמְגֻחָךְ.. צְחוֹק הָחֵל לְהִתְגַּלְגֵּל מִפִּיו בְּעַל כָּרְחוֹ, "אַשְׁרֵנוּ מַה טוֹב חֶלְקֵנוּ שֶׁזָּכִינוּ לְהִתְקָרֵב לְרַבֵּינוּ" הִצְטָרֵף לַשִּׁירָה.. אָסַף חִיֵּךְ בְּסִפּוּק.. "אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה, אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה, אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה.. אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה, אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה, אוּמַאן אוּמַאן רֹאשׁ הַשָּׁנָה..". שְׁנֵי הַחֲבֵרִים הִגִּיעוּ לְבַסּוֹף לַמַּגְלְשָׁה הַגְּבוֹהָה וְהַמְפֻתֶּלֶת, וְהִתְגַּלְּשׁוּ תּוֹךְ כְּדֵי שִׁירָה, צְחוֹק וְקוֹלוֹת מְשֻׁנִּים...

## דִּבּוּר אֶחָד שֶׁל אֱמֶת

"תּוֹדָה אָסַף, עָזַרְתָּ לִי קְצָת לִשְׁכֹּחַ מִכָּל הַמַּצָּב.." אָמַר חַגַּי הַמְּחֻיָּךְ לְאָסָף.. "בְּשִׂמְחָה, בִּשְׁבִיל זֶה יֵשׁ חֲבֵרִים...". "תִּשְׁמַע, נָתַתָּ לִי קְצָת כֹּחַ, אֲבָל אֲנִי עֲדַיִן צָרִיךְ לְבָרֵר לְעַצְמִי אֵיךְ אֲנִי מַמְשִׁיךְ.. אֲנִי רוֹצֶה לְהִשָּׁאֵר כָּאן וּלְהִתְבּוֹדֵד קְצָת, אוּלַי אֶמְצָא לִי אֵיזוֹ תובנה שֶׁתַּעֲזֹר לִי..". "מְעֻלֶּה" אָמַר אָסַף בְּסִפּוּק וְטָפַח עַל שִׁכְמוֹ, "אַל תִּדְאַג צַדִּיק, ה' אִתְּךָ, וַאֲנִי אִתְּךָ.." אָמַר וְהִתְחִיל לָלֶכֶת לְכִוּוּן הָרֶכֶב. חַגַּי חִכָּה עַד שֶׁאָסַף יֵעָלֵם מֵהָאֹפֶק וְהִתְחִיל לְדַבֵּר. "ה' יִתְבָּרַךְ...". פִּיו שֶׁל חַגַּי פִּתְאוֹם נֶאֱלַם... הוּא הִתְכַּוֵּן לְהִתְפַּלֵּל עַל הַמַּצָּב שֶׁלּוֹ, לְבַקֵּשׁ יְשׁוּעָה וְהַדְרָכָה.. זֶה אָמְנָם הָיָה הַדָּבָר הַהֶגְיוֹנִי לַעֲשׂוֹת, אַךְ לִבּוֹ לֹא הָיָה אִתּוֹ... "דִּבּוּרִים שֶׁל אֱמֶת.." מִלְמֵל לְעַצְמוֹ חַגַּי, "אֶפְשָׁר לְדַבֵּר מֵהַמַּדְרֵגָה שֶׁלִּי, מֵאֵיפֹה שֶׁאֲנִי בֶּאֱמֶת נִמְצָא..". חַגַּי נִסָּה לְהַרְגִּישׁ אֶת עַצְמוֹ, וְהִתְחִיל לְהִתְבּוֹנֵן בְּעוֹלָמוֹ הַפְּנִימִי. לְפֶתַע בָּקַע דִּבּוּר אַמִּיץ מִפִּיו, "אֲנִי יוֹדֵעַ, אֲנִי מֵבִין ה' יִתְבָּרַךְ, אֲנִי אָמוּר לִרְצוֹת עַכְשָׁיו לְהִשְׁתַּפֵּר, לְבַקֵּשׁ מִמְּךָ שֶׁתּוֹצִיא אוֹתִי מֵאֵיפֹה שֶׁאֲנִי נִמְצָא.. אָז זֶהוּ, מִקֹּדֶם אוּלַי רָצִיתִי לְהִשְׁתַּפֵּר, לָצֵאת מֵהַהֶעָרוֹת ומהביקורת בַּבַּיִת, אֲבָל עַכְשָׁיו.. אֲנִי רַק רוֹצֶה לִחְיוֹת חַיִּים נוֹרְמָלִיִּים וּשְׁלֵוִים, זֶה הַכֹּל, זֹאת הָאֱמֶת רִבּוֹנוֹ שֶׁל עוֹלָם, אֲנִי כָּל כָּךְ עָיֵף מֵהָרִיבִים וּמֵהַבִּזְיוֹנוֹת... מְחִילָה ה', אֲנִי מֵבִין שֶׁצִּפִּיתָ מִמֶּנִּי לְיוֹתֵר, לִכְסֹף, לִרְצוֹת, אֲבָל זֶה אֵיפֹה שֶׁאֲנִי נִמְצָא עַכְשָׁיו.. אֲנִי עַצְמִי לֹא רוֹאֶה שֶׁיֵּשׁ לִי אֵיזוֹשֶׁהִי תַּקָּנָה, אֲנִי פָּשׁוּט בֶּן אָדָם שֶׁמֵּעִיר, כָּכָה עָשִׂיתָ אוֹתִי, וְכָּכָה מִבְּחִינָתִי זֶה כַּנִּרְאֶה הוֹלֵךְ לְהַמְשִׁיךְ, כְּבָר אֵין לִי כֹּחַ לַעֲבֹד עַל זֶה יוֹתֵר...". חַגַּי הִרְגִּישׁ שִׁחְרוּר וַהֲקָלָה כְּשֶׁאָמַר סוֹף סוֹף אֶת מַה שֶׁהוּא מַרְגִּישׁ.. "אָז אֵיךְ אֲנִי אַמְשִׁיךְ עִם בְּנֵי בֵּיתִי? הֲרֵי אֲנִי אָמוּר לְהַפְסִיק לְהָעִיר..". חַגַּי עָצַר לְכַמָּה רְגָעִים.. "אֶת הָאֱמֶת, אִשְׁתִּי מִשּׁוּם מַה עֲדַיִן מְקַבֶּלֶת אוֹתִי.. נָכוֹן, רַבְנוּ, וַאֲנִי כָּל הַזְּמַן נוֹפֵל בְּאוֹתוֹ נִסָּיוֹן, שׁוּב וְשׁוּב.. אֲבָל כָּרֶגַע, אֲנִי יָכוֹל עוֹד קְצָת לְהַמְשִׁיךְ... וְחוּץ מִזֶּה, מַתְחִיל לִהְיוֹת כָּאן קַר.. מַה אִכְפַּת לִי לַחֲזוֹר הַבַּיְתָה לַמִּטָּה הַחַמָּה... מָחָר נִרְאֶה מַה נַעֲשֶׂה.." אָמַר חַגַּי בְּהַשְׁלָמָה וְהִתְחִיל לָלֶכֶת לְכִוּוּן הָרֶכֶב שֶׁלּוֹ.

## שַׁעֲרֵי הַקְּדֻשָּׁה

חַגַּי נִכְנָס לְבֵיתוֹ בְּשֶׁקֶט, אָמַר שָׁלוֹם מָהִיר וּבְקוֹל נָמוּךְ לְאִשְׁתּוֹ, וְהִתְחִיל לְהִתְקַדֵּם לְעֵבֶר הַמִּטְבָּח כְּדֵי לְהָכִין לְעַצְמוֹ מַשֶּׁהוּ לִפְנֵי שֶׁהוּא הוֹלֵךְ לִישׁוֹן. הוּא יָדַע שֶׁצְּרִיכִים לָבוֹא הַבַּיְתָה בְּחִיּוּךְ וּבְשִׂמְחָה, אֲבָל כָּרֶגַע הַדָּבָר הַיָּחִיד שֶּׁהֶחְזִיק אוֹתוֹ זֶה שֶׁהוּא הוֹלֵךְ לִישׁוֹן בַּמִּטָּה הַחַמָּה, וְזֶה הַשָּׁלוֹם שֶׁהִצְלִיחַ לְהוֹצִיא מֵעַצְמוֹ... לְפֶתַע הוּא רָאָה אֶת בַּקְבּוּק שֶׁמֶן הַזַּיִת פָּתוּחַ בְּלִי פְּקָק עוֹמֵד עַל הַשֻּׁלְחָן.. מִיָּד הִתְחִילוּ הַמַּחֲשָׁבוֹת לְהִתְעוֹפֵף בְּמוֹחוֹ.. "לָמָּה? לָמָּה? מַה הַהִגָּיוֹן? לָמָּה לֹא לִסְגּוֹר אֶת הַבַּקְבּוּק? יֵשׁ לָהּ אֵיזֶה מַשֶּׁהוּ נֶגֶד פְּקָקִים? אֲנִי אֶלֶף פְּעָמִים אָמַרְתִּי, אִם מִשֶּׁהוּ בְּטָעוּת יַפִּיל אֶת הַבַּקְבּוּק, חֲצִי מֵהַשֶּׁמֶן זַיִת הָלַךְ.. וְזֶה שֶׁמֶן זַיִת יָקָר..". חַגַּי כְּבָר הָיָה רָגִיל לְרֶצֶף מַחְשָׁבוֹת כָּזֶה, אֲבָל אָז פִּתְאוֹם מַחֲשָׁבָה אַחֶרֶת, לֹא מֻכֶּרֶת, צְלוּלָה וְזַכָּה, טְהוֹרָה וּבְהִירָה, הוֹפִיעָה בְּמוֹחוֹ.. "נוּ, אָז אִם פַּעַם בְּשָׁנָה, מִשֶּׁהוּ בֶּאֱמֶת יַפִּיל אֶת הַשֶּׁמֶן זַיִת.. וּבֶאֱמֶת חֲצִי יִשָּׁפֵךְ.. וְנַפְסִיד 20 שֶׁקֶל, אָז בִּשְׁבִיל זֶה כְּדַאי לִי לְשַׁגֵּעַ אֶת כָּל הַבַּיִת שֶׁלִּי?". מִצַּד אֶחָד חַגַּי הָיָה מֻפְתָּע.. הוּא מֵעוֹלָם לֹא הִצְלִיחַ לָצֵאת מֵהַפְּלוֹנְטֵר שֶׁהוּא הִכְנִיס אֶת עַצְמוֹ אֵלָיו... אֲבָל מִצַּד שֵׁנִי, הַמַּחְשָׁבָה הָיְתָה כָּל כָּךְ חֲזָקָה וְעָצְמָתִית, שֶׁהוּא פָּשׁוּט הִמְשִׁיךְ וְחָלַף עַל פְּנֵי הַבַּקְבּוּק... "אֲנִי לֹא מַאֲמִינָה.." לְפֶתַע אָמְרָה אִשְׁתּוֹ. "מַה?" אָמַר חַגַּי בְּהִסּוּס, מְנַסֶּה לֹא לִפְתֹּחַ רִיב חָדָשׁ.. "אַתָּה יוֹדֵעַ, אֲנִי בַּדֶּרֶךְ כְּלָל שׁוֹכַחַת לִסְגּוֹר אֶת הַפְּקָק.. אֲבָל אַחֲרֵי הָרִיב הָאַחֲרוֹן, אֲנִי בְּכַוָּנָה שַׂמְתִּי אֶת זֶה כָּכָה עַל הַשֻּׁלְחָן.. אָמַרְתִּי לְעַצְמִי, אִם כָּל כָּךְ פָּגַעְתָּ בִּי, לָמָּה שֶׁאֲנִי לֹא אֶפְגַּע בְּךָ בַּחֲזָרָה.. אֲנִי יוֹדַעַת כַּמָּה שֶׁזֶּה מְשַׁגֵּעַ אוֹתְךָ כְּשֶׁשּׁוֹכְחִים..". חַגַּי פָּלַט גִּחוּךְ קָטָן, הוּא לֹא תֵּאֵר לְעַצְמוֹ שֶׁאִשְׁתּוֹ תַּעֲשֶׂה דָּבָר כָּזֶה.. "אֲבָל הַפַּעַם, פָּשׁוּט עָבַרְתָּ וְלֹא אָמַרְתָּ כְּלוּם.." דִּמְעָה יָצְאָה מֵעֵינָהּ שֶׁל אִשְׁתּוֹ.. "אַתָּה בֶּאֱמֶת מִשְׁתַּנֶּה, בֶּאֱמֶת אִכְפַּת לְךָ.." אָמְרָה וְהִתְחִילָה לִבְכּוֹת. דִּמְעָה יָצְאָה מֵעֵינָיו שֶׁל חַגַּי, וְהוּא חִיֵּךְ חִיּוּךְ גָּדוֹל לְאִשְׁתּוֹ.. "אַל תִּשְׁאֲלִי.. זֶה הַכֹּל בִּזְכוּת הֶחָבֵר הַמְשֻׁגָּע שֶׁלִּי... תַּגִּידִי, אַתְּ רוֹצֶה אוּלַי לָצֵאת לְאַנְשֶׁהוּ, לְשַׁנּוֹת אֲוִירָה? אֲנִי רוֹצֶה לְהַרְאוֹת לָךְ מָקוֹם מְאֹד מְיֻחָד..". "בֶּטַח, אָמַרְתִּי לְךָ מִזְּמַן שֶׁאֲנַחְנוּ צְרִיכִים לָצֵאת..." קִנְּחָה אִשְׁתּוֹ אֶת פָּנֶיהָ, "אֵיזֶה מָקוֹם?". "אַל תִּדְאֲגִי.." אָמַר חַגַּי, "זֶה מָקוֹם טוֹב, תִּסְמְכִי עָלַי..".

## מֵאֲחוֹרֵי הַקְּלָעִים

אֵיךְ בֶּאֱמֶת הַגָּן שַׁעֲשׁוּעִים כָּל כָּךְ עָזַר? חַגַּי עָבַר כָּל כָּךְ הַרְבֵּה נְפִילוֹת, וְהִרְגִּישׁ שֶׁהוּא כָּל כָּךְ לֹא מֻצְלָח בַּנִשּׂוּאִים, שֶׁהוּא כְּבָר לֹא יָכַל לִסְבֹּל אֶת זֶה. כָּל הַחִזּוּקִים שֶׁבָּעוֹלָם לֹא הוֹצִיאוּ אוֹתוֹ מִזֶּה.. אֲבָל בְּעֶצֶם בְּעִקָּר מַה שֶׁהִכְנִיס אוֹתוֹ לְעַצְבוּת וּלְיֵאוּשׁ, זֹאת הַמַּחְשָׁבָה שֶׁהוּא אָמוּר לְהַצְלִיחַ בַּנִשּׂוּאִים. בָּרֶגַע שֶׁהוּא השתטה, וְלָקַח אֶת עַצְמוֹ פָּחוֹת בִּרְצִינוּת, לַמַּחְשָׁבוֹת הַמְיָאֲשׁוֹת הָיָה כְּבָר הַרְבֵּה פָּחוֹת כֹּחַ לְיַאֵשׁ וּלְהַעֲצִיב אוֹתוֹ. גַּם מַה שֶׁהוּא אָמַר בַּהִתְבּוֹדְדוּת, שֶׁכִּבְיָכוֹל הָיָה לֹא בַּכִּוּוּן, עָזַר מְאֹד. עֶצֶם זֶה שֶׁהוּא אָמַר שֶׁהוּא כְּבָר לֹא רוֹצֶה לְהִשְׁתַּפֵּר, מַרְאֶה שֶׁהוּא רוֹצֶה לִרְצוֹת לְהִשְׁתַּפֵּר, הוּא פָּשׁוּט כְּבָר לֹא יָכוֹל מֵרוֹב יֵאוּשׁ.. וְעֶצֶם זֶה שֶׁהוּא הִתְיָאֵשׁ מֵעַצְמוֹ לְגַמְרֵי, וְרַק שָׁמַר עַל הַמַּדְרֵגָה הַבְּסִיסִית שֶׁל לְהַגִּיעַ הַבַּיְתָה בְּסוֹף הַיּוֹם וּלְנַסּוֹת לֹא לָרִיב, בִּטֵּל אֶת הַמַּחְשָׁבָה שֶׁלּוֹ שֶׁהוּא עוֹבֵד עַל עַצְמוֹ, וְהוּא יַצְלִיחַ בַּנִשּׂוּאִים, וְנָתַן מָקוֹם לַה' לָתֵת לוֹ אֶת הַיְּשׁוּעָה הַנִּסִּית שֶׁבַּסּוֹף בֶּאֱמֶת עֶזְרָה לוֹ..

## מְגַלְגְּלִים זְכוּת עַל יְדֵי זַכַּאי

אֲבָל אִם חַגַּי הָיָה אָמוּר לַעֲמֹד בַּמִּבְחָן בְּעַצְמוֹ, הֲרֵי הוּא קִבֵּל עֶזְרָה מְאֹד גְּדוֹלָה מֵחֲבֵרוֹ אָסַף, מֵאֵיפֹה הָעֶזְרָה הַזֹּאת בָּאָה? כָּל הַהִתְחַזְּקוּיוֹת שֶׁהוּא עָשָׂה בַּדֶּרֶךְ, הַהַעְתָּקוֹת, הַתְּנוּעוֹת הַקְּטַנּוֹת, הִתְקַבְּצוּ וְעָזְרוּ לוֹ בְּעֵת הַדְּחַק וְהַצָּרָה, בְּעֵת הַנִּסָּיוֹן הַגָּדוֹל.. הֵם פָּשׁוּט הִתְלַבְּשׁוּ בַּחֲבֵרוֹ אָסַף, בִּבְחִינַת מְגַלְגְּלִים זְכוּת עַל יְדֵי זַכַּאי..

## כִּסּוּפִים אוֹ עֲנָוָה?

אָז יוֹצֵא שֶׁבַּסּוֹף הַמְּנִיעוֹת גַּם עוֹזְרוֹת לָנוּ לְהַגִּיעַ לַעֲנָוָה, וְרַק כְּשֶׁאֲנַחְנוּ מַגִּיעִים לְקַטְנוּת דְּקַטְנוּת, לַיֵּאוּשׁ הֲכִי גָּדוֹל, וְעוֹבְרִים אוֹתוֹ, רַק אָז מַגִּיעִים לְשַׁעֲרֵי הַקְּדֻשָּׁה.. אָז מַה הַמַּטָּרָה, עֲנָוָה אוֹ כִּסּוּפִים? אָז ה' דּוֹאֵג לָנוּ לִשְׁנֵיהֶם.. שֶׁלֹּא חַס וְשָׁלוֹם מֵרוֹב כִּסּוּפִים נַגִּיעַ לְרִבּוּי אוֹר, גַּאֲוָה, וּשְׁבִירַת כֵּלִים, לָכֵן יֵשׁ לָנוּ אֶת הַפָּתַח, אֶת הַמְּנִיעָה.. ה' דּוֹאֵג לָנוּ שֶׁהַכִּסּוּפִים יִהְיוּ נְקִיִּים, טְהוֹרִים, שֶׁנִּרְצֶה רַק אֶת ה', וְלֹא אֶת הַצְלָחַתֵנוּ בַּעֲבוֹדַת ה'...

## נְקוּדַת הַקָּמָץ

אָז מַה כָּל כָּךְ מְיֻחָד בַּקָמַץ? שֶׁהוּא מֵאַחֵד אֶת הַ'נְּקֻדָּה' וְהַ'פַּתַח', שֶׁהַמְּנִיעָה וְהַכִּסּוּפִים נִהְיִּים אֶחָד, שֶׁהַדִּין וְהָרַחֲמִים נִהְיִּים אֶחָד.. שֶׁאֲנַחְנוּ מְבִינִים כַּמָּה אַהֲבָה וְרַחֲמִים יֵשׁ בַּמְנִיעוֹת וּבַדִּינִים... רַק בָּעוֹלָם הַבָּא, נִזְכֶּה לְהַשִּׂיג אֶת אוֹר נְקוּדַת הַקָּמָץ, אוֹר הַכֶּתֶר, בִּשְׁלֵמוּת, אַךְ בָּעוֹלָם הַזֶּה, כַּמָּה שֶׁאֲנַחְנוּ מִתְחַזְּקִים בֶּאֱמוּנָה וּבְשִׂמְחָה בְּכָל מַצָּב עַל יְדֵי הָעֵצוֹת שֶׁל רַבֵּינוּ, אֲנַחְנוּ זוֹכִים לְהַשִּׂיג בְּחִינָה שֶׁל אוֹתוֹ הָאוֹר.. שֶׁנִּזְכֶּה כֻּלָּנוּ לְהַגִּיעַ לְשַׁעֲרֵי הַקְּדֻשָּׁה, כָּל אֶחָד וְעִנְיָנוֹ הוּא.. שַׁבָּת שָׁלוֹם!
