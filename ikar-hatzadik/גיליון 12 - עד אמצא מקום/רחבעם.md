# עצות מקולקלות

## ליקו"ה או"ח הל' תשעה באב ה"ד אות יא

כִּי הַגָּבִיעַ הוּא אֲשֶׁר יִשְׁתֶּה בּוֹ וְהוּא נַחֵשׁ יְנַחֵשׁ בְּחִינַת עֵצוֹת, כִּי כָּל הַמְנַחֲשִׁים וְהַקּוֹסְמִים הֵם בִּשְׁבִיל עֵצוֹת שֶׁמִּי שֶׁצָּרִיךְ לְאֵיזֶה עֵצָה לֵידַע אֵיךְ לְהִתְנַהֵג הָיָה הוֹלֵךְ לְהַמְנַחֲשִׁים וְהַקּוֹסְמִים, רַק שֶׁאֲנַחְנוּ בְּנֵי יִשְרָאֵל מֻזְהָרִים מִלְּקַבֵּל עֵצוֹת מֵהַמְנַחֲשִׁים וְהַקּוֹסְמִים, רַק לִהְיוֹת תָּמִים עִם ה' וּלְקַבֵּל כָּל הָעֵצוֹת מֵחֲכָמִים אֲמִתִּיִים. נִמְצָא, שֶׁנָּחָשׁ יְנַחֵשׁ הוּא בְּחִינַת עֵצוֹת, רַק שֶׁהִטְעָה אוֹתָם וְדִבֵּר עִמָּהֶם כְּאִלּוּ יוֹסֵף הוּא מֵאַנְשֵׁי מִצְרַיִם שֶׁעֲצָתָם עַל יְדֵי נִחוּשִׁים. וְזֶה שֶׁאָמְרוּ, כִּי הַגָּבִיעַ הוּא אֲשֶׁר יִשְׁתֶּה אֲדוֹנִי בּוֹ, כִּי שְׁתִיָּה בְּחִינַת שְׁלֵמוּת הָעֵצָה, בְּחִינַת וְאָכַלְתָּ וְשָבָעְתָּ. וְדָרְשׁוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה וְשָבָעְתָּ זֶה שְׁתִיָּה. כִּי עִקַּר הַשְּבִיעָה עַל יְדֵי הַשְּׁתִיָּה שֶׁהוּא מְעַכֶּלֶת הַמָּזוֹן וּמַשְבִּיעַ אֶת הַגּוּף, שֶׁזֶּהוּ הַהֵפֶךְ מִפְּגַם וְקִלְקוּל הָאֲכִילָה שֶׁעוֹלִים חַס וְשָׁלוֹם, עֲשָׁנִים סְרוּחִים וּמְבַלְבְּלִים אֶת הַמֹּחַ שֶׁמִּשָּׁם חֲלֻקַּת הָעֵצָה כַּנַּ"ל, שֶׁזֶּהוּ מֵחֲמַת שֶׁאֵין הַמַּאֲכָל מִתְעַכֵּל כָּרָאוּי וְאֵינוֹ מַשְבִּיעַ אֶת הַגּוּף כָּרָאוּי, אֲבָל שְׁתִיָּה הוּא בְּחִינַת עִכּוּל הַמָּזוֹן שֶׁלֹּא יַעֲלוּ הָעֲשָׁנִים סְרוּחִים אֶל הַמֹּחַ, רַק הַמַּאֲכָל יַשְבִּיעַ אֶת הַגּוּף כָּרָאוּי בְּחִינַת וְשָבָעְתָּ זוֹ שְׁתִיָּה. וְזֶהוּ בְּחִינַת כּוֹס שֶׁל בִּרְכַּת הַמָּזוֹן שֶׁהוּא בְּחִינַת כּוֹס יְשׁוּעוֹת, כִּי יְשׁוּעוֹת בְּחִינַת עֵצוֹת כַּיָּדוּעַ, כִּי עִקַּר הַיְשׁוּעָה הוּא שְׁלֵמוּת הָעֵצָה כְּשֶׁהַשֵּׁם יִתְבָּרַךְ מֵאִיר עֵינֵי הָאָדָם וְנוֹתֵן לוֹ עֵצָה אֵיךְ לְהִנָּצֵל מִמַּה שֶּׁצָּרִיךְ לְהִנָּצֵל בְּגוּף וְנֶפֶשׁ וּמָמוֹן. וּכְמוֹ שֶׁכָּתוּב, וּתְשׁוּעָה בְּרֹב יוֹעֵץ. וְכֵן בְּסִפְרֵי קַבָּלָה נִקְרָאִים הָעֵצוֹת יְשׁוּעוֹת כַּיָּדוּעַ, כִּי בִּרְכַּת הַמָּזוֹן הוּא בִּשְׁבִיל לְהַמְשִׁיךְ תִּקּוּן הָאֲכִילָה שֶׁלֹּא יַעֲלוּ חַס וְשָׁלוֹם, עַל יְדֵי הָאֲכִילָה עֲשָׁנִים סְרוּחִים לְבַלְבֵּל אֶת הַמֹּחַ שֶׁהוּא בִּלְבּוּל הָעֵצָה, רַק שֶׁיִּזְכֶּה לְעַכֵּל אֶת הַמָּזוֹן כָּרָאוּי. וְעַל כֵּן עִקַּר בִּרְכַּת הַמָּזוֹן הוּא עַל אֶרֶץ יִשְרָאֵל, כִּי שָׁם עִקַּר תִּקּוּן הָעֵצָה כַּנַּ"ל וְעַל כֵּן בִּרְכַּת הַמָּזוֹן עַל הַכּוֹס בְּחִינַת כּוֹס יְשׁוּעוֹת אֶשָּא, שֶׁזֶּהוּ בְּחִינַת תִּקּוּן הָעֵצָה כַּנַּ"ל, וְעַל כֵּן כּוֹס שֶׁל בִּרְכַּת הַמָּזוֹן הוּא בִּבְחִינַת מַלְכוּת דָּוִד כַּיָּדוּעַ, כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה שֶׁדָּוִד דַּיְקָא זָכָה לָזֶה בְּחִינַת כּוֹס יְשׁוּעוֹת אֶשָּא, בְּחִינַת לִי נָאֶה לְבָרֵךְ וְכוּ'. כִּי מַלְכוּת בֵּית דָּוִד זֶה בְּחִינַת תִּקּוּן הָעֵצָה הַשְּׁלֵמָה, שֶׁזֶּהוּ בְּחִינַת מַלְכוּת מָשִׁיחַ בֶּן דָּוִד שֶׁנִּקְרָא פֶּלֶא יוֹעֵץ וְכוּ' וְכַנַּ"ל. כִּי עִקַּר תִּקּוּן הָעֵצָה בִּשְׁלֵמוּת שֶׁלֹּא תִּתְחַלֵּק לִשְׁתַּיִם חַס וְשָׁלוֹם, זֶהוּ בְּחִינַת הִתְאַחֲדוּת הַמְּלוּכָה שֶׁל דָּוִד מָשִׁיחַ שֶׁלֹּא תִּתְחַלֵּק עוֹד מַלְכוּתוֹ לִשְׁתַּיִם, בְּחִינַת וְלֹא יֵחָצוּ לִשְׁתֵּי מַמְלָכוֹת שֶׁקּוֹרִין כָּל זֶה בְּהַפְטָרַת וַיִּגַּשׁ. וּכְלַל הַדָּבָר שֶׁכָּל הַמַּחֲלֹקֶת שֶׁהָיָה בֵּין יוֹסֵף וְהַשְּׁבָטִים הוּא בְּחִינַת חֲלֻקַּת מַלְכוּת בֵּית דָּוִד שֶׁנֶּחֱלַק עַל יְדֵי יָרָבְעָם שֶׁבָּא עַל יְדֵי עֵצוֹת מְקֻלְקָלוֹת שֶׁל רְחַבְעָם שֶׁשָּׁמַע לַעֲצַת הַיְלָדִים, וּמִכָּל שֶׁכֵּן יָרָבְעָם שֶׁנָּפַל לְעֵצוֹת רָעוֹת מְאֹד, כְּמוֹ שֶׁכָּתוּב, וַיִּוָּעֵץ הַמֶּלֶךְ וְכוּ'. וְכָל זֶה נִשְׁתַּלְשֵׁל מֵהַמַּחֲלֹקֶת שֶׁל הַשְּׁבָטִים עַל יוֹסֵף. וְעַל כֵּן פָּרָשַׁת וַיִּגַּשׁ שֶׁמְּדַבֶּרֶת מֵהִתְחַבְּרוּת הַשְּׁבָטִים עִם יוֹסֵף מַפְטִירִין אָז, וְאַתָּה קַח לְךָ עֵץ אֶחָד וְכוּ' שֶׁמְּדַבֵּר מֵהִתְחַבְּרוּת מַלְכוּת בֵּית יִשְרָאֵל וְרוֹעֶה אֶחָד יִהְיֶה לְכֻלָּם, וְעַבְדִי דָּוִד מֶלֶךְ עֲלֵיהֶם וְכוּ'. וְכָל זֶה הוּא בְּחִינַת הִתְאַחֲדוּת הָעֵצָה כַּנַּ"ל, בְּחִינַת וְלֹא יֵחָצוּ לִשְׁתֵּי מַמְלָכוֹת עוֹד וְכַנַּ"ל (וְכַמְבֹאָר בְּמָקוֹם אַחֵר מִזֶּה). וְעַל כֵּן כָּל עִנְיַן הַתּוֹרָה הַנַּ"ל שֶׁמְּדַבֶּרֶת מִתִּקּוּן הָעֵצָה מְרֻמָּז בְּפָרָשָׁה זֹאת וְכַנַּ"ל:

## ליקו"ה או"ח הל' הושענא רבה ה"ב אות ד'

כִּי עִקַּר בִּנְיַן הַמַּלְכוּת הוּא עַל יְדֵי בְּחִינַת עֵצוֹת, שֶׁהֵם בְּחִינַת נֶצַח הוֹד, שֶׁמִּשָּׁם עִקַּר בִּנְיַן הַמַּלְכוּת כַּיָּדוּעַ, וְעַל כֵּן כָּל מֶלֶךְ יֵשׁ לוֹ יוֹעֲצִים, כִּי עִקַּר תִּקּוּן הַמַּלְכוּת הוּא עַל יְדֵי עֵצוֹת, וְעַל כֵּן נִקְרָאִים הָעֵצוֹת בְּשֵׁם מַלְכוּת, כְּמוֹ שֶׁכָּתוּב מַלְכִּי יִשְׁפַּר עֲלָךְ, שֶׁזֶּה בְּחִינוֹת וְלֹא יֵחָצוּ לִשְׁתֵּי מַמְלָכוֹת, הַיְנוּ עֵצוֹת וְכוּ' (כַּנַּ"ל בַּמַּאֲמָר הַנַּ"ל), כִּי עִקַּר בְּחִינַת הַמַּלְכוּת עַל יְדֵי עֵצוֹת כַּנַּ"ל, וְעַל כֵּן חֲלֻקַּת מַלְכוּת בֵּית דָּוִד הָיָה עַל יְדֵי שֶׁנִּפְגַּם הָעֵצוֹת וְלֹא זָכָה רְחָבְעָם לָשִׁית עֵצוֹת בְּנַפְשׁוֹ, וַיַּעֲזֹב עֲצַת הַזְּקֵנִים וַיָּשֶׁב לָעָם קָשׁוֹת כַּעֲצַת הַיְלָדִים אֲשֶׁר גָּדְלוּ אִתּוֹ, וְעַל יְדֵי זֶה נֶחֶלְקָה מַלְכוּת בֵּית דָּוִד, שֶׁעַל יְדֵי זֶה נֶאֱמַר לֶעָתִיד וְלֹא יֵחָצוּ לִשְׁתֵּי מַמְלָכוֹת עוֹד, שֶׁפֵּרֵשׁ רַבֵּנוּ בַּמַּאֲמָר הַנַּ"ל שֶׁמְּרַמֵּז עַל חֲלֻקַּת הָעֵצָה שֶׁנֶּחֱצָה לִשְׁתַּיִם, שֶׁאֵינוֹ יוֹדֵעַ אֵיךְ לַעֲשוֹת, כִּי הַכֹּל אֶחָד, כִּי פְּגַם הָעֵצָה הוּא פְּגַם הַמַּלְכוּת, כִּי עַל יְדֵי חֲלֻקַּת הָעֵצָה לִשְׁתַּיִם, עַל יְדֵי זֶה נֶחֱצָה מַלְכוּת בֵּית דָּוִד כַּנַּ"ל, וּלֶעָתִיד שֶׁנִּזְכֶּה לְמַיִם טְהוֹרִים וְנִזְכֶּה לְעֵצָה שְׁלֵמָה, אָז יִתְקַיֵּם וְלֹא יֵחָצוּ לִשְׁתֵּי מַמְלָכוֹת עוֹד, דְּהַיְנוּ כִּפְשׁוּטוֹ שֶׁתִּהְיֶה מַלְכוּת אַחַת מַלְכוּת בֵּית דָּוִד וְגַם תִּהְיֶה הָעֵצָה שְׁלֵמָה, כִּי הַכֹּל אֶחָד כַּנַּ"ל, וְעַל כֵּן הָיָה דָּוִד הַמֶּלֶךְ עָלָיו הַשָּׁלוֹם מְבַקֵּשׁ מְאֹד לִזְכּוֹת לְעֵצָה שְׁלֵמָה, כְּמוֹ שֶׁכָּתוּב (תְּהִלִּים יג) עַד אָנָה אָשִׁית עֵצוֹת בְּנַפְשִׁי וְכוּ', וּכְשֶׁזּוֹכֶה לְעֵצָה שְׁלֵמָה נֶאֱמָר אֲבָרֵךְ ה' אֲשֶׁר יְעָצַנִי אַף לֵילוֹת וְכוּ' (שָׁם טז), כִּי עִקַּר מַלְכוּת תְּלוּיָה בָּזֶה בִּבְחִינַת עֵצוֹת שְׁלֵמוֹת כַּנַּ"ל, וְעַל כֵּן כְּשֶׁבָּרַח דָּוִד מִפְּנֵי אַבְשָׁלוֹם בְּנוֹ בִּקֵּשׁ דָּוִד מֵהַשֵּׁם יִתְבָּרַךְ, סַכֶּל נָא עֲצַת אֲחִיתֹפֶל ה', וְכֵן הָיָה, כִּי עִקַּר נְפִילַת אַבְשָׁלוֹם הָיָה עַל יְדֵי שֶׁהַשֵּׁם יִתְבָּרַךְ בִּלְבֵּל עֲצָתוֹ, שֶׁנֶּחֶלְקָה עֲצָתוֹ לִשְׁנַיִם וְלֹא יָדַע אֵיךְ לַעֲשוֹת, עַד שֶׁעָשָה כַּעֲצַת חוּשַׁי הָאַרְכִּי וְלֹא עָשָה כַּעֲצַת אֲחִיתֹפֶל שֶׁהָיְתָה טוֹבָה לְפָנָיו, וְעַל יְדֵי זֶה הָיְתָה מַפָּלָתוֹ, כְּמוֹ שֶׁכָּתוּב וַה' צִוָּה לְהָפֵר עֲצַת אֲחִיתֹפֶל הַטּוֹבָה לְבַעֲבוּר וְכוּ', כִּי עִקַּר הַמַּלְכוּת הוּא עַל יְדֵי עֵצוֹת כַּנַּ"ל, וְעַל כֵּן קֹדֶם שְׁמִינִי עֲצֶרֶת, שֶׁאָז הוּא גְּמַר תִּקּוּן הַמַּלְכוּת כַּיָּדוּעַ, עַל כֵּן אָנוּ לוֹקְחִין מִקֹּדֶם עַרְבֵי נַחַל שֶׁהֵם בְּחִינַת תִּקּוּן הָעֵצוֹת שֶׁנִּתְתַּקְּנִין עַל יְדֵי אֲפִיקֵי מַיִם, עַל יְדֵי בְּחִינַת מַיִם טְהוֹרִים שֶׁנִּתְגַּלִין אָז, בְּחִינַת בְּהַהוּא יוֹמָא וַיָּשָׁב יִצְחָק וַיַּחְפֹּר וְכוּ', וְעַל יְדֵי זֶה אָנוּ זוֹכִין בִּשְׁמִינִי עֲצֶרֶת לְהַשְׁלִים בְּחִינַת תִּקּוּן הַמַּלְכוּת, כִּי עִקַּר תִּקּוּן הַמַּלְכוּת, דְּהַיְנוּ שֶׁיִּתְגַּלֶּה מַלְכוּתוֹ יִתְבָּרַךְ עָלֵינוּ, הוּא עַל יְדֵי הַתּוֹרָה שֶׁהוּא תַּרְיַ"ג עֲטִין דְּאוֹרָיְתָא, כְּמוֹ שֶׁכָּתוּב לִי עֵצָה וְתוּשִׁיָּה:

## ליקו"ה יו"ד הל' הכשר כלים ה"ב

ז) וְעַל כֵּן הַכֵּלִים שֶׁל שִׁבְעָה מִינֵי מַתָּכוֹת צְרִיכִין תִּקּוּן בְּיוֹתֵר לְטָבְלָם וּלְטַהֲרָם מֵאֲחִיזַת הַשִּׁבְעָה עֲמָמִין, כִּי עִקַּר אֲחִיזַת הָעֲמָמִין הוּא מִבְּחִינַת שִׁבְעָה מִינֵי מַתָּכוֹת, שֶׁזֶּהוּ בְּחִינַת הַצֶּלֶם שֶׁהֶעֱמִיד נְבוּכַדְנֶאצַּר "רֵאשֵׁהּ דִּי דְּהַב טַב חֲדוֹהִי וּדְרָעוֹהִי דִּי כְּסַף וְכוּ'". שֶׁהוּא מְרַמֵּז עַל כְּלָלִיּוּת הָאֻמּוֹת עַל כָּל הָאַרְבַּע מַלְכֻיּוֹת, כַּמְבֹאָר שָׁם בַּפָּסוּק (דָּנִיֵּאל ב), כִּי הַשִּׁבְעָה עֲמָמִין כְּלוּלִין בְּאַרְבַּע מַלְכֻיּוֹת כְּמוֹ שֶׁבַּקְּדֻשָּׁה שִׁבְעַת יְמֵי הַבִּנְיָן כְּלוּלִים בְּאַרְבַּע אוֹתִיּוֹת הַשֵּׁם כַּיָּדוּעַ, כְּמוֹ כֵן אֶת זֶה לְעֻמַּת זֶה, שִׁבְעָה עֲמָמִין כְּלוּלִים בָּאַרְבַּע מַלְכֻיּוֹת. וְעִקַּר אֲחִיזָתָם בִּבְחִינַת שִׁבְעָה מִינֵי מַתָּכוֹת כְּשֶׁאֵין זוֹכִין לְקַדֵּשׁ הַכֵּלִים כָּרָאוּי, כִּי עִקַּר אֲחִיזָתָם הוּא רַק בִּבְחִינַת הַכֵּלִים כְּשֶׁאֵינָן כָּרָאוּי, כִּי מֵעֹצֶם הָאוֹר אֵין יְכוֹלִין לִנֹּק רַק מִבְּחִינַת הַכֵּלִים שֶׁהוּא בְּחִינַת צִמְצוּם הָאוֹר כַּיָּדוּעַ. וְזֶה שֶׁנֶּאֱמַר בְּעֵשָֹו, "שָֹא נָא כֵלֶיךָ תֶּלְיְךָ וְכוּ'", שֶׁמְּרַמֵּז עַל גָּלוּת אַרְבַּע מַלְכֻיּוֹת, כְּמוֹ שֶׁנֶּאֱמַר בַּמִּדְרָשׁ (בְּרֵאשִׁית רַבָּה, פֶּרֶק כ). וְזֶהוּ 'כֵּלֶיךָ' דַּיְקָא, כִּי עִקַּר הִתְגַּבְּרוּתוֹ הוּא עַל יְדֵי הַכֵּלִים, שֶׁהַכֵּלִים נוֹפְלִים אֶצְלוֹ, חַס וְשָׁלוֹם, בְּחִינַת חֻרְבַּן בֵּית הַמִּקְדָּשׁ, שֶׁעִקַּר הַחֻרְבָּן הָיָה שֶׁלָּקְחוּ אֶת הַכֵּלִים, כְּמוֹ שֶׁנֶּאֱמַר, "וְאֶת הַכֵּלִים הֵבִיא בֵּית אוֹצַר אֱלֹהָיו וְכוּ'". וְעַל כֵּן הַכֵּלִים שֶׁיּוֹצְאִין מֵרְשׁוּתָם אִי אֶפְשָׁר לָנוּ לְהִשְׁתַּמֵּשׁ בָּהֶם כִּי אִם עַל יְדֵי טְבִילָה וְעִקַּר הַטְּבִילָה הִיא לִכְלֵי מַתָּכוֹת שֶׁמִּשָּׁם אֲחִיזָתָם וְכוּ' כַּנַּ"ל. וּכְשֶׁהֵם נֶאֱחָזִין בָּהֶם אֲזַי אֵין עֵצָה שְׁלֵמָה, כִּי עִקַּר הָעֵצָה מִשָּׁם מִבְּחִינַת שִׁבְעָה מִינֵי מַתָּכוֹת שֶׁקְּדֻשָּׁתָם מִבְּחִינַת אֶרֶץ יִשְֹרָאֵל, מִבְּחִינַת סוֹד הָעִבּוּר, שֶׁהוּא שִׁבְעָה כּוֹכְבֵי לֶכֶת, שֶׁמִּשָּׁם זוֹכִין לְעֵצָה שְׁלֵמָה, כַּמְבֹאָר בַּמַּאֲמָר הַנַּ"ל. שֶׁעַל יְדֵי סוֹד הָעִבּוּר זוֹכִין לְעֵצָה שְׁלֵמָה וְכוּ'. כַּמְבֹאָר שָׁם בְּסֵדֶר הַמַּאֲמָר, עַיֵּן שָׁם, וְכַמְבֹאָר בְּהַמַּעֲשֶֹה הַנַּ"ל, שֶׁעַל יְדֵי זֶה שִׁבְעָה מִינֵי מַתָּכוֹת שֶׁמְּאִירִין בָּהֶם שִׁבְעָה כוֹכְבֵי לֶכֶת, עַל יְדֵי זֶה יוֹדְעִין עֵצוֹת וְכוּ' עַיֵּן שָׁם, כִּי אֶת זֶה לְעֻמַּת זֶה כְּמוֹ שֶׁזֶּה הַמֶּלֶךְ עָשָֹה זֹאת בִּשְׁבִילוֹ, כְּמוֹ כֵן בְּשָׁרְשׁוֹ בִּקְדֻשָּׁה נִמְשָׁכִין עֵצוֹת אֲמִתִּיּוֹת מִבְּחִינַת שִׁבְעָה מִינֵי מַתָּכוֹת עַל יְדֵי שִׁבְעַת כּוֹכְבֵי לֶכֶת שֶׁהֵם בְּחִינַת סוֹד הָעִבּוּר וְעַל כֵּן כְּשֶׁנִּפְגָּמִין, חַס וְשָׁלוֹם, הַשִּׁבְעָה מִינֵי מַתָּכוֹת וְנִמְשָׁכִין אֲלֵיהֶם, שֶׁזֶּה בְּחִינַת כְּלֵי מַתָּכוֹת שֶׁל עַכּוּ"ם, דְּהַיְנוּ הַכֵּלִים שֶׁנָּפְלוּ אֲלֵיהֶם. מִשָּׁם מִתְגַּבְּרִין, חַס וְשָׁלוֹם, עֵצוֹת נִבְעָרוֹת שֶׁלָּהֶם בְּחִינַת עַל עַמְּךָ יַעֲרִימוּ סוֹד וְיִתְיַעֲצוּ עַל צְפוּנֶיךָ (תְּהִלִּים פד), אֲשֶׁר עַל זֶה נֶאֱמַר (שָׁם לג), "ה' הֵפִיר עֲצַת גּוֹיִם וְכוּ'" בְּחִינַת עֻצוּ עֵצָה וְתֻפָר וְכוּ'. עַל כֵּן צְרִיכִין לְטַהֵר הַכֵּלִים שֶׁיּוֹצְאִין מֵרְשׁוּתָם, וְעִקַּר כְּלֵי מַתָּכוֹת כַּנַּ"ל, שֶׁצְּרִיכִין לְטַהֲרָם עַל יְדֵי מֵי מִקְוֶה שֶׁהוּא בְּחִינַת מַיִם טְהוֹרִים הַנֶּאֱמָרִים בַּמַּאֲמָר הַנַּ"ל, שֶׁעַל יְדֵי זֶה זוֹכִין לְעֵצָה שְׁלֵמָה, כְּמוֹ שֶׁנֶּאֱמַר שָׁם. בְּחִינַת וְלֹא יִטַּמְּאוּ עוֹד בְּגִלּוּלֵיהֶם וְלֹא יֵחָצוּ עוֹד לִשְׁתֵּי מַמְלָכוֹת. כְּלָל וְכַנַּ"ל. כִּי מִי שֶׁהוּא חָכָם אֲמִתִּי וְרוֹצֶה לְיַשֵּׁב דַּעְתּוֹ בֶּאֱמֶת וּלְהִסְתַּכֵּל עַל עַצְמוֹ וְעַל תַּכְלִיתוֹ בֶּאֱמֶת וְאֵינוֹ רוֹצֶה לְהַטְעוֹת אֶת עַצְמוֹ, בְּוַדַּאי רָאוּי לוֹ שֶׁיַּחֲשֹׁב רַק תַּחְבּוּלוֹת וְעֵצוֹת אֵיךְ לְהִנָּצֵל מִן הַשְּׁאוֹל תַּחְתִּיּוֹת וּמִתַּחְתָּיו, כִּי זֶה רוֹאֶה כָּל אָדָם, מִכָּל שֶׁכֵּן מִי שֶׁהוּא חָכָם קְצָת גֹּדֶל גְּנוּת זֶה הָעוֹלָם שֶׁהוּא מָלֵא יְגוֹנוֹת וְכָל יָמָיו כַּעַס וּמַכְאוֹבִים וְאֵין שׁוּם אָדָם בָּעוֹלָם שֶׁיִּהְיֶה לוֹ נַחַת וְטוּב בְּזֶה הָעוֹלָם הַמַּר הַמָּלֵא יִסּוּרִים מִתְּחִלָּה וְעַד סוֹף שֶׁאִי אֶפְשָׁר ד) כִּי עַכְשָׁו בַּעֲוֹנוֹתֵינוּ הָרַבִּים בְּעֹמֶק הַגָּלוּת הַזֶּה שֶׁנִּתְאָרֵךְ מְאֹד יָרַדְנוּ מְאֹד בַּעֲוֹנוֹתֵינוּ הָרַבִּים בִּבְחִינַת וַתֵּרֶד פְּלָאִים עַד אֲשֶׁר אָבְדָה עֵצָה מִבָּנִים. וּכְמוֹ שֶׁצָּעַק דָּוִד עַל זֶה, "עַד אָנָה אָשִׁית עֵצוֹת בְּנַפְשִׁי וְכוּ'. וּכְתִיב, "אָבַד מָנוֹס מִמֶּנִּי וְכוּ'". כִּי הַסִּטְרָא אָחֳרָא, שֶׁמִּשָּׁם הַיֵּצֶר הָרָע, מִשְׁתַּטֵּחַ וּמִתְפַּשֵּׁט עַל כָּל אֶחָד מְאֹד מְאֹד לְאָרְכּוֹ וּלְרָחְבּוֹ. וּבְכָל מַה שֶּׁרוֹצֶה לְהִתְקָרֵב לַה' יִתְבָּרַךְ מִתְפַּשֵּׁט כְּנֶגְדּוֹ מְאֹד עַד שֶׁנִּדְמֶה שֶׁכִּמְעַט אֶפֶס תִּקְוָה, חַס וְשָׁלוֹם. וּכְמוֹ שֶׁכָּתוּב , "רֶשֶׁת הֵכִינוּ לִפְעָמַי וְכוּ'". וּכְמוֹ שֶׁכָּתוּב, "בְּאֹרַח זוּ אֲהַלֵּךְ טָמְנוּ פַח לִי". שֶׁבְּכָל דֶּרֶךְ וְאֹרַח וְנָתִיב שֶׁרוֹצִים לֵילֵךְ לְהִתְקָרֵב לַה' יִתְבָּרַךְ טוֹמְנִים שָׁם פַּחִים לְלָכְדוֹ, חַס וְשָׁלוֹם. וּמֵחֲמַת זֶה עֲצָתוֹ חֲלוּקָה תָּמִיד שֶׁאֵינוֹ יוֹדֵעַ לָתֵת עֵצָה לְנַפְשׁוֹ אֵיךְ לְמַלֵּט נַפְשׁוֹ מִבְּאֵר שַׁחַת מִטִּיט הַיָּוֵן כָּזֶה. וּכְמוֹ שֶׁצָּעַק דָּוִד, "טָבַעְתִּי בִּיוֵן מְצוּלָה וְאֵין מָעֳמָד וְכוּ'". וְהַתִּקּוּן לְכָל זֶה שֶׁיָּשׁוּב בִּתְשׁוּבָה עַל פְּגַם אֱמוּנַת חֲכָמִים עַד שֶׁיַּחֲזֹר וְיֻחְשְׁבוּ בְּעֵינָיו כָּל סִפְרֵיהֶם הַקְּדוֹשִׁים, כִּי הֵם מְטַהֲרִים מִכָּל הַטֻּמְאוֹת וּמִכָּל הַטִּנֹּפֶת שֶׁמִּשָּׁם נִמְשָׁךְ חֲלֻקַּת הָעֵצָה:

כִּי עִקַּר הַתִּקּוּן עַל יְדֵי הַחֲכָמִים הַגְּדוֹלִים שֶׁיּוֹדְעִים סוֹד הָעִבּוּר, שֶׁיּוֹדְעִים לְעַבֵּר שָׁנִים וְלִקְבֹּעַ חֳדָשִׁים וִיכוֹלִים לַעֲשוֹת שְׁנֵי יָמִים רֹאשׁ חֹדֶשׁ וּשְׁנֵי יָמִים יוֹם טוֹב, אַף עַל פִּי שֶׁלְּפִי דַּעְתֵּנוּ בְּאֶחָד מֵהֶם בְּוַדַּאי אֵינוֹ יוֹם טוֹב וְלֹא רֹאשׁ חֹדֶשׁ, אַף עַל פִּי כֵן מָסַר לָהֶם ה' יִתְבָּרַךְ כֹּחַ כָּזֶה שֶׁהֵם יְכוֹלִים לִקְבֹּעַ רֹאשׁ חֹדֶשׁ וְיוֹם טוֹב כִּרְצוֹנָם כַּנַּ"ל, שֶׁבָּזֶה תָּלוּי כָּל קְדֻשַּׁת יִשְרָאֵל, כִּי כָּל הַתּוֹרָה תְּלוּיָה בִּיצִיאַת מִצְרַיִם שֶׁעַל זֶה נִקְבְּעוּ הַיָּמִים טוֹבִים הַתְּלוּיִים בְּרֹאשׁ חֹדֶשׁ שֶׁהֵם זֵכֶר לִיצִיאַת מִצְרַיִם, וּבִפְרָט מְבֹאָר בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה בְּסִימָן יוּד) שֶׁכָּל שֹׁרֶשׁ הַתְּשׁוּבָה הוּא רֹאשׁ חֹדֶשׁ וְכוּ', עַיֵּן שָׁם. נִמְצָא, שֶׁכָּל הַתְּשׁוּבָה וְכָל הַתּוֹרָה תְּלוּיָה בְּרֹאשׁ חֹדֶשׁ וְיוֹם טוֹב וְהַכֹּל תָּלוּי בְּהַצַּדִּיקֵי אֱמֶת שֶׁיּוֹדְעִים סוֹד הָעִבּוּר לִקְבֹּעַ רֹאשׁ חֹדֶשׁ כִּרְצוֹנָם עַד שֶׁבַּגָּלוּת הַזֶּה עוֹשִים לָנוּ שְׁנֵי יָמִים רֹאשׁ חֹדֶשׁ. וּמִכָּל זֶה רוֹאִים גְּדֻלַּת הַצַּדִּיקִים שֶׁכָּל הַתּוֹרָה תְּלוּיָה בָּהֶם, וְעַל כֵּן כְּשֶׁמַּאֲמִינִים בָּהֶם וּמִתְקָרְבִים אֲלֵיהֶם יְכוֹלִים לָשׁוּב וְלָצֵאת מִכָּל הַדְּרָכִים הַנְּבוֹכִים שֶׁנִּתְעֶה בָּהֶם. וּמִכָּל מִינֵי חֲלֻקַּת הָעֵצָה שֶׁבָּעוֹלָם, כִּי אֵיךְ שֶׁיַּעֲשֶה וְאֵיךְ שֶׁיִּתְנַהֵג בַּעֲבוֹדָתוֹ מִכֻּלָּם יוּכַל לָשׁוּב לַה' יִתְבָּרַךְ אֲפִלּוּ אִם יַעֲבֹר עָלָיו מָה בִּבְחִינַת, אִם אֶסַּק שָׁמַיִם שָׁם אָתָּה וְאַצִּיעָה שְׁאוֹל הִנֶּךָּ אֶשָּא כַּנְפֵי שַׁחַר אֶשְׁכְּנָה בְּאַחֲרִית יָם גַּם שָׁם יָדְךָ תַּנְחֵנִי וְתֹאחֲזֵנִי יְמִינְךָ וְכוּ' גַּם חֹשֶׁךְ לֹא יַחֲשִׁיךְ מִמֶּךָּ וְלַיְלָה כַיּוֹם יָאִיר וְכוּ'. כִּי הַצַּדִּיק הָאֱמֶת יָכוֹל לַהֲפֹךְ הַכֹּל לְטוֹבָה יִהְיֶה אֵיךְ שֶׁיִּהְיֶה וּבִלְבַד שֶׁיֹּאחֵז אֶת עַצְמוֹ בּוֹ בְּכָל כֹּחוֹ תָּמִיד כָּל יְמֵי חַיָּיו בְּכָל מַה שֶּׁיַּעֲבֹר עָלָיו, וּכְמוֹ שֶׁרוֹאִין בְּעִנְיַן קִדּוּשׁ הַחֹדֶשׁ שֶׁהוּא כְּלַל הַתּוֹרָה, שֶׁהַכֹּל תָּלוּי בָּהֶם כַּנַּ"ל:

# הטכנולוגיה באה מחידושים אמיתיים ועצות אמיתיות

## ליקו"ה חו"מ הל' פיקדון ה"ה אות טז

(טז) וְזֶה בְּחִינַת כָּל הָאֻמָּנִין שׁוֹמְרֵי שָֹכָר הֵן, כִּי צְרִיכִין לְהַאֲמִין שֶׁכָּל חָכְמַת הָאֻמָּנוּת שֶׁל כָּל הָאֻמָּנִין שֶׁבָּעוֹלָם כֻּלָּם נִמְשָׁכִין בְּשָׁרְשָׁן מֵחָכְמַת הַתּוֹרָה, כִּי הַתּוֹרָה הָיְתָה כְּלִי אֻמָּנוּתוֹ שֶׁל יוֹצֵר בְּרֵאשִׁית, כְּמוֹ שֶׁכָּתוּב, וָאֶהְיֶה אֶצְלוֹ אָמוֹן. אַל תִּקְרֵי אָמוֹן אֶלָּא אֻמָּן, כַּמְבֹאָר בְּדִבְרֵי רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה בְּכַמָּה מְקוֹמוֹת וּבִפְרָט בְּמִדְרָשׁ רַבָּה פָּרָשַׁת בְּרֵאשִׁית. אֲבָל ה' יִתְבָּרַךְ בָּרָא אֶת הָעוֹלָם בְּכַוָּנָה שֶׁיִּהְיֶה הָעוֹלָם דַּיְקָא מְחֻסָּר תִּקּוּן וְשֶׁהָאָדָם דַּיְקָא יְתַקֵּן הַכֹּל, כְּמוֹ שֶׁכָּתוּב אֲשֶׁר בָּרָא אֱלֹקִים לַעֲשֹוֹת, דְּהַיְנוּ שֶׁה' יִתְבָּרַךְ בָּרָא אֶת כָּל הָעוֹלָם וּמְלֹאוֹ לַעֲשֹוֹת שֶׁהָאָדָם יַעֲשֶֹה בְּכָל דָּבָר מְלָאכוֹת וְאֻמָּנוּת הַרְבֵּה שֶׁעַל יְדֵי זֶה דַּיְקָא יִהְיֶה כָּל תִּקּוּן הַבְּרִיאָה. וּכְמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה בַּמִּדְרָשׁ הַחִטִּים צְרִיכִים לִטְחֹן הַתֻּרְמְסִין צְרִיכִין תִּקּוּן וְכוּ'. וּכְמוֹ שֶׁרוֹאִין בְּחוּשׁ שֶׁאֵין שׁוּם דָּבָר שֶׁבָּרָא ה' יִתְבָּרַךְ שֶׁנּוּכַל לְהִשְׁתַּמֵּשׁ בּוֹ לִצְרָכֵינוּ בְּלִי שֶׁיַּעֲשֶֹה בּוֹ הָאָדָם כַּמָּה עֲשִֹיּוֹת וּמְלָאכוֹת וְתִקּוּנִים, כְּגוֹן הַחִטִּים אִי אֶפְשָׁר לֶאֱכֹל כִּי אִם עַל יְדֵי טְחִינָה וּרְקִידָה וַאֲפִיָּה וּלְכָל זֶה צְרִיכִין כַּמָּה אֻמָּנוּת וּמְלָאכוֹת בְּכַמָּה תַּחְבּוּלוֹת שׁוֹנוֹת. וְכֵן בְּמַלְבּוּשִׁים שֶׁאִי אֶפְשָׁר לִלְבֹּשׁ הַפִּשְׁתָּן וְהַצֶּמֶר כִּי אִם עַל יְדֵי כַּמָּה וְכַמָּה מְלָאכוֹת וְאֻמָּנוּת שׁוֹנוֹת וְכֵן בְּכָל דָּבָר שֶׁבָּעוֹלָם. וּבְכָל פַּעַם בְּכָל דּוֹר וָדוֹר נִתְחַדְּשִׁים כַּמָּה וְכַמָּה חָכְמוֹת וְתַחְבּוּלוֹת בְּעִנְיַן הָאֻמָּנוּת וּכְלֵי הָאֻמָּנוּת לְמַהֵר לְהוֹצִיא כְּלִי לְמַעֲשֵֹהוּ (שֶׁהוּא מַה שֶּׁעוֹשִֹין תַּחְבּוּלוֹת הַרְבֵּה שֶׁקּוֹרִין מַשִּׁינִין) וְכָל אֵלּוּ הָאֻמָּנוּת וְהַתַּחְבּוּלוֹת וּכְלֵיהֶם שֶׁיֵּשׁ בָּהֶם חָכְמָה גְּדוֹלָה כֻּלָּם נִמְשָׁכִין מֵחָכְמוֹת הַתּוֹרָה, וְכָל חִיּוּתָם וְקִיּוּמָם הוּא עַל יְדֵי אֱמוּנַת חֲכָמִים, כִּי הַחֲכָמִים הָאֲמִתִּיִּים שֶׁבְּכָל דּוֹר וָדוֹר מִתְיַגְּעִים וְטוֹרְחִים בְּכָל פַּעַם לְהַשְׁלִים אֶת הַתּוֹרָה לְחַדֵּשׁ בָּהּ חִדּוּשִׁים אֲמִתִּיִּים וּלְהַרְבּוֹת # רחבעם בחיקו של דוד

## כתם פז פרשת וירא

ועוד דרש רבא: "מ"ד: 'רבות עשית אתה יי' אלהי נפלאותיך, ומחשבותיך אלינו'. 'אלי' לא נאמר, אלא 'אלינו', מלמד שהיה רחבעם יושב בחיקו של דוד. א"ל: עלי ועליך נאמרו שתי מקראות הללו" - פסוק פתחת למוסרי ופ' נפלאותיך ומחשבותיך אלינו שאמרם דוד ברוח הקדש.‏

# מרחבעם נפגמה הלבנה

## כתם פז – פרשת וירא

מדוע המשיח יצא דווקא מהבכירה החצופה )אם מואב(‏

ת"ח: מואב ועמון אינון קראן לון שמהן וכו' - גם זה מענין הסוד, להודיע שלא בא דוד מהצעירה, כי אם מהבכירה החצופה שלא נתביישה מלקרות שם בנה ע"ש אביה ופרסמה הענין: מאבא הוא, כדי שיצא דוד ע"ה גבור מנצח לוחם מלחמות יי', לא ישא פנים לאויבי השם כאשר נשא שאול שחמל על אגג ולא הרגו, עד שכעס עליו שמואל הנביא בדבר יי'. ולכן בא מהבכירה החצופה דמהניא כלפי אויבי השם. ואמר: כי לכך נכתב בענין הבכירה בשכבה ובקומ"ה מלא בו'ו, לרמוז כי מיי' היתה לה שיצא ממנה שבט מושלים - והוא המלך המשיח. וניקוד על ו'ו, בגין דסיועא דלעילא הוה אשתכח בההוא עובדא - וכבר ידעת סוד הו'ו שהוא המשפיע בה'א מלכות בית דוד.‏

ואחרא כתיב: 'ובקמ"ה' חסר, בגין דהא לא נפק מינה חולקא לקב"ה כהאי אחרא - כלומר: שלא יצא מנעמה העמונית בן שיהיה בשלימות מלא כדוד, כי רחבעם שיצא ממנה כבר התחילה הלבנ"ה להתפגם ולהיות חסירה מאור הו'ו שהוא שמש ומגן. וכבר ידעת, כי משלמה ואילך התחיל החסרון בסוד הל' מלכים המיוחסים לימי הלבנה. ולרמוז זה, נכתב "ובקמה" של צעירה חסר, כי רחבעם לא היה שלם בסוד הו'ו כדוד ע"ה.‏

ר"ש אמר: כתיב 'ולא ידע בשכבה' - לא ידע, דזמין קב"ה לאוקמא מינה דוד מלכא ושלמה וכו' - הנה ר"ש ביאר, כי הידיעה הזאת לא תתפרש כענין "וידע אדם את חוה אשתו" שהוא ענין הרגש ידיע' הזכר לנקבה, כי גם כי יהיה האדם שכור או חולם חלום, לא ימנע מהיותו מרגיש בפועל ההוא כשהוא מתעסק בו, וכ"ש עם אשה. כי הידיעה הזאת היא הרגש חומרי, ואיברי הגוף מרגישים בו, ואיך א"כ יכתוב הכתוב ולא ידע - והוא ידע, ואשם כדרך כל הארץ?! ולכן ביאר: כי ענין הידיעה הזאת ענינה ידיעת העתיד, שלא ידע שעתיד הקב"ה להוציא מהזיווג ההוא דוד ושלמה ויתר המלכים עד כי יבא שילה שיבא במהרה בימינו.‏

תו 'ובקומה' דכתיב: 'ותקם בטרם יכיר איש את רעהו' וכו' - יבאר כי בכאן נרמז המאורע ההוא של רות המואביה, ושמקרה שתיהן מקרה השגחיי ומקרה א' להן, כי מה שנא' בזו נאמר בזו. כי בבכירה כתיב: "בשכבה", וברות כתיב: "ותגל מרגלותיו ותשכב". ובבכירה נאמר: "ובקומה", וברות כתיב: "ותקם בטרם יכיר איש את רעהו".‏

ומ"ש ר"ש: בגין כך נקוד בו'ו - יתכן שרמז גם כן ר"ש לתוספת הו'ו אשר במלת 'בטרם' - "בטרום" כתיב, ונרשמה הו'ו ב"ובקומה" לרמוז על הו'ו הנוספת במלת "בטרם". ואני אגיד לך את הרשום בכתב אמת, ואם לא ביאר ר"ש הענין בארוכה. כבר ידעת, כי כל מלא וחסר שבא בתורה הוא במקום החולם - שתבוא הו'ו במקום החולם, או שהחולם ימלא חסרון הו'ו. וכאן בא הו'ו במלת "בטרם" שאין לה שייכות כלל לא בחסר ולא במלא, ואם כן יותר היה ראוי שתבוא הו'ו הנוספת במלת "ותקם" תחת החולם, ולא במלת "בטרם", ויהיו נא אז שניהם מלאים, וזה יורה על זה, וזה על זה. אמנם בהיות הו'ו הנוספת במלת "בטרם", רמז לנו דבר גדול. והוא להודיענו: שכבר קדמה קימה אחרת מלאה קטרת לקימה זו, בטרם שתבוא זו לעולם בבכירת לוט, להודיע: דאסתייעא מלתא מן שמיא בטרם שהגיע תור רות לבא לעולם, והיא שעמדה לה. ולכן באה הו'ו במלת בטרם שלא כמשפט המלא והחסר, כי תורתנו הקדושה כלה רמזים - מליאיה וחסיריה מלאים כל טוב, והכל עשוי בדפוס עליון.‏

# רחבעם מלך רשע

## כתם פז – השמטות הזוהר

### שבעה צדיקים כנגד שבעה רשעים‏

ומה שהמשיך אל הענין הזה ספור השבעה הצדיקים, וז' המלכים החטאים, בא להודיע מה שרמזנו למעלה ב"צור ילדך תשי"; וכי יש ז' כנגד ז', וכי קמו בעולם ז' צדיקים שהיו דוגמא לשבעה עמודים אשר חצבה החכמ"ה, המעמידים את העולם. והם: אברהם, יצחק, יעקב, יוסף, משה, אהרן, דוד. ולנגדם בקליפה ז' עמודי חשך ברגליה המטונפים הידועים לבעלי הקבלה, נקראים ספירות הטומאה. וקמו ז' החטאים האלה בנפשותם מחריבי עולם, והם: ירבעם, בעשא, אחאב, יהוא בן נמשי, פקח בן רמליהו, ומנחם בן גדי, והושע בן אלה. הז' הצדיקים עזרו אל כח אדנ"י להשפיע בתחתונים, ולהעמיד העולם; ושבעה הרשעים תששו כביכול כחה, כמו שביארנו למעלה; ושחוייב כפי החכמה שתתגרש הא"ם שלא יסרה אותם, מבית בעלה עם בניה. והגירושין כבר רמזנום לסבת כי לא מצאה היונ"ה מנוח לכף רגלה, לשישרה טובה ושפעתה עליהם. ולפי שלא יסרה אותם ז' על חטאתיהם, מנעו הטוב ממנה - מצדם לא מצדה.‏

ויש מקום עיון בכאן למה לא יסרה אותם, ולא אוכל לפרש כי הדברים עתיקים.‏

ושם תמצא שקבלו ע"ה בקבלתם, שאלה החטאים בנפשותם הולידו י"ב בנים רשעים כאבותיהם - כי חוייב כפי הדוגמא כי זה לעומת זה עשה האלהים. כי כשם שיש בעולם הקדושה ז' וי"ב: שבעה בסוד "חצבה עמודיה שבעה", וי"ב אלכסונות היוצאות מהנקודה; כן יש בקליפה לנגדם ז' וי"ב אלכסונות, אשר להוראתם באו בעולם הגלגלים ז' כוכבי לכת וי"ב מזלות. וזה היה סבה שנשברו כביכול הז' הצנורות הקדושות מהשפיע מימיהם אלינו, וי"ב הגבולות אשר באו בדוגמתם ז' אבות וי"ב שבטים. ויאמר שלא חרבה ירושלם עד שנשלמו מנין עשרה מלכים רשעים, שהיו נגדיים לעשרה קדושות. והם: רחבעם, אביה, אחזיהו, אחז, מנשה, אמון, יהואחז, יהויקים, יהויכין. ולפי שמספר אלה אינם אלא ט', אמר כי שלמה אחוז בהם מצד אחד, ואחוז עם הצדיקים מצד אחד. וכן היה: כי לעת זקנתו נשיו הטו את לבבו, ועל סוד זה סבר מי שסבר, שאנשי כנסת הגדולה לדעת רב יהודה למנותו עם השלשה מלכים לדון אותו כפי הפגימה. יצתה בת קול ואמרה להם: "חזית איש מהיר במלאכתו לפני מלכים יתיצב" - בגן עדן. "ולא לפני חשוכים" - כמו שפי' רש"י ז"ל וביארו הסוד באמרם שם: 'אמרה בת קול: מי שהקדים ביתי לביתו, ולא עוד אלא שבנה ביתי בשבע שנים" - וכבר ידעת סוד ז' שנים העליונים שרמזו בזה, כי ראשו האחד אחוז שם. ושתקן ז' עמודי אור - ועכ"ז אמרו שלא השגיחו בבת קו"ל, זה כשם שלא השגיחו בדמות דיוקנו של אביו ולא באש שירדה מן השמים ולחכה בספסליהם עד שיצתה בת קול פעם שנית ואמרה: "המעמך ישלמנה כי מאסת"? ופי' רש"י ז"ל: וכי עליכם לשלם תשלומי עונשו של אדם שאתם מואסים בו בשלמה לומר שאין לו חלק?!‏

הנה נתבאר מדבריהם כי סוד הענינים האלה הנסתרים מהחיוב הנראה לעין כפי הידיעה העליונה, הוא נעלם ונסתר. אמנם למדנו מדבריהם ששלמה היה העשירי כמדתו שהיא העשירית, אשר חציה העליון הפונה אל שמש ומגן יי' צבאות מליאה - ודמיון זה הלבנה התחתונה, כי לעולם אינה חסרה יותר מחציה. ואם היא חסרה הוא מהצד הפונה אלינו כמו שכבר ביארנו. וכן היה שבתחלת מלכותו בנה בית המקדש ולבסוף נפגמה "הלבנה", והלכה הלוך וחסור, וכן המלכים הרעו אחריו.‏

והנה הסוד הזה ביארו רז"ל בפסוק: "החדש הזה לכם ראש חדשים". אמרו: סימן מסר להם בימי החדש, שט"ו ימים לחדוש הלבנה הולך ומתוסף האור, וט"ו אחר המילוי הולך הלוך וחסור. ולפי שימי הלבנה אינם שלשים מלאים כי אם כ"ט י"ב תשצ"ג, נמצא יום ט"ו מתווך בין י"ד ובין ט"ו,. והוא סוד אמרם: "אור לי"ד", כי ימי המלוי י"ד וימי החסרון י"ד בסוד העתים של ס' קהלת הנודעים מדבר והפכו, שהם כ"ח שתי ידות. א"כ הט"ו הוא האמצעי האוחז בי"ד של אורה, ואוחז בי"ד של פגימה. וכן על המספר הזה מנו רז"ל י"ד מלכים עד שלמה ע"ה בסוד מלוי הלבנה, וי"ד מלכים משלמה עד צדקיהו, שאזי היתה הלבנ"ה חסרה מכל אורה. בסוד: "ואת עיני צדקיהו עור".‏

הנה נתבאר כי שלמה ע"ה היה זמנו בין האור והחשך, והנסתרות ליי' אלהינו. האמנה כי מובטחים אנו באל יתברך שתשוב הלבנה ותשב על מלאת, כמ"ש רשב"י ע"ה: זמין קב"ה לאתבא שכינתא לאתרהא ולישראל לאתרייהו ולאתפרעה מאינון דסטרא דשמאלא דעאקו לון וכו' - שנאמר "ועלו מושיעים בהר ציון". וזה יהיה ביום הששי עצמו, אשר אנו בו היום שנת כ"י פאר"ך לפ"ק. ואמרי ביום הששי עצמו, לאפוקי מסברת הפוקרים שהם סוברים כי אין גאולה עד סוף - השמטה תפח רוחם ונשמתם! וכבר בא לידי ספר חדש, חדשו אחד מבעלי ברית ולא באמונה, וזרק בו חצים ומות בענין תחיית המתים, וגלה דעתו כי תחיית המתים הוא בשב"ת עצמו ולא בערב שבת. והוכחתיו בשבט אנשים רבים ונכבדים - קדמונינו ז"ל - שהגאולה עתידה להיות ביום הששי, וכן תחיית המתים וחזר ותיקן את אשר עות בספרו. ולא ידעתי אם מאימה עשה זה או מאמונה. והאל יכפר בעד:‏

# רחבעם התגאה, ע"כ נקרעה מלכותו

## ספר הסולם – פרשת מקץ אות יד

דבר אחר מלך וכו': פירוש אחר. מלך במשפט יעמיד ארץ זהו דוד המלך, שכתוב, ויהי דוד עושה משפט וצדקה‏ לכל עמו. והוא קיים הארץ בחייו, ובזכותו היא עומדת לאחר מכאן, דהיינו לאחר פטירתו מן העולם. ואיש תרומות יהרסנה, זהו רחבעם. פירוש. שאיש תרומות, פירושו איש גאה, ורחבעם התגאה שאמר, אבי יסר אתכם בשוטים ואני בעקרבים (מלכים א י"ב), ע"כ נקרעה ממלכתו.‏

## זוהר חי על בראשית קצג:

ד"א מלך דא קב"ה כמו שבארנו על סוד צ' דצלם במשפט הם הגבורות שבזעיר ויורדין ליסוד וגורמין הקמת ברית וחימום לעת הזיוג כמבואר בקול ברמה יעמיד ארץ דכתיב וכל הארץ באו מצרימה לשבור אל יוסף בסוד קנס אחד יוסף נער מטטרון שמשם יוצא השפע מהאכל יצא מאכל ור' יוסי מפרש מלך צ' דצלם דעת דזעיר ששם יסוד ונקרא יסוד עליון במשפט דא יעקב ת"ת המכריע בין חו"ג המתפשטין ומסלק בזה הרעב כי בגלות מצרים החו"ג סתומים במצר וגורם רעב ויניקות הקליפות כמבואר בדרוש הספירה וכשבא יעקב נתפשטו החסדים והחיות והשפעות מזון ואח"כ מפרש על דוד המלך יסוד ומלכות דרחל ושורשו מלאה ומבואר בזוהרקי"ע קכ"ד שלא תאמר כשהמלכות שותת ומקבלת שפע מימי הנקודה עליונה אחר רדת למטה השפע דרך המדריגות אלא מההוא אתר דשקיו דנחלין נפקין ולא פסקין שהוא עלה למעלה ומקבלת בתחלת יציאות המים מן יסוד דבינה ומקבלת מן נתיב עליון נקודת ציון שבבינה ששם נקראין צדקה הגבורות מים רבים מנצפך שבבינה ודוד היה מלאה נקודת ירושלים ונטל חלקו ירושלים וחלק חבירו שאול נקודת ציון הפנימי ואותו נקודת ציון מקבלת מנקודת ציון שבבינה ששם משפט וצדקה ואיהו קיים ארעא מלכות ירושלים:‏

ואיש תרומות פירש בו אדונינו הקדוש המק"מ דקאי על מי שיש בו גסות הרוח והתרוממת של גאוה כגון רחבעם אבי ייסר ואני בעקרבים וכן פרעה הרשע לי יאורי:‏

# נועם וחובלים

## רש"י על זכריה יא פסוק ז'

וארעה את צאן ההרגה - כל אלה דברי הקב"ה לנביא ואני רעיתי אותם בימים הקדמונים. לכן עניי הצאן - אמת עניי הצאן היו כשהתחלתי לרעותם. ואקח לי שני מקלות - לסוף ימים חלקתים לשתי ממלכו' בעוונם. לאחד קראתי נועם - ירבעם אמר לנהלם לעט. ולאחד קראתי חובלים - רחבעם אמר להם ליסרם בעקרבים ולפי שדרך צאן לנהלן במקלו' קרא למושליהם מקלות:

# רצה לבנות בית המקדש, אך בנין ילדים סתירה

## שפתי חכמים על מסכת מגילה לא:

תניא וכו'. וז"ל התוספתא (ע"ז פ"א) רשב"א אומר אם יאמרו לך ילדים בנה בית המקדש אל תשמע להן, ואם יאמרו לך זקנים סתור בית המקדש שמע להן, מפני שבנין ילדים סתירה, וסתירת זקנים בניין, וראיה לדבר רחבעם בן שלמה עכ"ל:

וסימן לדבר רחבעם בן שלמה. עיין (מלכים א' י"ב). בענין כשמת שלמה המלך, ומלך רחבעם בנו, ויבא ירבעם בן נבט והרבה מישראל לפני רחבעם לאמר, אביך שלמה הקשה עולנו, ואתה הקל מעולו הכבד אשר נתן עלינו ונעבדך, ויען רחבעם המלך עם הזקנים אשר היו עומדים את פני שלמה אביו, ויאמרו אליו הזקנים אם היום תהיה עבד בהכנעה (שתדבר בהכנעה לעם הזה) ודברת אליהם דברים טובים, והיו לך עבדים כל הימים, אבל ויעזוב רחבעם עצת הזקנים, ויועץ עם הילדים, ויעצו הילדים לרחבעם כה תדבר עליהם אבי העמיס עליכם עול כבד, ואני אוסיף על עולכם, אבי יסר אתכם בשוטים (ל' שבט) ואני איסר אתכם בעקרבים (מין קוצים לעוקצין כעקרב), ויעזוב המלך רחבעם את עצת הזקנים, וידבר אליהם כעצת הילדים, ולבסוף נחרב בית המקדש על ידי עצת הילדים, וזה הסימן לדבר כאן:

# רעים כנעמן ולא כרחבעם

## בן יהוידע נדרים ט.

שם שיזדמנו לו ריעים כנעמן. פירשו רש"י ז"ל והרא"ש ז"ל, כאלישע הנביא ע"ה שהשיאו עצה לנעמן שיטבול בירדן ויתרפא. וקשה דברי אלישע הנביא ע"ה בזה אינם בסוג עיצה, אלא גזרה גזר שיהיו מי הירדן מרפאין אותו, כי מי הירדן אינם מרפאים צרעת מצידם, מה שאין כן דברי הילדים לרחבעם היו בסוג עיצה. ונ"ל בס"ד העיצה היא מה שיעץ לנעמן שישפיל עצמו ויעזוב גאותו, כדי שאז יתרפא מצרעתו במי הירדן, וכמו שאמרו המבארים ז"ל, הא דהניחו לעמוד בחוץ ושלח שלוחו אליו לומר לו שיטבול ז' פעמים, היינו כדי שישפיל גאותו בזלזול זה שזלזל בו, לבלתי עשות לו כבוד הראוי לשר גדול כמוהו, מפני שידע שהצרעת באה לו מצד הגאוה שלו:

ועוד נ"ל בס"ד, מה שאמר ריעים כנעמן קאי על הנערה, שנתנה עצה שילך נעמן אצל הנביא, ויבקש ממנו רפואה, דכתיב [מלכים ב' ה' ג'] ותאמר אל גברתה אחלי אדוני לפני הנביא אשר בשומרון אז יאסוף אותו מצרעתו, וכתבו המפרשים שנתנה לו עצה שילך לפני הנביא, ויחלה פניו בעבור דבר זה, כי הנערה הבינה מדעתה שהצרעת באה לו מחמת גאותו, ולכך יעצתו שישפיל גאות לבו, וילך ויפול לפני הנביא ויחלה פניו, ולזה אמר ריעים כנעמן זו הנערה שהשיאה עצה טובה זו לנעמן, ולא רעים כרחבעם אלו הילדים שהשיאו עצה לרחבעם להפך לדבר עם ישראל בגאוה וגבהות לב, שיעצוהו להשיב להם קטני עבה ממתני אבי ועתה וכו', אבי יסר אתכם בשוטים ואני איסר אתכם בעקרבים [מלכים א' י"ב י' וי"א]:

שם שסתירת זקנים בנין ובנין ילדים סתירה. נ"ל בס"ד, זקנים מספר אור, כי עצתם היא אור, וילדים מספר דמים, כי ילדים מעצתם יהיה שפיכות דמים, וכמו שהיה מעצת הילדים לרחבעם, והפרש שיש בין מספר זקנים ובין מספר ילדים, הוא מספר מחסה, שיש יתרון לזקנים על מספר ילדים, רמז דעצת זקנים היא מחסה לאדם המקבל עצה, וראיה לדבר רחבעם בן שלמה, שאם שמע לעצת הזקנים היה אור לו ואור לישראל, ואחר שלקח עצת הילדים בא דמים לעולם, שנחלקה מלכות בית דוד, והיה אח"כ מלחמות בין מלכי בית דוד, ובין מלכי ישראל, ונשפכו דמים לרוב:

# בגלל לשון הרע שקיבל דוד נחלקה המלכות

## שמירת הלשון ח"ב פרק יב

וּרְאֵה עוֹד עִנְיָן נוֹרָא מְאֹד, שֶׁנִּסְבַּב בְּמֶשֶׁךְ הַיָּמִים מִדָּה בְּמִדָּה, שֶׁהִנֵּה יוֹסֵף נִכְשַׁל בְּחֵטְא הַלָּשׁוֹן, כַּכָּתוּב בַּתּוֹרָה, וְלֹא רָצוּ אֶחָיו לִמְחֹל לוֹ בְּשׁוּם אֹפֶן, שֶׁהָלְכוּ אִתּוֹ בְּעֹמֶק הַדִּין, שֶׁאָמְרוּ (בְּרֵאשִׁית ל"ז כ'), "לְכוּ וְנַהַרְגֵהוּ", דְּהַיְנוּ, נְשַׁסֶּה בּוֹ אֶת הַכְּלָבִים, וּכְהַאי גַּוְנָא, וְנִתְרַצּוּ לְמָכְרוֹ לְעֶבֶד עֲבוּר זֶה, אַף שֶׁיְּהוּדָה הֵגֵן בַּעֲדוֹ, וְלֹא הִנִּיחָם לְהָרְגוֹ, שֶׁאָמַר (בְּרֵאשִׁית ל"ז כ"ו), "מַה בֶּצַע" וְגוֹ', אֲפִלּוּ הָכֵי לִמְחֹל לוֹ לֹא רָצָה בְּשׁוּם אֹפֶן, שֶׁאָמַר (שָׁם כ"ז), "לְכוּ וְנִמְכְּרֶנּוּ", וְתַחַת זֶה בַּדּוֹרוֹת הַעֲתִידִים, שֶׁנִּכְשַׁל אֶחָד מִזֶּרַע יְהוּדָה בְּאִסּוּר קַבָּלַת לָשׁוֹן הָרָע, לֹא רָצָה אֶחָד הַמְיֻחָד מִזֶּרַע יוֹסֵף לִמְחֹל לוֹ. וּמִי הוּא שֶׁנִּכְשַׁל בִּשְׁגָגָה כָּזוֹ? אֲדוֹנֵנוּ דָּוִד הַמֶּלֶךְ עָלָיו הַשָּׁלוֹם, שֶׁהֶאֱמִין לְצִיבָא מַה שֶּׁדִּבֵּר עַל מְפִיבֹשֶת רָעָה, וְאָמַר (שְׁמוּאֵל ב' י"ט ל'), "אַתָּה וְצִיבָא תַּחְלְקוּ אֶת הַשָֹּדֶה", יָצְתָה בַּת קוֹל וְאָמְרָה, רְחַבְעָם וְיָרָבְעָם [שֶׁהוּא מִזֶּרַע יוֹסֵף] יְחַלְּקוּ אֶת הַמְּלוּכָה. וְכַכָּתוּב בַּקְּרָא, שֶׁאַחֲרֵי מוֹת שְׁלֹמֹה בָּא יָרָבְעָם וְכָל יִשְֹרָאֵל לִרְחַבְעָם וּבִקְּשׁוּ שֶׁיָּקֵל מֵהֶם מְעַט עֹל הַקָּשֶׁה וְיַעַבְדוּהוּ, וְהֵשִׁיב לָהֶם קָשׁוֹת (מְלָכִים א' י"ב י"ד), "אָבִי יִסַּר אֶתְכֶם בַּשּׁוֹטִים, וַאֲנִי אֲיַסֵּר אֶתְכֶם בָּעַקְרַבִּים". מִיָּד הֵשִׁיבוּ כֻּלָּם (דִּבְרֵי הַיָּמִים ב' י' ט"ז), "אִישׁ לְאֹהָלֶיךָ יִשְֹרָאֵל" וְהִמְלִיכוּ אֶת יָרָבְעָם. וְהַסִּבָּה הָיְתָה מִפְּנֵי קַבָּלַת לָשׁוֹן הָרָע שֶׁל דָּוִד, כְּמוֹ שֶׁאָמְרוּ חֲכָמֵינוּ זִכְרוֹנָם לִבְרָכָה, אִלְמָלֵא קִבֵּל דָּוִד לָשׁוֹן הָרָע, לֹא נֶחְלְקָה מַלְכוּת בֵּית דָּוִד, וְלֹא עָבְדוּ יִשְֹרָאֵל עֲבוֹדַת כּוֹכָבִים וּמַזָּלוֹת, וְלֹא גָּלִינוּ מֵאַרְצֵנוּ. וְזֶהוּ הַכֹּל מִדָּה בְּמִדָּה, כְּמוֹ שֶׁכָּתַבְנוּ:‏

# מלכים זה לא טוב לישראל

## יוסיפון פרק ארבעים ושמונה

עוד - בני עמי! התיצבו על עמדכם, ואוכחה עמכם! הגידו נא לי: מתי היו אבותיכם חפשים, ולא היה אתם סער מבוכה, ממלחמות גוים וממשלות לאומים? הלא מיום צאתכם מארץ מצרים, עד יום מלוך שאול בן קיש, הייתם נוצחים וגוברים, כי יי היה מלככם, והייתם אדונים, ולא הייתם עבדים לאדם - כי אם לאלהים. ואחר כן, כי השטין אתכם יצרכם הרע, ותמאסו את יי מהיותו לבדו מלך עליכם, בחרתם לכם אדם למלוך עליכם, ולשפוט אתכם - כיתר הגוים, המולכים והשופטים עליהם שופטי אדם ומלכי איש! וימלוך עליכם שאול בן קיש ויתר המלכים, ועבדתם אותם, והיו פניכם לעבדים לו, והטובים מכם למשרתים, ובנותיכם היקרות לרקחות ולטבחות ולאופות. ואחריו מלך דוד עליו השלום עליכם, אשר משל בגוים רבים אשר סביבותיכם - אחרי מות שאול. וגם הוא העביד אתכם כרצונו, עד אשר המית רבים מכם - להשלים תאות לבבו, ואחרי מות דוד בן ישי, עבדתם את שלמה בנו עליו השלום - וגם הוא רדה בכם כנפשו, וישלח יד בבנותיכם ובבניכם, ויקח מהם לעבוד לפניו, ולשרתו כחפצו. ואחריו באו מלכים רשעי ארץ עליכם, וקמו עליכם לרעתכם - ומאז התחילה ההשמדה בארצכם. והטוב שבהם אחרי שלמה עליו השלום, היה רחבעם בנו, אשר אמר אליכם: (מלכים א' י"ב) "אבי - יסר אתכם בשוטים, ואני - איסר אתכם בעקרבים!" וכה עשו יתר המלכים, אשר מלכו אחריו - עליכם. כי מעט מהם היו רצויים ליי, ויתרם השחיתו התעיבו אין עושה טוב מהם אין גם אחד, מושל ביראת אלהים. ונקשרה הממלכה בכם, ממלך אחר מלך - שנים רבות. עד אשר באו מלכי כשדים, והגלו אתכם מעל אדמתכם - בבלה. עד אשר בא כורש מלך פרס, והשיב את שבותכם אל אדמתכם - ולא רקים השיב אתכם, כי אם בכסף וזהב ועושר וכבוד, וביד גדולה לאלהים, ותחשב לו התשובה ההיא לצדקה. ואחרי מות כורש, קמו עליכם מלכי יון הרשעים, להשמיד ולהחרים ולהאביד, וילחמו עמכם, וילחצו אתכם בלחץ רב ובעול קשה. ויער יי את רוח כהניו, החכמים השלמים בני חשמונאי וינקמו נקמתכם מכם. ואז - הייתם אחים ורעים לרומיים, ותגדל אהבתכם שנים רבות. ואחרי כן פשעתם כלכם בבני חשמונאי, אשר הושיעו אתכם ויגדילו שמכם, ותבחרו לכם איש מעמו - ושמו הורודוס, הוא המלך אשר הכביד את עולו עליכם. וימת הורודוס, וימלוך תחתיו ארקילאוס בנו. ויקשה גם הוא את עולו עליכם, ותפשעו בו כולכם, ותאמרו: "לא נעבוד עוד את מלכי יהודה - כי אם נעבוד את מלכי הרומיים, ותבקשו לעבוד את מלכי הרומיים כרצונכם. ותטו שכמכם לעבוד את אגוסטוס המלך, כי הוא חמל עליכם, ולא היה לכם לאכזר. ותעבדוהו - ככל הגוים אשר עבדו אותו - בלא חרפה. ויהי לכם לשם טוב, כי עבדתם את מלך טוב.‏

ועתה - אחי, בני עמי! מה לכם היום הזה, כי אתם הבוערים מכל הגוים, ומחפשים את המות בנרות, ולא תחמלו - לא על נפשותיכם, ולא על צאצאי מעיכם?!.‏

בינו נא בצאצאי הארץ כלם, זולתכם - באדם ובבהמה ובחיה, ובשרץ ובעוף השמים ובכל דגי הים! וראו - כי הגדול ימשול בצעיר! ואין כלימה ובושת לצעיר, כי יעבוד את הגדול ממנו. כי השור והגדי - ייראו את הארי, והאיל והשה - ייראו את הזאב, ופרה ורחל - ייראו את הדוב, והעזים - ייראו מן הנמר, והנץ - יירא מן הנשר, והיונה - תירא מן הנץ! ובינו לכם בחיות ובעופות ובבהמות, גם תראו, אשר יתגאה הפר הגדול - על הפרות הקטנות ממנו, והאיל יתגאה - על הצאן, והשעיר יתגאה - על עדר העזים! וככה - כל הבהמות והחיות הגדולות מושלות על הקטנות! ולמה לא ילמדו בני האדם מן הבהמות והחיות והעופות והלא אל אחד ברא הכל, ומושל בכל. והכל נקשר זה עם זה, ועומד זה בזה, וזה מעמיד לזה, והכל - מעמד לבורא העולם, ישתבח לנצח! והכל - ישוב אל העפר ברצון הבורא, יתעלה שמו!.

קחו לכם דמיון מחלקי העולם כולו, וראו - איך חלק ימשול בחלק, וחלק יכנע אל חלק. ולא תקשו ערפכם, לאנוס את מקרה טבע העולם ומקריו! ותמשיכו בחירתכם אחרי סבותיו, ואז - יקרא שמכם עם חכם.‏

ועתה - בני עמי! בהיותכם צעירים ונכנעים לרומיים, לא תחשבו - כי זה יהיה חסרון לכם וקלון, כי עתכם יבא, ותשובו אל יי בכל לבבכם, ותהיו גם אתם מושלים בעמים - כרצונכם, רק כי תלכו אחרי אלהיכם, הסומך אתכם בכל עוז! ואין הרומיים אשר ירדו בכם היום, גוי שפל או חלש, יותר מהגוים אשר רדו בכם מלפנים - כי גוי איתן הוא, ומאת השמים המליכוהו להיותו מושל בעמים. וקחו לכם עדות ודמיון ומשל, מהבהמות והחיות המתגאים ומושלים זה על זה, לפי טבע העולם וסדרו. ודעו, כי המקרה בני אדם, לא יוכל הגדול למשול בקטן - כי אם בעונותיו אשר יגדלו עליו, יגדל מי שכנגדו להעבידו, ויטה שכמו לסבול עולו ולקבל ממשלתו.‏

ועתה - בני עמי! לבשו ענוה, וקחו לכם דרך להיותכם ככל הבריות, ולא תרצו להיותכם בעלי תמורה ושינוי בעולם. קחו עמכם דברי אלה, ושובו לעצתי הטובה ושמעו אלי, ועבדו הרומיים - ותחי נפשכם, כי יכרתו לכם ברית בחסדם!".‏

ויהי, כדבר יוסף הכהן באזני עם ירושלם את הדברים האלה, וישאו קולם ויבכו, ויחרקו שיניהם עליו, ויחרפוהו, ויקללוהו קללות נמרצות מעל החומה, וישליכו עליו אבנים, ויורו עליו חצים הרבה - ויבקשו להמיתו. וכאשר ראה יוסף כי לא שמעו לעצתו, ויקשו ערפם, ויעיזו פניהם נגדו, ויחל להוכיחם עוד - כדברי הספר.‏

# יש לו כח ואומץ יותר מאביו, לכן דיבר אל העם קשות

## רד"ק על מלכים א' פרק יב פסוק י'

קטני עבה - האבר הקטן שבי עבה ממתני אבי ומלת קטני קשה בדקדוק כי אם היה מגזרת קטן משפטו קטני כמו מן גדול גדולי אלא נשאר הקמץ בקו"ף כמו שהיה טרם התחברו אל הכנוי כמו שאמר מן בגוד בגודה אחותה ומן שליש בדקר שלישו והדומים להם שנשארו על תנועתם והנח נבלע בדגש ושב החולם לקמץ בחטף וענינו יותר יש לי כח ואומץ ממה שהיה לאבי ולמה אקל עולכם:

# יש לו זכות כי ירש מאביו ההנהגה

## מלבי"ם על מלכים א' פרק יב פסוק י'

וידברו אליו הילדים אשר גדלו אתו - ספר כי דבור זה לא היה מצד שהם עומדים לפניו דהיינו שהם שרי העצה, כי כפי שהיו שרי העצה העומדים בראש העם לא היה ראוי שידברו כן והיה להם להשכיל כי מרה תהיה באחרונה, רק דברו כן מצד שהם ילדים אשר גדלו אתו בשטות וידברו כחסרי דעת, כה תאמר אשר דברו אליך - ר"ל הנה הם טוענים לאמר אביך הכביד את עולנו - והחזיק בזה שלא כדת ולכן אתה הקל מעלינו - אנו יועצים כה תאמר כה תדבר אליהם - יש הבדל בין דבור ואמירה, האמירה היא המוחלטת והדבור הוא הצעת הדברים, שמטעים את האמירה ומדבר באורך מדוע אמר כן (כמ"ש בחבורי התו"ה ויקרא סי' ג') ור"ל בין האמירה עצמה שהוא הענין שנדונים עליו ובין הדבור שהוא איכול הדברים והוצעתם יהיה באופן זה, שתאמר להם, אתם טוענים שאבי השתרר בחזקה שלא כדין, הלא קטני עבה - יותר יען שבא ממתני אבי - ר"ל הלא אבי התחיל בהנהגה זו ובכ"ז הייתם מחוייבים לשמוע לו כ"ש אנכי שיש לי זכות על הנהגה זו שירשתי מאבי:

# גם הזקנים וגם הנערים כיוונו לדבר אחד, שמלכותו תהיה בלתי מוגבלת

## מלבי"ם על מלכים א' פרק יב פסוק יג

ויען המלך - ר"ל באמת עצת הזקנים ועצת הנערים שניהם כוונו לתכלית אחת שימלוך מלכות בלתי מוגבלת, רק היו מחולקים על האמצעיים שיבחר לזה, שהזקנים יעצוהו שזה יהיה ע"י שיענם מענה רך, והנערים יעצוהו שזה ישיג ע"י מענה קשה, שיפיל אימתו עליהם, ולפ"ז בגוף התשובה שהשיב שאינו רוצה להגביל את מלכותו לא עזב את עצת הזקנים כלל, רק במה שענה את העם קשה - בזה כבר עזב את עצת הזקנים - כי עי"כ חרה אף העם בו, אבל עדיין היתה תקנה אם היה ממלא רצונם להגביל את מלכותו באמת ושיאסר במסורת משפטי המלוכה כפי חפץ העם, שאז לא היו מקפידים על דבריו הקשים אחר שהשיגו בקשתם, אבל עם זה:

# בימי רחבעם מאסו ב-ג' דברים

## נבואת הילד – נבואה ה'

והנה נשלם ביאור חזיונות נחמן המקודש מבטן אמו. ואע"פ שהצרות העתידות עדיין הם חזקות, וכמו הרה תקריב ללדת תחיל תזעק בחבליה כן הם אלינו חבלי משיח. ומי מכלכל את יום כוחו ומי העומד בהראותו כי הוא כאש מצרף וטהר וזקק את בני דורו כזהב וככסף. ולפי רוב הצרות אמרו בפרק חלק, אמר עולא ייתי ולא אחמיניה, וכן אמר רבא ייתי ולא אחמיניה. מ"מ בידנו הוא להקל מעלינו את הצרות ואת התלאות כשנעתר אל אלוה וירצנו ונרבה בתפלה בכונה ונשפוך שיחה לפניו. וכבר זכר זה נחמן בחזיוניו, שהזהירים בתפלה ומצות ומעשים טובים יהיו נצולים. ואי משום שמא יגרום החטא יפשפש האדם בכל יום במעשיו וימהר יחישה תשובה ויתודה בכל לילה טרם ישכב וימחול לאשר אשם לו ויסיר את לב האבן מבשרו ולא ישנא את אחיו בלבבו. כי בית ראשון מפני שהיו בו ע"ז וגלוי עריות וש"ד חרב, כאשר ארז"ל ושבו לסוף ע' שנה ובית שני שלא היו בו כל אלה חרב מפני שנאת חנם. ואתה רואה שעדיין לא שבו יען כי כל יום קללתו מרובה משל חברו באיבה ומשטמה ולשון הרע ורכילות אשר כלם ענפים מתפשטים מן השורש הפורה רוש ולענה אשר הוא שנאת חנם. ואיש את רעהו בפיו שלום ידבר ובקרבו ישים אורבו, והרוצה להיות נושע צריך שיטהר לבו מכל אלה. ככתוב "כבסי לבך ירושלם למען תושעי" (ירמיה ד' יד'). ויזהר אדם בשמירת שבת בכל יכולתו. כי אתה ידעת מה שדרשו ז"ל על פסוק "אשר ישמרו את שבתותי" אם ישמרו ישראל שתי שבתות כהלכתן מיד הן נגאלין. ויקיים ג' סעודות לשום שמים על השולחן הטהור לפני יי'. והיה כשבתו על כסא שולחנו עיניו אל קדוש ישראל תראינה. וישב במורא במיעוט שחוק במיעוט קלות ראש ויעסוק בתורה ובגמילות חסדים. וכן אמרו בפרק חלק, אמרו לו לר' אליעזר מה יעשה אדם וינצל מחבלו של משיח, אמר להם יעסוק בתורה ובגמילות חסדים. וכל אדם יחזיק במעוזו ויאחז צדיק דרכו וטהור ידים יוסיף אומץ בטהרת הלב והעינים שהם סרסורי חטאה, כדכתיב "ולא תתורו אחרי לבבכם" וכו'. וארז"ל אמר הקב"ה הב לי לבא ועינא ואנא ידע דאת דילי. וכן הוא אומר "תנה בני לבך לי" וכו' (משלי כג' כו'). ויכניע האדם לבבו הערל ויפשיט מעליו מעיל הגאוה וגסות הרוח ויזהר בל ימנה מד' כתות שאינן מקבלות פני שכינה כי משחתם בהם מום בם. והסר ממך עקשות פה ולזות שפתים ושבועות לבטלה והזכרת השם הנכבד, כי אם במה שהורשית באימה וביראה. ותן כבוד לשם הגדול לאשר כל צבאות עולם ירופפו ויתמהו מגערתו ויתחלחלו צבאות מעלה מהזכרתו. ותתן על פניך יראתו ותקבע בלבך אהבתו ותלמוד תמיד ספר היראה ואגרת התשובה לרבינו יונה החסיד ע"ה. כי הם ספרים קטנים כוללים הרבה מדרכי התשובה ותורת האדם ותתנהג בהם ותצליח. ויותר מהמה בני הזהר לצום ולהתענות עם אנשי המשמרות ולקרוא אל אלהים בחזקה ככל אשר הם עושים להתודות עונם ועון אבותם ועון אחיהם כל בית ישראל, ולהתפלל על כל קהלות הקדש, יי' צבאות יגן עליהם ויצילם מכל צרה. ולהתודות על הג' דברים שמאסו ישראל בימי רחבעם שהם מלכות שמים ומלכות בית דוד ובית המקדש, ולבקש שלשתם. וגם בלא תענית התפלל על כל אלה כי זה נבחר ליי' מזבח ובזה תקובל תשובתך כשתתפלל על כל אחיך בני בריתך. כדכתיב "ויהוה שב את שבות איוב בהתפללו בעד רעהו" (איוב מב' י'). ותתן פרוטה על כלם לצדקה בכל שבוע הן בכל את שתוכל ותהיה מצדיקי הרבים ויי' אלהיך ירצך:

# אחז באמת

## דובר צדק – אות ח'

שם (עירובין י"ג ע"ב) שלוש שנים נחלקו וכו'. פירוש שלוש שנים הוא שלושה מדריגות דחשיב להלן שהם שלושה האבות אברהם יצחק ויעקב המקבילים נגד שלושה מיני רע, שלעומתן שלוש אומות אדום ומואב ועמון. כמו ששמעתי שהם השלושה מדות רעות כעס [אדום שהוא נגד לא תרצח כמו שאמרו ז"ל (זוה"ק ח"ג קצ"ב ע"ב) וכמו שנאמר (בראשית כ"ז, מ') על חרבך תחיה וממנו לא נשתלה מלכות בית דוד שלא היה תיקונו על ידו כידוע. וכמו שאמרו דכעס אינו ממחנה ישראל ומלכות בית דוד רומז לכללות הכנסת ישראל כידוע כי המלך הוא העם. אבל מפלתם על ידי בית יוסף כמו שאמרו ז"ל (בבא בתרא קכ"ג ע"ב) כי הוא היפך הכעס כמו שנתבאר במקום אחר ואין שתילת הכנסת ישראל ממחנה ישראל מזה וכמו שאמרו ז"ל (יבמות ע"ט.) שלושה סימנים וכו' שהם כולם נגד עשו שרומז בשלוש אותיותיו היפך זה. והם שלוש פעמים שנזכר מחייתו כמו שאמרו במגילה (ז'.) ואין כאן מקום להאריך] וגאוה [מואב כמו שנאמר (ישעיה ט"ז, ו') שמענו גאון מואב גא מאוד וגו' וממנו נשתל בישראל ותיקונו דוד המלך ע"ה שאמר (תהלים כ"ב, ז') ואנכי תולעת ומשפיל עצמו על כבוד קונו] ושקר [עמון ולכך קראתו בן עמי שהוא העלמת האמת. ומכל מקום אמרו ז"ל (נזיר כ"ג ע"ב) שקיבלה שכר על זה. כי יש שקר טוב ובגין דא מותר לשנות (עיין בבא מציעא כ"ג ע"ב) וגם שזהו מצד בריאת העולם דלכן נקרא עלמא דשיקרא (ויקרא רבה כ"ו, ז') שכל העולם הזה כן. ולפיכך לא נשתלה עד רחבעם שהוא אחר בנין בית המקדש ושלמה המלך ע"ה הרומז למשיח אחר כך הוא זביחת היצר לגמרי ואז נסתלק עלמא דשיקרא ורחבעם תיקונו לכן אחז באמת ועזב עצת הזקנים] ושלוש מדות אלו לא נתקנו עדיין לגמרי עד עת קץ. לכן לא זכו בני ישראל אז לנחול שלוש אומות אלו ולעתיד יזכו גם להם כמו שאמרו ז"ל (בראשית רבה מ"ד, כ"ג) ולא יתעסקו עוד בסביבת ההר. [כמו שאמרו (סוכה נ"ב.) הצדיקים נדמה להם כהר]:

# שלמה גרם לחורבן הבית ע"י שינה

## כוכבי אור – ששון ושמחה אות ו'

כִּי זֶה יָדוּעַ שֶׁכָּל בְּחִינָה כְּלוּלָה מִכָּל הַבְּחִינוֹת, וְכֻלָּם הֵם בִּכְלָלִיּוּת וּבִפְרָטִיּוּת וּבְכָל דּוֹר וָדוֹר וּבְכָל אָדָם וְאָדָם וְגַם בְּעֵת הַחִפּוּשֹ אַחַר הַר שֶׁל זָהָב וּמִבְצָר שֶׁל מַרְגָּלִיּוֹת, צְרִיכִים לְהִזָּהֵר וּלְהִשָּׁמֵר מִתַּאֲוַת אֲכִילָה, וּמִכָּל שֶׁכֵּן מִשְּׁתִיַּת הַיַּיִן וּמִלְּהִשְׁתַּקֵּעַ בְּשֵׁנָה כַּמּוּבָן כָּל זֶה בִּדְבָרָיו בְּמָקוֹם אַחֵר וְכָל צַדִּיק לְפִי בְּחִינָתוֹ כֵּן הוּא מוֹצֵא אוֹתָהּ וְדָוִד (שֶׁחַי שִׁבְעִים שָׁנָה) שֶׁמִּמֶּנּוּ יֵצֵא מָשִׁיחַ (שֶׁהוּא יִהְיֶה גְּמַר הַתִּקּוּן שֶׁל מְצִיאָתָהּ כַּמּוּבָא שָׁם בַּהַקְדָּמָה שֶׁל הַסִּפּוּרֵי מַעֲשִֹיּוֹת) עַל כֵּן הוּא אֲשֶׁר זָכָה לִמְצֹא אֶת מְקוֹם הַר בֵּית ה' דַּוְקָא עַל יְדֵי נִדּוּד הַשֵּׁנָה, בִּבְחִינַת "נִשְׁבַּע" וְכוּ' (כט) "אִם אֶתֵּן שְׁנַת לְעֵינַי לְעַפְעַפַּי תְּנוּמָה עַד אֶמְצָא מָקוֹם לַה'" וְכוּ', וְגַם צִוָּה וְהִזְהִיר אֶת שְׁלמֹה בְּנוֹ שֶׁיִּבְנֶה וִיתַקֵּן אוֹתוֹ בִּשְׁלֵמוּת וְעַל כֵּן דַּיְקָא בְּעֵת שֶׁבָּנָה שְׁלמֹה אֶת הַבַּיִת שֶׁאָז הָיָה צָרִיךְ לִהְיוֹת גְּמַר הַתִּקּוּן שֶׁל מְצִיאָתָהּ, אָז דַּיְקָא נִתְגַּבֵּר עָלָיו הַשָּטָן "זֶה לְעֻמַּת זֶה", עַד שֶׁנָּשָֹא אָז דַּיְקָא אֶת בַּת פַּרְעֹה שֶׁזֶּה גָּרַם חֻרְבַּן הַבַּיִת (כְּמַאֲמַר חֲכָמֵינוּ זַ"ל (וַיִּקְרָא רַבָּה פֶּרֶק י"ב, ד) עַל פָּסוּק "כִּי עַל אַפִּי וְעַל חֲמָתִי" וְכוּ') (אָמַר הַמַּעְתִּיק, וְעַל כֵּן כְּשֶׁנָּשָֹא אֶת בַּת פַּרְעֹה הָיָה יָשֵׁן שְׁלמֹה עַד אַרְבַּע שָׁעוֹת בַּיּוֹם כְּמוֹ שֶׁאָמְרוּ חֲכָמֵינוּ זַ"ל (שָׁם) כִּי אָז הִתְגַּבֵּר הַשָּטָן לְהַכְשִׁילוֹ בְּשֵׁנָה דַּיְקָא) וְעַל כֵּן גַּם יַעֲקֹב אָבִינוּ כְּשֶׁהָלַךְ אַחַר זִוּוּגוֹ "וַיִּפְגַּע בַּמָּקוֹם" הַזֶּה, וְאָז אָמַר "מַה נּוֹרָא הַמָּקוֹם הַזֶּה" (בְּרֵאשִׁית כ"ח) וְאַחַר כָּךְ נִסְתַּכֵּל עַל חֻרְבָּנוֹ וַיֹּאמֶר, "אֵין זֶה כִּי אִם בֵּית אֱלהִים", (וְעַיֵּן בַּמִּדְרָשׁ (בְּרֵאשִׁית רַבָּה פֶּרֶק ס"ט) שֶׁתֵּבוֹת אֵלּוּ מְרַמְּזִים עַל הַבַּיִת שֶׁיִּבָּנֶה לֶעָתִיד) שֶׁשָּׁם יִמָּצֵא בִּשְׁלֵמוּת וְעָלָיו נֶאֱמַר, "מִקְּדָשׁ ה' כּוֹנְנוּ יָדֶיךָ" שֶׁלֹּא יֵחָרֵב עוֹד לְעוֹלָם:‏

# רחבעם חכמתו קודמת ליראת חטאו

## צדקת הצדיק אות רכה

רכה) דברי תורה שעוברים דרך הלב פירוש שהלב מרגיש ומתפעל בהם הוא נקרא עץ חיים וסמא דחיי. כי הלב מקור החיות כמו שנאמר (משלי ד' כ"ג) כי ממנו תוצאות חיים וכאשר הוא מתפעל הרי שורש חיותו מתפעל על ידי הדברי תורה ומולידה חיות באדם. וזהו על ידי שיראת חטאו קודמת לחכמתו כי היראה בלב כמו שאמרו ז"ל (יומא ע"ב ע"ב) על פסוק (משלי י"ז ט"ו) ולב אין ויראה נוקבא ותורת אמת דכר להשלים ולשמח הלב כמו שנאמר (תהלים י"ט ט') פקודי וגו' אחר היראה המולדת לב נשבר ועצבות. ואז הדברי תורה נבלע בלב ומתקיימת כי בלב מקום רתיחת הדם ועל ידי תוקף החשק וכיסופין למתיקות הדברי תורה מתגבר כח האש שבו וכדרך שאמרו (תענית ד'.) אורייתא דקא מרתחא ליה. וידוע כח הבישול להבליע והבלוע אינו יוצא עוד. מה שאין כן בצונן וקרירות דהיינו שאינו עובר דרך הלב שהגם שמבין חכמה במוחו ומשיג עניני אלהות ודברי תורה אין הלב מתפעל על ידי זה כלל כי הוא כלמד חכמה חיצונית שאין לו שייכות לזה לפי שלא קדם יראת חטא שיתבונן על חסרונו וצורך הדברי תורה להשלימו אז אין חכמתו מתקיימת כדרך שאמרו בדואג (סנהדרין ק"ו ע"ב) דשכח הכל משום דרחמנא ליבא בעי שיהיה הדברי תורה נרגש בלב זהו העיקר שאז יש בו חיות והוא נבלע בו:

ובלאו הכי אמרינן בשבת (ל"א.) מוטב שלא העלית היינו כי חס ושלום יוכל להיות סמא דמותא כמו שאמרו ביומא (ע"ב ע"ב) כי בתלמיד חכם היצר מתגבר כמו שאמרו בסוכה (נ"ב.) וגם זה נקרא תלמיד חכם וכמו דואג ואחיתופל שנקראו תלמידי חכמים בכל מקום. וכשאין לו יראת חטא להגביר על היצר אז יוכל להיות חס ושלום כענין אחר שאמרו ז"ל (חגיגה ט"ו ע"ב) ספרי טועין וכו' וזמר יוני וכו' פירוש שכל חכמתו היה דוגמת לימוד חכמה חיצונית בתורת חכמה לבד ולא להשלמת החסרון שזהו עיקר שם תורה. ומשום דלא היה נקי אגב אימיה כידוע דאבא ואמא הם מוח ולב והוא חכמת מוחו מרובה רק דלב אין דהיינו אימא:

וכן בפשטיה מסתמא זוכין זה כל אחד מאב ואם הגשמיים כמו שאמרו (עדיות סוף פרק ב') האב זוכה לבנו בחכמה ואב מזריע לובן שממנו מוח ואשה אודם שממנו בשר ודם (נדה ל"א. לגירסת שאלתות) והלב בכלל. ומזה בא טומאת זיבה כי תורה נמשל לאשה (יבמות ס"ג ע"ב) והתלמיד חכם הוא האיש המשפיע בה. וקרי יוצא מאבר חי וזיבה מאבר מת. והיינו שלא על ידי התפעלות החשק שבלב כלל כנודע:

וזה שאמרו ז"ל (סנהדרין מ"ח ע"ב) זב מרחבעם שנאמר בו (מלכים - א י"ב י"ח) התאמץ לעלות במרכבה. וכמו ששמעתי על (שמות י"ג י"ט) ויקח משה עצמות יוסף שהעצמות של יוסף הוא יראת חטאו קודמת לחכמתו. וזה צריך ליקח בעת שרואה שינוי בהתנהגות שאינו מבין כהך דולא נחם אלהים דרך ארץ וגו'. ולכך ירבעם מרד בשלמה המלך ע"ה שטען עליו שחכמתו קודמת ליראת חטאו והוא מזרעא דיוסף. ולכן כשניצח למלוך ולמרוד ברחבעם נאמר ברחבעם התאמץ וגו' ונעשה זב שנתברר שהדין עמו ושהוא באמת יש עליו טענה לומר שהוא כן. וטהרת הזב הוא במים חיים פירוש על ידי חשק שבלב בחיות גמור לדברי תורה. כי החכמה תעוז לחכם ואי אפשר כל כך לבוא למדריגת היראה כמו לאהבה וחשק עצום לדברי תורה בלב גם זה מזכך החכמה להיות נבלע בלב ולהיות עץ חיים על ידי תוקף החשק שבלב בחיות גמור היינו בדביקות במקור החיים יתברך שמו שנקרא חי העולמים שהוא מקור החיות:

# רחבעם היה צריך להיכנע

## היכל הברכה על פרשת תולדות

[אתה עתה] תלת דין ואתה עתה הקל מעבדת אביך אתה עתה תעשה מלוכ' על ישראל בסוד אתה אבינו חכמה עתה מלכות כד אית בישראל משכילים דידעין לזרקא להאי אבנא להאי אתר דאתגזרת בסוד והחכמ' מאין תמצא ובאמת אם רחבעם היה נכנע והי' שומע אל עצת הזקנים לא היה נחלק' מלכות בית דוד כי הכניע' מבטל כל גזירות רעות של עתה שנגזר בעת' אע"פ שכבר נגזר שירבעם הרשע הצורר יהיה מלך הית' מתבטלת על ידי הכנע' וכן אם היה נכנע צדקיה הצדיק לפני נבוכדנצר הרשע לא נחרב מקדשנו אע"פ שכבר נחתמת הגזיר' כי הכנע' מבטלת כל הגזירות רעות וזהו ואת' עתה הקל לשעת' בעצת הזקנים אזי אתה עתה תעשה מלוכ' על ישראל בהנהגות אתה עליון לזמן עתה בסוד קץ משיח תלוי בחוט השער' בהכנע' אמת בסוד אתה עת"ה ברו"ך ה' בוא ברוך ה' שאז יצא העבד מקלל' לברכ' ותחזור למעלת אין עליון רדל"א והבן עיין גלגולים ותבין:‏

## נתיב מצוותיך נציב התורה שביל הראשון אות כח

כח. עיבור יניקה מוחין, הכנעה הבדלה המתקה, אורות ניצוצין כלים, עולמות נשמות אלהותו יתברך, ובימי עיבור לומד כל התורה בכח, ונר דולק על ראשו, ומביט מסוף העולם עד סופו (נדה ל:), כי בחינות עיבור ישנו בכל זמן, כשמסתלקין החיות והמוחין, ומתגבר החשכות והיסורין וחלישות דעת, ולא נשאר אלא רושם התורה, ולזה לעתיד כשיבטל הרע והסיגים 'הרה ויולדת יחדו' (ירמיה לא, ז), ומיתוק היסורין והחשכות הוא הכנעה, נכנע באמת בלב אמת, כי אם היה נכנע רחבעם כעצת הזקנים 'היום תהיה לעבד', לא נחלקה מלכות בית דוד, ולא עבדו ישראל עבודה זרה על ידי הרשע ירבעם המתגאה, ולא גלינו מארצנו. ואלו היה צדקיהו הצדיק (קיבל) [מקבל] הכנעה לפני נבוכדנצר שחיק טמיא, לא נחרבה בית תפארתנו ולא גלינו מארץ.‏

הכלל, הכנעה אמת מבטלת כל הגזירות רעות והיסורין, ושקול כנגד כולם. וכשאדם בעת צרה, יקבל על עצמו הכנעה עצומה בלב אמת, יראה פלאות ישועת השם בהרף עין, והוא הכנעה המבררת הסיגים והרע מן האדם בעת עיבור וקטנות, ונמשך הדעת משיח בן דוד בבחינתו.‏

ובימי הגלות המר שהוא בחינות עיבור, נתגלה אור התורה ורזין פליאין ונשגבים, מה שלא היה בהתגלות כל כך בזמן המקדש, ובנקל להשיג השגות רוח הקודש ומראות עליונות, כמו לתינוק בימי עיבור.‏

ואחר כך באין ימי היניקה, ללמוד תורה, 'דדיה ירוך בכל עת' (משלי ה, יט), תורה לשמה, שורש המאיר לנשמות, ואז צריך הבדלה בין סם חיים לסם המות, כי בכל אות יש בחינות נוקבא, כטעם 'נשיו הטו את לבבו' (מ"א יא, ד), כנודע ממרן הריב"ש טוב. וזאת התורה יכוין להעלות כל כ"ב אתוון מתי"ו עד אל"ף, פלא עליון, לאור זין, 'אור זרוע לצדיק' (תהלים צז, יא), צנור הימיני.‏

והענין, כי כל מעשה בראשית, הן דבר קטן הן דבר גדול, נברא על ידי אותיות, ועל ידי פגם ירדו נשמות קדושות שהם אותיות הקודש ממקום חיותן אל מקום הקליפות, ונתלבשו בכמה לבושין חומריים, דומם צומח חי מדבר רע ומדבר טוב, ובתיקון האדם מעשיו ולומד לשמה היינו לשם האות בעצמה, להעלות האותיות שנפלו בעמקי הקליפות, על ידי שלומד בדביקות באהבה ויראה בחיות אלהות, ועל ידי החיות הקדושה מסלק החיות מן הקליפה, ומבדיל בין טוב לרע, ומעלה האותיות והנשמות מעמקי הקליפות עד למעלה למעלה.‏

והנה אות שנפלה בקליפה, ותלוי בזה האות כמה בריות ודברים הרבה כטעם 'נפל אות מתורתך' (תהלים קיט, יח), והוא מעלה את הכל לשורש האותיות לשורש הקדושה, ומעלה כמה אלפים ורבבות ענינים התלוים בזה האות, וזהו זוכה לדברים הרבה (אבות ו, א), כיון שמעלה כל המדריגות וענינים ששורש בריאתן מזה האות, עד למעלה לשורשן, זוכה בהם אף בגשמיות כסף וזהב ובתים, כל אחד לפי ענינו, ויש בקרקעות, כטעם 'אוהב אדמה היה' (דה"ב כו, י), שהעלה בחינה זאת לשורשה. והסימן של תורה לשמה אמר הרמב"ן, כשנתמלא אהבה ויראה מוחין דביקות נועם זיו השכינה, סימן שעלה לרצון.‏

ולפעמים צריך לירד כדי להעלות, כטעם קבעת עתים לתורה (שבת לא.), וביאור מרן הריב"ש טוב מאמר כפה עליהם ההר (שם פח.), ללמד שאף בעת שנסתלק החיות והתורה הוא עליו למשא כבד, יקבל על עצמו עול תורה, כהאי תורא דיהבין עליה עול בקדמיתא בגין לאפקא מניה טב לעלמא, ואי לא קביל עליה האי עול, לא עביד מידי (זוהר ח"ב קח.).‏

ועול מלכות שמים ויראה הוא העיקר, כי אינו בן חורין להיבטל ממנה (אבות ב, טז), רק יעשה בעל כרחו, ויקבע ויגזול עתים לתורה, כמו מי שכופין אותו בעל כרחו, והוא דרך טוב לאיש הישראלי, שלא יבטל מתורה ותפלה אף בימי הקטנות והחשך, דמכל מקום עושה מעשה, ונקרא עבד, ועקימות שפתיו הוה מעשה (ב"מ צ:).‏

כטעם 'כי טוב חסדך' (תהלים סג, ד), זהו החסד שעשית עמדי טוב, 'מחיים שפתי ישבחונך', בשפתים בעלמא בדבור פה בעלמא, גם זה לשפל כמוני חסד גדול, ואחר כך ביקש, יהי רצון שאזכה 'כן אברכך בחיי', עם חיות אלהות, 'כמו חלב ודשן תשבע נפשי'. אימתי 'תאות לבו נתת לו' (שם כא, ג), בזמן 'וארשת שפתיו בל מנעת'. אבל אותן ההולכין בגדלות לא יזכו לעולם לאור הטוב והחיים, אבל ההולך בהדרגה הכנעה הבדלה, בא לבחינות המתקה, 'מתוקים מדבש ונופת צוף' (שם יט, יא), עייל ונפיק בלא בר ובבר לבב.‏

# מצד אמו היה יכול ללכת במקומות חשוכים

## נועם מגדים פרשת פנחס

ומעתה נפתחו לנו שערי אורה להבין ולהשכיל פירוש המאמר המוצב בראש דברינו, דהנה כבר כתבנו דבפסוק שמע בני, יש להעיר דאצל אביך, נקט לשון שמיעה ואצל אם, נקט לשון שלילה אל תטוש. וגם מה זה שחלק מוסר אב ותורת אם, אטו שני תורות הם הלא תורה אחת לכולם. אך יתכן לפרש הכתוב בדרך נפלא לדרכינו, דהנה אמרו רבותינו ז"ל (ב"ק לח ע"ב) שתי פרידות טובות יש לי להוציא מהם רות המואביה ונעמה העמונית ונעמה הלזו היא אשתו של שהמע"ה אמה של רחבעם כידוע. והנה מדהזהיר הקדוש ברוך הוא אל עמון לא תצורם בעבורה. בודאי כי זכותה היה רב ועצום וצדקת גדולה היתה:

ובזה יבואר הכתוב ויתכן שהמלך שע"ה אמר הכתוב הלז לרחבעם בנו ויבא שמע בני כמשמעו ממש על נכון. והכוונה דבתחלה הזהירו שמע בני מוסר אביך, היינו התורה שאני מלמדך להועיל כדברי הזוהר הקדוש מוסר אביך דא אורייתא ואל תסור ממנה ימין ושמאל. אך בהיותך בדרך זה לקיים את כל דברי התורה לבלתי עבור מאחת סמהנה ה"ז הדרך של הצדיק הגמור בלתי בעל תשובה. אם כן איפוא תחשוב שאסור לך הליכה לדרך דעבירה כמ"ש משמי' דהרב כלי יקר אין צדיקים גמורים יכולים לעמוד בו לזה אמר לו דבאמת מצידי אסור לך הדבר הזה אך מירושת אמך משרי שרי לך הדבר הזה דהא גיורת הואי ותברא חשוכא ולא יגרע מצדיק עינו. וזהו ואל תטוש כו', גם שתשמע מוסר אביך ותהיה צדיק ולא בעל תשובה על כל זה הותרה הרצועה לילך למקום חשוכא לתברא חשוכא. וזהו תורת אמך, שהורידה אור זה ומורה תורה לך שתוכל לזרוח בחושך אור והבן היטב כי זה נחמד להשכיל בפירוש הכתוב:

# איך שלמה נשא את נעמה

## עמק המלך – שער יד – פרק קיב

יום הכיפורים אין רצונו של הקב"ה לשמוע קטרוג על ישראל, והוא אוהב משפט, הוא נותן עצה בשעיר המשתלח ל'עזאזל', והוא מלה מורכבת מעזא ועזאל, ומחלק אותם, כדי שיגיע מיתוק לאחד שלא יקטרג, וכיון שלא יהיה לעזאל מענה מעזא, לא יפעל עזאל מאומה. כמו להבדיל בין טמא לטהור, נצח והוד לא יפעול אחד, בלי חבירו, כדפירשנו בסוד היניקה, ומקויים בם 'יתפרדו כל פועלי און'. וצריך שתדע, כי כשכבש יהושע את הארץ שלח ב' אנשים מרגלים, והיו מגבורות עזא ועזאל שנמתקו בכבישת הארץ, ולכן כששלחם הלכו בית אשה זונה, שהיא הקליפה הגדולה, והקשה למתקה. ובהיות שתכף כשנכנסו שם נמתקה, לכן אמרה 'נפלה אימתכם עלינו', מפני כי היו באים הגבורות במתוק, ולכן 'ולא קמה עוד רוח באיש' [יהושע ב' ט י"א]. ואמרו רז"ל, [זבחים קט"ז ב'] כי באו שרי יריחו לשכב עמה, מפני שכבר נמתקה צד הגבורה, והטעם הוא, כי דינין דדכורא דקיק, ותכף הם נמתקים, ולכן לא 'קם עוד רוח באיש', ואמרו רז"ל [כנראה שחסר פה], אבל דינין דנוקבא בהיות כי הם קשים, לא נמתקו עד זמן שלמה המלך ע"ה, וזהו סוד 'אז תבאנה שתים נשים זונות' [מלכים א' ג' ט"ז], פירוש אז נמתקו ולא עד עתה, עיין בספרא דצניעותא בסוד בני אלהים [ספרא דצניעותא. ביאור הגר"א פ"ד ד"ה ויראו בני. "שלא נתבסם העולם עד שלמה" זוהר חלק ב' קע"ח ב'] עד דאתא יהושע, 'בנות האדם' דהיינו 'תענוגות בני האדם שדה ושדות' [קהלת ב' ח'], שהם נשים זונות של שני מרגלים ששלח, וגם באדרת נשא [זוהר במדבר קמ"ג א] והוא סוד גדול, ובהיות כי עדיין לא הגיע הזמן לגמרי, עד לעתיד לבוא יגיע מיתוק לנקיבות הקשות, לכן 'נשיו [של שלמה] הטו את לבבו' [מלכים א' י"א ד']. ולכן יבוא משיח על ה'חמור' כדי להכניע הקליפה הקשה מצד הזכר, הנקרא 'חמור', כי מצד הנקיבה נקראת אתון. וזהו סוד 'ושעשע יונק על חר פתן' [ישעיהו י"א ח'], שהוא הקליפה, ופירושו, 'ושעשע יונק' שהוא מטטרון שנקרא נער, על חור פתן שהוא הקליפה, מפני שכבר נמתקה, 'ועל מאורת צפעוני' שהיא הנקיבה שנקראת לילית, בהיות שכבר נמתקה, 'גמול ידו הדה'. ולעתיד לבוא כשמתמתקת כל הקליפה, אז כח הטהרה גובר, ונראית כמעט קליפה כאלו היא טהרה. וזהו סוד המאמר, 'לעתיד לבוא הקב"ה מביא ליצר הרע ושוחטו, בפני הצדיקים ובפני הרשעים, צדיקים נראה להם כהר גבוה' מפני שאינם רואים אלא טהרה שבה, ואינם רואים צד הקליפה שבה, והם בוכים מרוב השמחה , 'ורשעים נראה להם כחוט השערה' [סוכה נ"ב א'] מפני שאינם רואים אלא קליפה. פירוש רואים איך נתמעטה הקליפה, ואינם רואים צד הקדושה שבה כמו הצדיקים, אלא צד הקליפה שבה לבד, ולכן רואים אותה כחוט השערה מפני שכבר נמתקה, ולכן בוכים איך לא יכלו לגבור עליה, כי כל כוונות שצריך האדם לכוין ולעשות במעשיו הטובים, הוא להכניע הקליפה. ובזה תבין למה שלמה המלך ע"ה לקח את בת פרעה לו לאשה, אבל כוונתו היה לכוף הקליפה הקשה, כי היא בת מלך, ובהיות כי עדיין לא הגיע הזמן לא יכול עמה, אלא 'נשיו הטו את לבבו' [מ"א י"א ד]. ומפני העבירה הזאת שעבר על שלשה לאוין שבתורה, ד'לא ירבה לו סוסים, לא ירבה לו נשים, ולא ירבה לו כסף וזהב', נגזרה עליו לטרוד ממלכותו ג' שנים לבקש על הפתחים, והיה בדלי דלות, ושם השיג את נעמה העמונית וגיירה, ומפני שאין מקבלין גרים בימי שלמה [יבמות כ"ד ב'], נעמה נתקבלה בזמן הג' שנים שהיה בדלי דלות ובעניות. ולא חזר למלכותו אלא בזכותה, כדאיתא באגדה. שלמה היה שט בכל יום לרקיע לשמוע סודות מפי עזא ועזאל, ולא היה לו פחד ואימה, וכל צבא מעלה היו כורעים ומשתחוים לפני הקב"ה, והיו משבחים אותו שהעמיד מלך כזה בישראל, והיו ממלאים לו כל רצונו, שנאמר, 'וישב שלמה על כסא ה' למלך', והיה מולך על העליונים והתחתונים. גזר והביאו אבנים וצרכי בנין לבית המקדש, וכשבקש השמיר היה מביא אשמדאי 'מלכא דשדים' בשלשלאות של ברזל וטבעת שעליו שחקוק עליו שם המפורש [גיטין ס"ח ב'], והיה תפוס אצלו ימים רבים אפילו אחר בנין בית המקדש. וכשגרמה החטא בקש שיתיר אותו ויגלה סוד אחד, שהוא ענין גדול ונורא מאוד, והתירו ושאל ממנו שיתן לו טבעתו שהיה חקוק בו שם המפורש, והוא האמינו כי היתה סבה מאת ה' לשלם לו כפעלו, על שעבר ג' לאוין שילך ג' שנים בגלות. וכשלקח הטבעת השליכו לים ובא דג אחד ובלע אותו, אז השליכו ארבע מאות פרסאות בארץ העמים, וגרשו מעל מלכותו ואבד כל תפארתו, כי השליכו למרחוק והיה שואל על הפתחים, ואמר 'אני שלמה הייתי מלך בירושלים'. והיו משחקים מדבורו והיו אומרים מלך כזה ישאל על הפתחים, ועמד בזה הצער ג' שנים, על שעבר ג' מצות התורה, על 'לא ירבה לו נשים ולא ירבה לו סוסים וכסף וזהב לא ירבה לו מאוד', ונכשל בכולם. ובסוף הזמן הזה בסוף השלש שנים, רצה הקב"ה לרחם עליו למען דוד עבדו, ולמען הצדקת נעמה בת מלך עמון שיצא ממנה משיח בן דוד, שיתחבר אליה ויביאנה עמו לארץ ישראל. והביאו הקב"ה לארץ עמון ובא לעיר המלוכה שמה משכמם, והיה עומד ברחוב העיר משכמם, ומלצר המלך שר הטבחים, שעושה ומבשל מאכל המלך, בא לקחת מה שצריך לעשות למלאכתו, ומצא שם את שלמה עומד שם, ולקח אותו באנגריא, שהוליך מה שלקח, והוליכו לבית התבשיל וראה מה שהוא עושה. ואמר שלמה שיעמוד עמו וישרתיהו, ואינו רוצה אלא המאכל בלבד, ונתרצה וישב עמו וישרתיהו ויעזרהו, אחר איזה איזה ימים אמר לו שיבשל למלך מאכלים כמנהגו, כמו שהוא היה אומן גדול באלו המאכלים, ונתרצה לו המלצר, והוא עשה ובשל מעדני המלך. וכשאכל המלך אותם המטעמים שהביאם המלצר וטעם אותם התבשילין, אמר למלצר, ושאל אותו מי בישל אלו המאכלים, שעד עתה לא הבאת לי כהם, וסיפר לו כל <ק"ט א> המאורע שאותו האדם בשלם. וצוה לעבדיו שיקראו אותו, ובא לפני המלך, ואמר לו אם רוצה אתה להיות לי מלצר, ואמר הן, ושלח את המלצר מלפניו ושם אותו במקומו, ושמש לו ובשל לו כל מאכליו. ויהי אחר הדברים האלה ראתה אותו בת מלך עמון, שמה נעמה, ואמרה לאמה שרצונה לקחת את האיש הזה המלצר לבעל, ואמה גערה בה, ואמרה לה יש במלכות אביך שרים רבים ונכבדים שתקח אחד מהם הטוב בעיניך, ואמרה אינני רוצה אלא זה המלצר. והפצירה בה הרבה ולא הועיל, שאמרה על כל פנים אין רצוני באיש אחר אלא זה, עד שהוכרחה אמה לגלות הענין לבעלה להמלך, שרצון בתו היא ליקח את המלצר לאיש. וכששמע זה הדבר, חרה לו זה עד מאוד, ורצה להרוג את שניהם, אבל לא היה רצונו של הקב"ה, ונעשה להם נס שנכמרו רחמי המלך עליהם, ולא רצה לשפוך דם נקי, וקרא לאחד מעבדיו וצוהו שיוליך אותם למדבר שמם ושם ימותו מאליהם. והסריס עשה כאשר צוהו המלך והניחם במדבר והלך לדרכו אל המלך לשרתו כבראשונה. והם הלכו משם למצוא אוכל להחיות את נפשם, ובאו לעיר אחת שהיתה על שפת הים, והוא הלך לבקש אוכל לנפשם, ומצא מוכרים דגים, וקנה אחד מהם והביא את הדג הזה לאשתו שתבשל אותו, וכשפתחה אותו הדג מצאה הטבעת בתוכו שחקוק עליה שם המפורש, ונתנה את הטבעת הזה לבעלה. ותכף הכיר את הטבעת הזה ושם אותו באצבעו ותכף שב רוחו אליו ודעתו נתישבה עליו ועלה לירושלים וגרש את אשמדאי, וישב על כסא מלכותו ושם כתר מלכות בראשו. ואחר כך שלח לקרוא לאביה שהוא מלך בני עמון, ואמר לו למה הרגת שני נפשות בלי רשות, ופחד ואמר חס ושלום לא הרגתים אלא גרשתים למדבר שמם, ולא ידעתי מה נעשה בהם, אמר שלמה המלך ע"ה ואם תראה אותם יוכל אתה להכיר אותם, דע שאני המלצר ובתך אשתי, ושלח לקראתה ובאה ונשקה ידיו, והוא שמח שמחה גדולה והלך לו אל ארצו. [אוצר המדרשים אייזנשטיין חלק ב' עמוד תק"ל מדרש נשכח המובא על ידי ילניק]. ואחר זה המעשה לא לקח במחשבתו עוד לשוטט עוד אל פני רקיע השמים כבראשונה, ונפל עליו פחד ואימה עד שהיה מתירא אפילו על מטתו בלילה, שנאמר, 'הנה מטתו שלשלמה ששים גבורים סביב לה מגבורי ישראל' [שיר ג' ז']. ולא מלך על העליונים כבראשונה, ונפגמה הלבנה, עד יערה רוח ממרום לנחמנו על ידי משיח בן דוד, להעלות אותה בחידושה אל מקומה הראשון, שנאמר, 'אלכה ואשובה אל אישי הראשון כי טוב לי אז מעתה' [הושע ב' ט'], ואז היא עולה עמו ואינה יורדת. וזהו שנאמר על שלמה ע"ה, 'אשר בהעוותו והוכחתיו בשבט אנשים ובנגעי בני אדם' [שמואל ב' ז' י"ד], 'בשבט אנשים' זהו שנאמר, 'ויקם אלהים לו שטן [לשלמה] את רזון בן אלידע' [מלכים א' י"א כ"ג], 'ובנגעי בני אדם' זהו השלכתו על ידי אשמדאי מלכא דשדים. וזהו מחלוקת רב ושמואל, 'חד אמר מלך והדיוט ומלך, שהיה חוזר על כל פנים למלכותו אחר ג' שנים, וחד אמר מלך והדיוט ולא שב למלכות העליונים [סנהדרין כ' ב'] כמו שנאמר בראשונה, 'וישב שלמה על כסא ה' למלך' [דברי הימים א' כ"ט כ"ג], אותו כסא ה' לא היה לו באחרונה, ולא חזר אליו. ומר אמר חדא ומר אמר חדא ולא פליגי, כי חד סובר ומדבר על מלוכת התחתונים שחזר אליו, וחד אומר במלוכת העליונים שלא חזר עוד אליו, ואינו מזכיר מלוכת התחתונים, כי בזה גם כן הוא מודה שחזר אליו, והלא נאמר במלכים, 'והימים אשר מלך... על כל ישראל ארבעים שנה, וישכב שלמה עם אבותיו' [מלכים א' י"א מ"ב מ"ג], אלא מחוורתא כדאמרן. ואחר כך נתגלגל בירמיהו כשסיהרא בפגימותה, ולכן אמר 'אני הגבר ראה עני בשבט עברתו' [איכה ג' א'], ואין לו למי שהוא, כל כך צרה, כמו לאדם שרואה אותה בשלותה, לכן אמר משלמה קאתי, ממלכת שבא אחרי שגיירה, אלא שלבה לא נכון עמה, ולבסוף גלתה נבלותה וחזרה לסורה, לכן רחקו בניה מישע. אם כן לא על חנם בקש דניאל להשיאו עצה טובה באמרו 'מלכי ישפר עליך, וחטאך בצדקה פרק' [דניאל ד' כ"ד], ונענש על זה, כי סוף סוף ולדה כמותה. ועל דא קרא 'ומלך מלכיא' ואף על פי שבמסכת שבועות [ל"ה ב'] דרשוהו כלפי מעלה, מכל מקום אין מקרא יוצא מידי פשוטו, שכן שלמה ישב על כסא ה', ונובלות מגדלותו, נתנה לו לפי שעה, לנבוכדנצר, בשכר שיחה נאה וארבע פסיעות כד הוה ספריה דמרודך בלאדן [סנהדרין צ"ו א']. ולא יתואר הקב"ה ב'מלך מלכיא' בלבד, אלא מלך מלכי המלכים, ובארנוהו, ולא מצינו לאחד מגדולי המלכים שנקרא בפי הנביאים מלך מלכיא זולתו [יחזקאל כ"ו ז'], כי הנה סנחריב, קראו הושע מלך שרים, אבל לא מלך מלכים. אבל נבוכדנצר נתנבא עליו ירמיהו שהמליכוהו מן השמים על כל הגוים ועל הממלכות, גוים רבים ומלכים עצומים, וכתיב, 'וגם את חית השדה נתתי לו לעבדו' [ירמיהו כ"ז ו']. ומהרבוי הזה נתמעטו 'שבעה עדנין' [דניאל ד' י"ג] בשני רבויין הללו, כי באו למעט בגדרי ה'נתינה' שלא היתה לעבדו כל אותן השנים, וריבה בנתינה אפילו אותן השנים לשמרו מן הנזקין והמיתה, שנדמה לכל מין שהוא מינו [תנחומא וארא ט'] ונשמר מזה שלא הזיקוהו, עד שחזר ליקר מלכותו והדרו וזיוו. כל מלך באומות לא דמה אליו בגדלו, אמנם משלו נתן לו הקב"ה לנבוכדנצר, זהבה של ירושלים שהרבה ממנו נתנה אמו סבתא, למלך שלמה. ולא נמסר בעונו של שלמה אלא בידו, ומזליה חזי, לחוס הרבה על ירמיה, כדמפורש בכתוב, שצוה עליו רב טבחים 'קחנו ועיניך שים עליו' [ירמיהו ל"ט י"ב]. הנה בקש דוד להחליש כחו לא להאבידו, כי בו יכופר עון יעקב, שהוא האריה החצוני, שהכה דוד המלך ע"ה שנאמר, 'גם את הארי גם הדוב הכה עבדך' [שמואל א' י"ז ל"ו]. ולפי גדולתו מפלתו, כי בתחלה הוכה ולא מת, ובסופו 'והכרתי לבבל שם ושאר ונין ונכד' [ישעיה י"ד כ"ב]. וממה שהוכה הארי ולא מת יוקח מופת וראיה, היא נפלאת בעינינו, למוסר ירמיהו שאמר, 'כי אם הכיתם כל חיל כשדים הנלחמים אתכם ונשארו בם אנשים מדוקרים' דייקא, מכח ההכאה 'איש באהלו יקומו ושרפו את העיר הזאת באש' [ירמיהו ל"ז י'] ולבסוף נתקיימה בה נבואת ישעיהו, שנהפכה כמהפכת סדום ועמורה. יצא לנו מנדון דידן, כי שלמה המלך היה חייב גלות, כדי לאסוף את נעמה העמונית, לא בשלותו, כמו שנעמי אשת אלימלך, אספה אליה את רות המואביה, בשעת עניה ומרודיה. ואז כתיב, 'ורות דבקה בה' [רות א' י"ד] דוקא לא בשלותה, כי 'המר שדי לי מאוד' [שם א' כ'], וכן מאלף נשים שגייר שלמה לא נתקבלה מכולן להעמיד ממנה ממלכת ישראל, אלא נעמה העמונית שנשאת לו כשהיה בדלי דלות, לקיים מה שאמרו חז"ל במסכת יבמות על פסוק 'מי גר אתך בעניותך הוא דעליך יפול' [כ"ד ב']. וכשם שגלתה נעמי לאסוף את רות, כך גלה שלמה לאסוף את נעמה, היא נעם ה' עלאה, אשר שם רחובות הנהר, כאשר אבאר. לפיכך נקרא בנה רחבעם, כי בקש הקב"ה לעשותו משיח, וירבעם יהיה לו למשנה, שכנגדו במרום, דוד עילאה ושר העולם מטטרון, רק הכסא דבריאה יגדל ממנו דאמר מר אין ישיבה בעזרה [יומא ס"ט ב'] דבריאה, אשר שם ירושלים של מעלה, אלא למלכי בית דוד בלבד. וראוי היה שירחיב ה' לבו של רחבעם בשביל עם הקודש, שיהיה 'מורח ודאין' [סנהדרין צ"ג ב'], 'והריחו ביראת ה" [ישעיהו י"א ג'] אלמלא זכינו, וכשצפה בו דוד ברוח הקודש, אמר עליו ועל עצמו, 'רבות עשית אתה ה' אלהי, נפלאתיך ומחשבתיך אלינו' [תהילים מ' ו'], שאסר עמון ומואב והתיר בנותיהן. ובו ביום נולדה אמו של רחבעם, ואותה עריסה נגלה לדוד ברוח הקודש [יבמות ע"ז א'. רחבעם נולד שנה לפני ששלמה מלך. ולכן "רחבעם יושב בחיקו של דוד"] ואמר שירה עליה, וזכתה יותר מרות שנקראת אם המלך סתם. כדאיתא בפרק המוכר את הספינה, על פסוק, 'וישם כסא לאם המלך' [בבא בתרא צ"א ב'], ונעמה היא אם המשיח לדעת חז"ל, ושמה יעיד על נועם ה' אלהינו, כשם שנקראה רות על שם ש'יצא ממנה דוד שריוהו להקב"ה בשירות ותשבחות' [ברכות ז' ב']. וכשרצה לתקן כל ישראל בימי מורינו ורבינו קודש קדשים האר"י זלה"ה, אחר שהלך ירמיהו בגלות שכינה בכל מקום, ומחדש קינות בכל יום על ירושלים עיר הקודש, ועל קב"ה ושכינתיה. בא באחד מתלמידיו, ושמו מורינו ורבינו הרב רבי אברהם הלוי זלה"ה, שהיה חסיד קדוש ופרוש כל ימי היותו באדמת עיר הקודש בקהילת קודש צפת תוב"ב, והוא היה כל ימיו אפוטרופוס גדול לשבת, מכריז ומזהיר את ישראל תמיד אנשים ונשים על שמירתו. וכה משפטו כל ימי חייו, מידי שבת בשבתו לכתת רגליו בשווקים וברחובות בבתים ובחצירות, לילה ויום מחצות ליל ששי, לזרז את העם שיכינו את אשר יביאו, ושתהא אשה משכמת ואופה, ושיהיו נועלין החנויות סמוך למנחה. כדי שתהא פנאי ביום, להתקדש לקראת שבת כלה מלכתא, בטבילה וברחיצת פניו ידיו ורגליו בחמין, וחלוף שמלותיו, ובקריאת הפרשה שנים מקרא ואחד תרגום. וידוע שירמיהו תמיד היה מוכיח את ישראל שיחזרו בתשובה [י"ז כ"א], ולא יחרב בית המקדש, אף על פי שעשו לו עינויים, והושיבוהו אותו בתוך הבור בטיט, עם כל זאת לא הניח מלהוכיח את ישראל. וכן תיקון שבת היא מביאה הגאולה, כדברי חז"ל, אלמלא שמרו ישראל שני שבתות כהלכתן מיד נגאלין [שבת קי"ח ב'], כנודע בתיקון שבת, כי בעליה תלייא מלתא, ואם יהיה עת רצון שיעלו העולמות מעלה, שהיתה בימי דוד ושלמה, אז יגבר הטוב על הרע, כי המקום גורם והדינים והקליפות ימתקו, 'לא ירעו ולא ישחיתו בכל הר קדשי' [ישעיהו י"א ט']. וכנודע גם כן שרצה האר"י <ק"ט ב> זלה"ה להביא הגאולה בזמן קבלת שבת, כי יש לנו קבלה אמתיות שבאותה העת רצון, תהיה הגאולה במהרה בימינו. וכן היתה המעשה, פעם אחת בערב שבת סמוך להכנסת כלה, יצא עם תלמידיו חוץ לעיר צפת לבוש ד' בגדי שבת, דהיינו, היילריק ושופיצא [מעיל בית דק ועליו מעיל עליון] וכתונת ומכנסים כדי לקבל השבת. והתחיל 'מזמור לדוד הבו ליי' בני אלים', ופזמון מיוסד לקבלת השבת, ו'מזמור שיר ליום השבת', ומזמור 'י"י מלך' בניגון נאה, לסבב העיר בסוד יריחו, וסוד הקפת המזבח ביום הערבה. שהוא להכרית מעיר ירושלים של מעלה האויבים, שנאמר, 'סביב רשעים יתהלכון' [תהילים י"ב ט'], ונאמר, 'יהמו ככלב ויסובבו עיר' [תהילים נ"ט ז'], ובזה נכרת כל אחיזת הקליפות מהשכינה כנודע, וכך הרב ע"ה היה מסבב עיר צפת תוב"ב בסוד 'לכה דודי נצא השדה' בשירים נאים. ובתוך שהיו משוררים אמר הרב ע"ה לתלמידיו, חברים רצונכם שנלך לירושלים, קודם השבת, ונעשה שבת בירושלים, וירושלים היא רחוקה מצפת יותר מכ"ה פרסאות, השיבו קצת מן התלמידים אנו שומעים בכך, רצוננו לילך לשם, וקצתם השיבו ואמרו נלך מקודם ונודיע את הדבר לנשינו. כיון שאמרו נלך מקודם לבתינו, נתחרד הרב חרדה גדולה, והכה כף אל כף, ואמר אוי לנו שלא היה לנו זכות להגאל, שאלמלא הייתם כולכם משיבים בפה אחד, שרצונכם לילך בשמחה גדולה, תכף היו נגאלים כל ישראל, שעתה היתה השעה עומדת להגאל, ומתוך שמאנתם בדבר חזר הגלות לאיתנו בעונותינו הרבים. וידוע שמפני אותה העת רצון בא האר"י זלה"ה לעולם, והוא נשמת משיח בן יוסף, כנודע בביאור, בשער גלגול נשמות, נמצא שתיקון שבת היא מביאה את הגאולה. גם היה הזקן מוהר"ר אברהם הלוי ע"ה אשכנזי, דורש ומזהיר על התשובה, מעשים המעוררים בני אדם לתשובה, כאשר אבאר בשער התשובה, והיה שוכב עצמו במטה של חרולים ערום, ואמר שמי שרוצה להנצל מעונש שריפה יעשה כך, וצוה לתלמידיו שיזרוקו עליו אבנים, ואמר כך ינצל הבן אדם מעונש סקילה וכן כולם דוגמתו. והוידוי יועיל יותר, שנזכר בו ד' מיתות בית דין על דרך הסוד כאשר אכתוב בשער הנזכר, ונעשה התעוררות גדול על ידו, עד שמצאו לפני בית הכנסת הגדול של האיטליאני, אדם אחד יושב בשק ג' ימים וג' לילות בבכייה רבה. וכן היה מקדים והולך בכל חצות לילה, ודופק על פתח הבתים של ק"ק צפת, וקורא עליהם, עמדו לתיקון חצות, ללמוד בעת רצון הזאת, ולא היה מאמין שיעמוד, עד ששמע אותו, שלמד מה ששכלו נותן, הן מקרא ומשנה ותלמוד, ולא היה חצות לילה עד שנשמע קול לומדי תורה עד למרחוק. ובתוך כך חלה אותו הצדיק והגיע עד שערי מות, ונכנס אצלו רבו האר"י זלה"ה לבקרו ואמר לו רצונך שתלך לירושלים ותתפלל כנגד כותל מערבי תתרפא מחוליך. מיד נדר לילך לירושלים עיר הקודש, ותכף נתרפא מחליו, ואז מכר מטלטלי הבית לצורך הוצאות הדרך, כי היה עני גדול. והלך לירושלים ובהגיעו כנגד כותל מערבי, היה קורע כדין הרואה בית מקדש בחרבנו. ותכף נתגלתה עליו שכינה, שראה אותה כביכול כמו שראה אותה ירמיהו הנביא הולכת מבית קודשי הקדשים ראשה פרועה, כדאיתא במדרש איכה [פסיקתא רבתי פרשה כ"ז], וגם על זה נתקנה קינה 'אז במלאת ספק', ומה טוב ומה נעים שיאמר אותה, גדול ורב שבעיר. וכיון שראה אותה רבי אברהם זלה"ה בצער גדול כזה, אז הרים קולו בבכי וצעק צעקה גדולה ומרה, והיה רץ מכח רוב הצער בראשו כנגד בית אחד אשר שם, והכה ראשו בכותלי הבית, ונתעלף מרוב צער, ונפל לארץ. ואז ראה שהשכינה לקחה ראשו בין ברכיה, וקנחה את דמעות עיניו ואמרה לו, אברהם בני התנחם 'יש תקוה לאחריתך ושבו בנים לגבולם' [ירמיהו ל"א ט"ז], ואחר זה שב אל דעתו כבראשונה, וחזר לצפת תוב"ב. ותכף שאל אותו רבו, ראית כזה וכזה ? והיה אומר הן, ואמר לו את כל זה ראית בזכות התשובה שנתעוררה על ידך, ובזכות זה הוסיפו לך מן השמים כ"ב שנה, בעבור השכינה היא תורה שבעל פה, הנבנית מכ"ב אותיות, ובעבור נשמת ירמיהו הנביא ע"ה אשר היא עמך. רק חזק ואמץ ביראת ה' להוסיף בהתעוררות התשובה לבני העיר ובזכות חדא כנישתא יכנש גלוותא [זוהר חדש סוף נח], ונשמתו של ירמיהו היא הועילה לך לעזרך ולסייעך באלו המצות, וזהו סוד אמרם ז"ל, 'אלמלא הקב"ה שעוזר אינו יכול לו' [סוכה נ"ב ב']. עוד נדבר בענין סוד רות המואביה ונעמה העמונית מה ענין ניצוצם, ושם יתבארו לך סוד יניקת הקליפות עם כמה סודות של פסוק, 'על כן יאמרו המושלים בואו חשבון תבנה ותכונן עיר סיחון' [במדבר כ"א כ"ז], עם פסוק 'רשפי אש שלהבת יה' [שיר ח' ו'], ועל ידו יתבארו לך כמה מאמרים של הזוהר, בעזרת קב"ה ושכינתיה יתברך יי' לעולם אמן ואמן:

# רחבעם היה אמור להיות משיח

## עמק המלך – שער טז – פרק כד

'דבר טוב אל י"י' כי המשיח הזה יהיה בן טוב אל י"י שנאמר, 'י"י אמר אלי, בני אתה אני היום ילדתיך' [תהילים ב' ז'] וזהו סוד אומרם ז"ל שבקש הקב"ה לעשות רחבעם משיח, וירבעם יהיה לו למשנה, שכנגדם במרום, דוד עילאה ושר העולם, רק הכסא יגדל ממנו, דאמר מר אין ישיבה בעזרה אלא למלכות בית דוד לבד [יומא כ"ה א']. והמשיח לקח נשמת ירבעם, שהלכה ממנו קדם שחטא, ועליה לא נגזר העונש המיועד לבית ירבעם, כמו שפרחה נשמת אדם הראשון ממנו בזמן מחשבת החטא, והיא זיהרא עילאה, וזכה בה מטטרון הוא חנוך, ולכן לא נגזר עליו המיתה כמו על אדם הראשון ובניו:

ואלו השני מלכים אי אפשר להם להשתמש בכתר אחד, המשנה יצא ויבוא [רק] על פי בן דוד, כמו מטטרון הנקרא יהו"ה קטן [זוהר רעיא מהימנא במדבר פינחס רכ"ג ב'], והוא אח היבום כביכול, בסוד הגלות. וכל זה בא מיסוד עולם התוהו, בסוד 'מלכות בית דוד פליגי', שנחלקה שפע המלכות לשנים, ושפע מלכות היה מושאל ליסוד לפי שעה, בזמן שנקרא חי, כדי שלא ישיגה מלכות שבירה, בסוד מלך שאול ויהונתן בנו. שנאמר 'ואנכי אהיה לך למשנה וגם שאול אבי יודע כן' [שמואל א' כ"ג י"ז] בהסוד הזה, לכן היה רודף אחריו, כמו השפע שהיה רודף לשבור כלי המלכות בית דוד, ומי שיש לו לב להבין, יבין הסוד של הדורות הללו, על מתכונתם. ישמע חכם יוסיף לקח, יגל יעקב ישמח ישראל. יתברך י"י לעולם אמן ואמן:

## מגלה עמוקות על התורה – פרשת תולדות

בזוהר פ' וירא שהקב"ה אמר לאברהם ויצחק שילכו לקבל אולפין דצדיקייא בשעת פטירת הצדיק והם אינם רוצים לילך עד שיעקב הולך לדרכו ואח"כ באין תרין משריין דאברהם ויצחק ויפגעו בו מלאכי אלהים הנה הן ג' בארות שעל משריין דאברהם קראו עשק כי התעשקו עמו ואינם רוצים לילך עמו גם משיריית' דיצחק שטנה אבל משריית' דיעקב רחובות כי שם הרחיב ד' לנו והולך עמו אח"כ ויעל משם באר שבע לקביל שבע היכלין דכסיפין ושבע ברכאין דבריך הקב"ה לאברהם ג' בארות ע"ס שהידית וישב יצחק בגרר לגייר את הגרים כטעם יצחק בן אברהם שעשה כמו שעשה אביו ומלא מקומו וכבר נתברך יצחק בעקידה והתברכו בזרעך כל גויי הארץ שהוא לשון הברכה והשפלה שמבריכין גוף א' בארץ. (איוב יד) גם אם יזקין בארץ ישריש שרשו פתוח עלי מים ג' אלו שסתמום פלשתים עפר ג' בארות הם נפש רוח נשמה הראשון נקרא עשק השני שטנה כי בעשיה שם מדורין בישין נפש נקרא עשק [לקביל עולם היצירה ורצה ליכנס בבריאה מקום כה"ג תמן רוח] בריאה תמן נשמה שאין הקליפה כ"כ רק שהם מעורבת ע'ופרת ש'ל ק'ין אותו עשק איתמר גבי קרח (קהלת ז) כי העשק יהולל חכם שבעה לקביל עולם היצירה ורצה לכנוס בבריאה מקום כה"ג תמן רוח בריאה תמן נשמה נקראת רחובות הנהר אימא עילאה מקננת בכורסי' גם מתמן (בראשית ל') שאול מרחובות הנהר (שמואל ט) אשר הי' משכמו ומעלה גבוה מכל עם וגם אם זכה רחבעם הי' משיח וירבעם משנה ושר העולם:

# סוד רחבעם וירבעם

## טל אורות חלק ראשון פרק שבעה עשר אות ז'

והנה שלשה ראשונים שהם כח"ב, באשר שהם קרובים אל המאציל שכל שלשה כלבוד דמי והיה בהם שורש חזק כי כל אחד מהם נבנה משתי אותיות ע"י יו"ד האומנית שנחשב לערכם למאציל (עיין בע"ח היכל שביעי ש"א) והיה אורם זך מאוד, לכן היו יכולים לקבל גם החלק התחתונים שכולם בטלים לערכם. אבל הז"ת שנתרחקו יותר מג' מדריגות ואין בהם שורש חזק כי כולם נבנו רק מהארת שתי אותיות בלבד, וגם הז"ת כולם שוים בכמות ואין אחד גדול מחבירו כי לכל אחד מיוחד פעולה מיוחדת, לכן כשירד כל האור שהם ק"מ ריבוא אורות אל הגדולה והוא חשב לקבלם למלוך על האחים שלמטה ממנו היינו שאלו דלמטה יזונו על ידו ולפי שעה נחשב לו למלוכה, אך לא היה לו כח לסבלם ונשבר הכלי שלו, ועליו נאמר 'וימלוך באדום בלע בן בעור', ואח"כ מיד כתיב בתריה 'וימת כו' וימלוך תחתיו יובב', הוא גבורה שקבל ג"כ המלוכה לפי שעה, ואח"כ וימת, וכן כולם. ומה שקבלו כל אחד ואחד האורות ממה שלמעלה אחר שראה שבירת הראשון, הוא מחמת שמלך תחתיו, ר"ל דתדע שלאחר שנשבר כלי החסד נשארו האורות השייכים לחסד במקום שהיה עומד כלי החסד והשאר ירד למטה, והיה חושב כל אחד בדעתו שאחר שנתמעט האור וגם הוא תחתיו יותר בריחוק מקום, יהיה יכול לקבל השאר. ועל כל זה, מאחר שרצה לקבל מה שאינו שלו ורצה להתנשא על חבירו לכן כולהו לא אתקיימו, לבד מבעל חנן הרומז לספירת היסוד, דנודע דספירת המלכות לית לה מגרמה כלום, שכל האורות יש להם צינור מכמה ספירות שמקבלים השפעתם ויש להם ג"כ צינור מה שמקבלים מהמאציל בלי אמצעי כמו שנכתוב כל זה לקמן, אך ספירת המלכות אינה מקבלת רק מצינור אחד המיוחד לה והוא מספירת היסוד, ולכן היסוד מקבל לעולם שני חלקים חלקו וחלק המלכות. ולכן כשבאו כל האורות אל היסוד, לא קבל את כולם בבת אחת, אלא קבל תחילה חלק המיוחד למלכות ונתן אותו בפקדון (בסוד הפס' 'והיה האוכל לפקדון') כחצי חלק התחתון שלו, כי בכלי שלו יש בו סוד לחם משנה יומיים, והיה בו סוד מלכות פליגא בסוד רחבעם וירבעם, ולפי שעה נתקיים מלכותו והוא המלך השביעי הנקרא בעל חנן בסוד 'צדיק חונן ונותן'.‏

# רחבעם משורש לוט והבל

## שער הגלגולים – הקדמה לג

ואמנם צריכים אנו לבאר ענין שמואל, ולכן נתחיל נדב ואביהוא, דע, כי כמו שנדב ואביהוא חטאו במעשה הקטרת וגם ישראל שחטאו במעשה העגל, גרמו להם לישרף, כנז"ל בפסוק ואחיכם כל בית ישראל יבכו את השריפה אשר שרף ה'. כן אהרן גרם מיתתם, לפי שהוא עשה העגל ומתו בסבתו, כמש"ה ובאהרן התאנף ה' מאד להשמידו, זה כלוי בנים וכו'. ואמנם אהרן הכהן, הוא משרש הבל בן אדה"ר, כמו שנתבאר בדרושים שקדמו, כי שרש הבל נחלק לכמה בחי' שרשים. והנה שני שרשי נשמות, יש בשרשי הבל, והם, שרש הרן אחי אברם. ושרש נחור. והנה נחור, נתגלגל בחור בנה של מרים. והרן, נתגלגל באהרן הכהן. וכבר הודעתיך, כי חו"ר לקח ג' אותיות מן נחור, ואות נ' של נחור נשארה לשרש אחאב מלך ישראל, כמבואר אצלינו ואמנם אהרן, ג' אותיות מן הרן, ונוסף עליו אות א'. וגם לוט בן הרן, משרש הבל, ולכן ר"ת ואת "לוט "בן "הרן, הוא הבל למפרע, כי לוט והרן שניהם משרש הבל, ומן לוט נמשך רחבעם המלך, כמבואר אצלינו. ונמצא כי מרע"ה, ואהרן אחיו, וחור בן מרים, ולוט, ואחאב, כלם משרש הבל בן אדם, רק שהם שרשים מחולקים זה מזה, וכלם משרש הבל:

# המלך והמשנה למלך – רחבעם וירבעם, דוד ויהונתן, לאה ורחל

## עשרה מאמרות – מאמר חקור דין – חלק ד' פרק יז

והמשכילים יזהירו עם מה שאמר יהונתן לדוד אתה תמלוך על כל ישראל ואנכי אהיה לך למשנה וגם שאול אבי יודע כן כי הידיעה לא תאמר אלא בדברים האמתיים העתידים להתקיים על כל פנים וגם שאול כשפייס את דוד אחרי כרתו את כנף המעיל במערה לאות שלא רצה לחבול בו אמר לו ידעתי כי מלך תמלוך וקמה בידך ממלכת ישראל והן שלש מלכיות לאדם לדוד ולמשיח כי ביד משיח לבדו תתקיים ממלכת ישראל כאמרם ז"ל שפירושו ממלכת שפלה היא חלק המשנה אתמול אלא אצל יום חמישי ראשון למפרע שבו היו מסדרים דבריהם הרי חמש שנים ואמרם בהיות שאול מלך עלינו יוכללו שם בהבלעה גם שנתים של איש בשת אע"פ שבאותן שתי שנים לא היה דוד מוציא ומביא רק שבט יהודה והוא טעם בלבול אותיות של מסורת בשתי מלות הללו אבל בימי שאול כתיב להדיא כי דוד היה לו שר אלף לו דוקא קרוב למלכות ויצא ויבא לפני העם ולעולם יש הבדל בחמש משפחות הללו בין איש בשת לדוד שכלן קבלו עליהם את איש בשת תוך שתי שנים ודוד לא קבלוהו כל ישראל אלא בתוך חמש שנים אחרות כדאמרן ולענין ראשון אע"פ שדרשו על פסוק קום משחהו כי זה הוא זה טעון משיחה ואין אחר טעון משיחה לא נאמר אלא על הזמן העתיד למעט שאר מלכי ישראל שלא היו כשרים ואין שמן המשחה מתכבד בהם אף המה לא רצו להיות סרים אל משמעתם של מלכי בית דוד על כן החזיקו כלם בחטאת ירבעם והיתה עם זה קלקלתם תקנתם להתכפר גם הם בשעיר ויש בפסוק קום משחהו כי זה הוא מצות עשה לשמואל שיקום מפני דוד אף קודם משיחתו שכן כתיב גבי ישי ראיתי בבניו לי מלך פי' מלך הוא לי מתחלת המחשבה וכן זה הוא מעקרא אבל שאול כשמו כן הוא בהשאלה לא בקביעות זה יעצור בעמי דרך עראי ועוד למדנו מפסוק זה שמצות משיחה בשמן הקדש אינה אלא מעומד למושח ולנמשח ויש בראשי תיבות אלה קם כזה כי בשעה שנמשח דוד למטה דוד עלאה אית ליה לקב"ה שקם גם הוא למעלה עוד קם כזה השתא דאתית להכי להחליף מם דקרא במם דמנצפ"ך הנה עם הכולל הוא בגי' חמשה שמות כוללים האצילות אריך אבא אמא זעיר נוקבא שכלם שמחו לקראת דוד יעויין מאמר מאה קשיטה סי' צ"ג וצ"ד:

אך המשנה משפטו שיהיה מבניה של רחל דאתמר ביה עבד מלך מלך ומאז ראויה היתה מלכות שמים להתגלות עלינו בלי הפסק בסוד מלך מלכי המלכים אלו ישראל ממלכת כהנים וגוי קדוש ומלכיהם שנים הנשיא שאין על גביו אלא ה' אלקיו והמשנה המוציא והמביא את ישראל כי ה' עליון נורא מלך גדול על כלם הוא הדבר הנאמר לשאול בתחלת המרותו את פי נביא שאם היה שומר מצוה כי עתה הכין ה' את ממלכתך אל ישראל עד עולם דייקא נמי דכתיב ממלכתך ולא מלכותך אל ישראל ולא על ישראל וכן הובטח ירבעם בשם ה' אלמלא מרה ועזב את רוח קדשו ובניתי לך בית נאמן כאשר בניתי לדוד דאי לא הא לא קיימא הא אם אין מלך אין משנה ואם אין משנה נכנע למרבה המשרה זה וזה לא יעלה כהוגן והדוגמא למעלה אריך על אבא ואמא דבדבורא אסתיים וכלם על זעיר ונוקביה מקור ישראל וכתיב עיניו כיונים אלו הסנהדרין עיני העדה רוחצות בחלב יושבות על מלאת כאן רמז רוח הקדש בר"ת אלה רחבעם וירבעם לפיכך באה מלת רחצות בחסרון וא"ו ראשונה ושניהם יחד נרמזו עד"ז בסוף פסוק ויהי בישרון מלך:

וראוי היה לנו להקדים בנין בית המקדש בירושלים לפריקת משכן שילה כפי הדין המפורש לבית הכנסת בפ' בני העיר לא ליסתור איניש בי כנישתא עד דבני אחריתי אך גרם החטא אי דגרים ושילה שהיה ביניהם ג' מילין בלבד ועשן פסלו של מיכה מתערב עם עשן המערכה של קדש ומימי עתניאל בן קנז התחיל סרחון זה ואז אירע הפורענות של פלגש בגבעה ולא פסק פסלו של מיכה ימים רבים כתוב א' אומר עד יום גלות הארץ ופירשו המתרגם ורש"י עד גלות ראשונה שהיא היתה גלות שלמה ע"י סנחריב לשני המטות וחצי המטה וכתוב אחד אומר כל ימי היות בית האלקים בשילה הרי אלה שני כתובים דסמיכי אהדדי ונראים מכחישים זה את זה ולא נאמנו דברי נביאות שכתב הרלב"ג ואחרים בדרכים שונים מסברת עצמם לא נאבה ולא נשמע אליהם אכן לראשונים אנו שומעים שדבריהם דברי קבלה ופשטיה דקרא מסייעם ונראים הדברים שבזמן ראשון דכתיב אין חזון נפרץ וכתיב אין מלך בישראל היה הפסל נעבד בפומבי ואחרי שהוחזק שמואל הרמתי נביא ונאמן לה' והעמיד מלכים כשרים בישראל וגם שאול התאמץ במלכותו והכרית את האובות ואת הידעונים מן הארץ לא היה נעבד פסלו של מיכה אלא בהחבא סוף דבר פרסומו של פסל מתחלת הקמתו שהיה קרוב לבנין שילה והתמדתו בפומבי כל ימי היות המשכן על עמדו הם גרמו חרבן שילה לפני זמנו וימאס באהל יוסף הואיל ומיכה נמי מניה קאתי מהר אפרים וביום גלות הארץ גלות ראשונה לתגלת פלאסר הוא אחד משבעה שמות של סנחריב גלו ראובן וגד ומנשה דאתי מיוסף ומעשה הפלגש ג"כ עשה פרצה בבנימין הן הם שני בניה של רחל דכתיב בה ויחר אף יעקב ברחל ועשה רושם גם למהר חרבן שילה כדאמרן, ונוסף על זה עון בית עלי שגרם אח"כ מיתת כהני נוב וגברו עשר קליפין עד שראו חכמים שבאותו הדור לצמצם עשרה מילין מתחומי שבת זכר עשו לאותה קלקלה הואיל וביום שבת בא דוד אל בית אחימלך ונשארו שני מילין על שני מילין ומקום השביתה באמצען דוקא שאפילו הוא באנטוכיא ובמערת צדקיהו לעולם אינו מן המדה אלא מסויים לבדו באמצע התחומין ויש רמז להגבלתם חוצה לו בשעור הממועט הזה בפסוק וגם נצח ישראל לא ישקר בסודו המבואר במקומו בשני דרכים יעויין שם. ...
