\documentclass[12pt]{article}

% פונטים ו־RTL
\usepackage{fontspec}
\usepackage{polyglossia}

\newfontfamily\headingfont{David}
\newfontfamily\quotefont{David}

\usepackage{geometry}         % שוליים
\geometry{margin=2.5cm}

% כדי שפנדוק לא יקרוס ברשימות

% עיצוב ציטוטים
\usepackage{etoolbox}
\AtBeginEnvironment{quote}{\quotefont\itshape\small}

\begin{document}

$endif$

$if(author)$
{\headingfont\Large $author$\par}
$endif$

$if(date)$

$endif$
