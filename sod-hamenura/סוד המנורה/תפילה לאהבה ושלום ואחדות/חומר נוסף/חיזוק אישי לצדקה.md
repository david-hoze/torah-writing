# ליקו"ה יו"ד הל' ריבית ראשי פרקים ה"ה

וּגְמִילַת חֶסֶד גָּדוֹל יוֹתֵר מִן הַצְּדָקָה, כִּי גְּמִילַת חֶסֶד בֵּין לַעֲנִיִּים בֵּין לַעֲשִׁירִים, כִּי צְדָקָה אֵינוֹ נוֹתֵן לוֹ כִּי אִם דֵּי מַחְסֹרוֹ אֲשֶׁר יֶחְסַר לוֹ, כִּי אֵין עַתָּה מִצְוָה לְעַשְּׁרוֹ, אֲבָל גְּמִילוּת חֶסֶד וְהַלְוָאָה בָּזֶה מַחֲזִיק בְּיָדוֹ וְיָכוֹל לְעַשְּׁרוֹ עַל יְדֵי זֶה. וְהָעִקָּר הוּא הַהַמְתָּנָה כְּשֶׁהַמַּלְוֶה הַגּוֹמֵל חֶסֶד מַמְתִּין לְהַלּוֹוֶה אֵיזֶה זְמַן עַד אֲשֶׁר יִתְחַזֵּק יָדוֹ וִיסַלֵּק לוֹ, כִּי זֶה עִקַּר בִּטּוּל הַשִּׁנּוּיִים הַתְּלוּיִים בַּזְּמַן, כִּי בֶּאֱמֶת אֵין זְמַן כְּלָל, רַק שֶׁאִי אֶפְשָׁר לְהָבִין זֹאת כְּדֵי שֶׁיִּהְיֶה בְּחִירָה עַל יְדֵי שִׁנּוּי הַזְּמַן. וְכָל הַתּוֹרָה וְהַמִּצְוָה הוּא שֶׁאָנוּ מַאֲמִינִים בְּבִטּוּל הַזְּמַן שֶׁיִּתְחַדֵּשׁ הָעוֹלָם וְנִהְיֶה בִּבְחִינַת לְמַעְלָה מֵהַזְּמַן. וְזֶה נִזְכֶּה עַל יְדֵי שֶׁאָנוּ מַאֲמִינִים בַּה' יִתְבָּרַךְ וּמְקַיְּמִין הַתּוֹרָה וְהַמִּצְווֹת שֶׁהֵם הַיּוֹם לַעֲשוֹתָם בַּזְּמַן וּלְהַאֲמִין לַה' יִתְבָּרַךְ וּלְהַמְתִּין בְּקַבָּלַת הַשָֹּכָר עַד לְמָחָר, שֶׁהוּא עוֹלָם הַבָּא, שֶׁהוּא לְמַעְלָה מֵהַזְּמַן. וְזֶה עִקַּר מַעֲלַת הַצְּדָקָה וְגַם שֶׁהוּא כְּמוֹ מַלְוֶה לַה' יִתְבָּרַךְ, כְּמוֹ שֶׁנֶּאֱמַר, "מַלְוֶה ה' חוֹנֵן דָּל". וְעִקָּר הוּא הַהַמְתָּנָה שֶׁמַּמְתִּין לַה' יִתְבָּרַךְ, כִּי בְּוַדַּאי אִם הָיָה הָאָדָם רוֹאֶה בְּעֵינָיו שֶׁעַל יְדֵי הַדִּינָר שֶׁנּוֹתֵן לְהֶעָנִי נִמְשָׁךְ לוֹ מִיָּד שֶׁפַע בִּמְקוֹמוֹ אֶלֶף אֲדוּמִים, מִי פֶּתִי שֶׁלֹּא יָרוּץ בְּכָל כֹּחוֹ לִתֵּן צְדָקָה?! וּבֶאֱמֶת אֵין זֶה צְדָקָה, רַק חִלּוּף, שֶׁמַּחֲלִיף דִּינַר כֶּסֶף בְּאַלְפֵי דִּינְרֵי זָהָב. וַהֲלֹא מִתְיַגֵּעַ וְנוֹשֵֹא וְנוֹתֵן עַל סָפֵק רֶוַח חֵלֶק עֲשִֹירִית מֵהַקֶּרֶן וְלִפְעָמִים יֵשׁ הֶפְסֵד. וּמִי אֲשֶׁר יִהְיֶה כְּסִיל שֶׁלֹּא יָרוּץ לְמַשָֹּא וּמַתָּן כָּזֶה?! וְאִם כֵּן לֹא הָיָה בְּחִירָה כְּלָל. וְעַל כֵּן עִקַּר הַבְּחִירָה הוּא עַל יְדֵי הַהַמְתָּנָה וְהַהַעֲלָמָה, הַיְנוּ מֵחֲמַת שֶׁנֶּעְלָם מִבְּנֵי אָדָם, כִּי הֵם סְתִימִין עֵינִין וְאֵינָם רוֹאִין שֶׁתֵּכֶף נִמְשָׁךְ הַשֶּׁפַע בְּכִפְלֵי כִּפְלַיִם לַהֶעָשִׁיר הַנּוֹתֵן, וְגַם הַדָּבָר צָרִיךְ לְהִתְמַהֲמֵהַּ כְּפִי הָעִנְיָן וְהַבְּחִינוֹת עַל כֵּן יֵשׁ בְּחִירָה, אֲבָל הָאָדָם נִבְרָא שֶׁיְּשַׁבֵּר הַבְּחִירָה וּלְבַטֵּל רְצוֹנוֹ וְלִתֵּן צְדָקָה וְלַעֲשוֹת גְּמִילוּת חֶסֶד וְהַלְוָאַת חֵן וּלְהַאֲמִין לַה' יִתְבָּרַךְ לְהַמְתִּין עַד בּוֹא שְכָרוֹ. נִמְצָא, שֶׁעִקָּר הוּא הַהַמְתָּנָה, וְעִקַּר הַכֹּחַ לְהַמְתִּין בְּכָל הַדְּבָרִים וּלְקַוּוֹת לִישׁוּעַת ה' וְשֶׁלֹּא יְקַפַּח שְֹכַר כָּל בְּרִיָּה הוּא עַל יְדֵי שֶׁמְּבַטְּלִין כְּסִילוּת דַּעְתּוֹ וּמַאֲמִינִים בִּבְחִינַת לְמַעְלָה מֵהַזְּמַן, כִּי הַבַּר דַּעַת רָאוּי לְהָבִין בְּדַעְתּוֹ שֶׁצְּרִיכִין לְהַאֲמִין בָּזֶה כַּאֲשֶׁר הִסְבִּיר לָנוּ הוּא, זִכְרוֹנוֹ לִבְרָכָה, בְּהַתּוֹרָה הַנַּ"ל עַל פָּסוּק, "אֲנִי הַיּוֹם יְלִדְתִּיךָ וְכוּ'". וְעַל כֵּן אָסְרָה לָנוּ הַתּוֹרָה לִקַּח רִבִּית שֶׁל הַלְוָאָה וְהוּא עָווֹן חָמוּר מְאֹד, כִּי מֵאַחַר שֶׁרוֹצֶה, חַס וְשָׁלוֹם, לִקַּח רִבִּית, שֶׁהוּא אֲגַר נְטַר, שֶׁנּוֹטֵל רִבִּית בִּשְׁבִיל הַמְתָּנַת הַזְּמַן, זֶהוּ בְּחִינַת פְּגַם כָּל הַתּוֹרָה כֻּלָּהּ, כִּי בֶּאֱמֶת אֵין שׁוּם זְמַן כְּלָל. וְעִקַּר אֲחִיזַת הַזְּמַן נֶאֱחָז בְּתַאֲוַת מָמוֹן שֶׁהוּא בְּחִינַת כְּסִילוּת, בִּבְחִינַת וְלֵב כְּסִיל לִשְֹמֹאלוֹ, שֶׁהוּא בְּחִינַת מָמוֹן שֶׁהוּא בִּשְֹמֹאל, כְּמוֹ שֶׁנֶּאֱמַר (מִשְׁלֵי ג, טז), "בִּשְֹמֹאלָהּ עֹשֶׁר וְכוּ'", הָרוֹצֶה לְהַעֲשִׁיר יַצְפִּין, כְּמוֹ שֶׁמְּבֹאָר בְּמָקוֹם אַחֵר. וְעַל כֵּן שָׁם עִקַּר הַתִּקּוּן, דְּהַיְנוּ כְּשֶׁמְּשַׁבֵּר תַּאֲוַת מָמוֹן שֶׁעִקָּרוֹ עַל יְדֵי צְדָקָה וְהַלְוָאַת חֶסֶד. וְעִקַּר הַתִּקּוּן עַל יְדֵי שֶׁמְּבַטֵּל הַזְּמַן מֵאַחַר שֶׁמַּאֲמִין בַּה' יִתְבָּרַךְ וּמַמְתִּין כַּנַּ"ל. וְעַל כֵּן זֶה הַמַּלְוֶה הָאַכְזָר שֶׁאֵינוֹ רוֹצֶה לְהַמְתִּין וְדוֹחֵק וְנוֹגֵשׁ אֶת חֲבֵרוֹ וְלוֹקֵחַ מִמֶּנּוּ רִבִּית שֶׁהוּא שְֹכַר הַמְתָּנָה, הוּא פּוֹגֵם מְאֹד בְּכָל הַתּוֹרָה כֻּלָּהּ שֶׁתְּלוּיָה בְּהַהַמְתָּנָה בְּחִינַת הַיּוֹם לַעֲשוֹתָם וְכוּ', כִּי הַהַמְתָּנָה הִיא בְּחִינַת בִּטּוּל הַזְּמַן, שֶׁעַל יְדֵי זֶה מִתְבַּטְּלִים כָּל הַשִּׁנּוּיִים וְכַנַּ"ל וְעַל כֵּן אָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה; שֶׁהַלּוֹקֵחַ רִבִּית כְּכוֹפֵר בְּכָל הַתּוֹרָה כֻּלָּהּ וְכַנַּ"ל וְזֶהוּ וּכְאִלּוּ כּוֹפֵר בִּיצִיאַת מִצְרַיִם שֶׁאָז שִׁדֵּד ה' יִתְבָּרַךְ כָּל הַמַּעֲרָכוֹת שֶׁבָּהֶם תָּלוּי הַזְּמַן, כִּי אָז הָיָה בְּחִינַת בִּטּוּל שִׁנּוּי הַזְּמַן, כִּי אָז הָיָה לַיְלָה כְּיוֹם יָאִיר בְּלֵיל שִׁמּוּרִים וְכוּ'. וְעַל יְדֵי זֶה נִמְשָׁךְ בְּחִינַת קַבָּלַת הַתּוֹרָה שֶׁאָז הָיוּ כֻּלָּם כְּאִישׁ אֶחָד, כְּמוֹ שֶׁנֶּאֱמַר, "וַיִּחַן שָׁם יִשְֹרָאֵל וְכוּ'" לְשׁוֹן יָחִיד, כְּמוֹ שֶׁפֵּרֵשׁ רַשִׁ"י שָׁם. וְהַכֹּל הָיָה עַל יְדֵי מֹשֶׁה שֶׁהָיָה לְמַעְלָה מֵהַזְּמַן, כְּמוֹ שֶׁמְּבֹאָר בְּמָקוֹם אַחֵר:
