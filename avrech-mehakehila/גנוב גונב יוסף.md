# גנוב גונב יוסף

נקדים בהתבטאות של מורנו הרב שליט"א על התפילה האחרונה שביקש להפיץ וללמוד לאהבה ושלום ואחדות; "אם ילמדו את התפילה הזאת בעיון, אולי יבינו למה הרב לא חוזר". לכאורה התפילה מדברת על שלום ואחדות, אז בפשטות אפשר אולי לומר שהרב התכוון שכולנו נהיה באחדות, ואז הוא יחזור. אך קשה לומר זאת, כיון שאם נדקדק בדברי הרב, הרב אמר אם ילמדו את התפילה בעיון, אולי יבינו למה הרב לא חוזר, והרי ענין השלום כתוב כבר בכותרת.. לכן מכון סוד המנורה וכולל כנישתא חדא לקח על עצמו ללמוד את התפילה בעיון, והתוצאה מוצגת כאן לפניכם. לפעמים כשמכבירים בתיאורים מאבדים את הנקודה, אך כשלמדנו את התפילה והגענו למסקנות נדהמנו מעומק כוונת הרב בדבריו, עומק והיקף התפילה שבחר שנלמד, ופלאי ההשגחה שהובילה אותנו לאן שהובילה. על כן ראוי לגשת לדברים ברצינות, כובד ראש, וחרדת קודש, מפאת חשיבותם.

# קישור הרצונות לשורש הרצון

מסביר הליקו"ה שגניבה באה לעולם כיון שלא מקשרים את הרצונות לשורש הרצון:

## ליקו"ה חו"מ הל' שומר שכר ה"ב אות ט'

וְזֶה בְּחִינַת גְּנֵבָה וַאֲבֵדָה וְאוֹנְסִין שֶׁיֵּשׁ חִלּוּקֵי דִּינִים בֵּינֵיהֶם לְעִנְיַן שׁוֹמֵר שָֹכָר וְשׁוֹאֵל כַּנַּ"ל, כִּי שְׁנֵי מִינֵי הֶזֵּקוֹת אֵלּוּ הַבָּאִים מִפְּגַם הָרָצוֹן כַּנַּ"ל הֵם נִמְשָׁכִים חַס וְשָׁלוֹם כְּפִי הַפְּגָם בָּרָצוֹן:

כִּי מְבֹאָר בְּהַתּוֹרָה הַנַּ"ל שֶׁיֵּשׁ חִלּוּק בֵּין הָאֶפִּיקוֹרְסִים, כִּי יֵשׁ אֶפִּיקוֹרְסִים שֶׁנִּקְרָאִים בְּחִינַת חַיּוֹת רָעוֹת שֶׁהֵם הָאֶפִּיקוֹרְסִים לַהֲנָאָתָם, כִּי יֵשׁ לָהֶם הֲנָאָה מִזֶּה לְהַחֲזִיק דַּעַת וְשֶׁקֶר שֶׁלָּהֶם וְכוּ'. אֲבָל יֵשׁ אֶפִּיקוֹרוֹס שֶׁהוּא בִּבְחִינַת מֵצַח הַנָּחָשׁ, שֶׁהוּא רָשָׁע וְכוֹפֵר בְּלִי שׁוּם הֲנָאָה וְכוּ'. וּמֵאֵלּוּ שְׁנֵי מִינֵי אֶפִּיקוֹרְסִים נִמְשָׁכִין שְׁנֵי מִינֵי הֶזֵּקוֹת שֶׁהֵם גְּנֵבָה וַאֲבֵדָה וְאוֹנְסִין. כִּי יֵשׁ אֶפִּיקוֹרְסִים שֶׁנָּפְלוּ לְאֶפִּיקוֹרְסוּת שֶׁלָּהֶם עַל יְדֵי כַּמָּה טָעֻיּוֹת שֶׁהִטְעָהוּ הַבַּעַל דָּבָר עַל יְדֵי רִבּוּי תַּאֲוֹתָיו ... וְזֶה בְּחִינַת גְּנֵבָה וַאֲבֵדָה, ... וְעַל כֵּן מֵאֶפִּיקוֹרְסִים כָּאֵלּוּ יוֹנְקִים בֶּאֱמֶת מַזִּיקֵי עָלְמָא הַגּוֹרְמִים גְּנֵבוֹת וַאֲבֵדוֹת, כִּי מִשָּׁם שָׁרְשָׁם כַּנַּ"ל וְאֵלּוּ הָאֶפִּיקוֹרְסִים נִקְרָאִים חַיּוֹת רָעוֹת הַכּוֹפְרִים בָּרָצוֹן, שֶׁהַכְנָעָתָם עַל יְדֵי הֶחָכָם שֶׁבִּקְדֻשָּׁה שֶׁמְּקַשֵּׁר כָּל הָרְצוֹנוֹת לְשֹׁרֶשׁ הָרָצוֹן שֶׁהוּא בְּחִינַת מֵצַח הָרָצוֹן, בְּחִינַת רַעֲוָא דְּרַעֲוִין שֶׁשָּׁם נִסְתַּלֵּק מֹשֶׁה רַבֵּנוּ עָלָיו הַשָּׁלוֹם וְכוּ', עַיֵּן שָׁם כַּמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל, עַיֵּן שָׁם.

# מכירת יוסף

וממשיך וכותב שזה היה כל ענין מכירת יוסף ומחלוקת האחים על יוסף:

## ליקו"ה חו"מ הל' שומר שכר ה"ב אות יג

וְהַשְּׁבָטִים שֶׁפָּגְמוּ בְּיוֹסֵף שֶׁהוּא בְּחִינַת הִתְגַּלּוּת הָרָצוֹן כַּנַּ"ל, עַל כֵּן כְּתִיב, וְאָמַרְנוּ חַיָּה רָעָה אֲכָלָתְהוּ וְכוּ'. וְכֵן יַעֲקֹב אָמַר, חַיָּה רָעָה אֲכָלָתְהוּ טָרֹף טֹרַף יוֹסֵף, זֶה בְּחִינַת הַחַיּוֹת רָעוֹת הַדּוֹרְסִים וְטוֹרְפִים שֶׁהֵם בְּחִינַת חַכְמֵי הַטֶּבַע שֶׁהֵם פּוֹגְמִים בָּרָצוֹן שֶׁהוּא בְּחִינַת יוֹסֵף, כַּמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל. וְהִנֵּה הַבַּעַל דָּבָר הִתְגָּרָה בָּהֶם אָז וְרָצָה לַהֲסִיתָם שֶׁיַּהַרְגוּ אֶת יוֹסֵף לְגַמְרֵי, כִּי הַבַּעַל דָּבָר הוּא רוֹצֶה לִפְגֹּם וְלִכְפֹּר בְּשֹׁרֶשׁ הָרָצוֹן לְגַמְרֵי חַס וְשָׁלוֹם, אֲבָל ה' יִתְבָּרַךְ חָמַל עֲלֵיהֶם וְעַל כְּלָלִיּוּת הָעוֹלָם וְסִבֵּב בְּרַחֲמָיו שֶׁלֹּא הֲרָגוּהוּ לְגַמְרֵי רַק מָכְרוּ אוֹתוֹ בְּכֶסֶף לְמִצְרַיִם, הַיְנוּ כַּנַּ"ל, כִּי הֵם פָּגְמוּ רַק בִּבְחִינַת גְּנֵבָה וַאֲבֵדָה, כִּי גָּנְבוּ אוֹתוֹ וּמְכָרוּהוּ, כְּמוֹ שֶׁכָּתוּב, כִּי גֻנֹּב גֻּנַּבְתִּי מֵאֶרֶץ הָעִבְרִים וְכוּ'. הַיְנוּ שֶׁהַפְּגָם בָּרָצוֹן בִּבְחִינַת יוֹסֵף לֹא הָיָה רַק בִּבְחִינַת גְּנֵבָה וַאֲבֵדָה כִּי גָּנְבוּ אוֹתוֹ וּמְכָרוּהוּ כְּמוֹ שֶׁכָּתוּב כִּי גֻנֹּב גֻּנַּבְתִּי מֵאֶרֶץ הָעִבְרִים וְכוּ'. הַיְנוּ שֶׁהַפְּגָם בָּרָצוֹן בִּבְחִינַת יוֹסֵף לֹא הָיָה רַק בִּבְחִינַת גְּנֵבָה שֶׁהִוא בְּחִינַת פְּגָם הַנִּמְשָׁךְ עַל יְדֵי חַיּוֹת רָעוֹת שֶׁהֵם חַכְמֵי הַטֶּבַע לַהֲנָאָתָם שֶׁמַּכְנִיסִים טָעוּת בְּלֵב הָאָדָם בִּגְנֵבָה וְהֶסַּח הַדַּעַת, ... אֲבָל לֹא הָרְגוּ אֶת יוֹסֵף לְגַמְרֵי, שֶׁזֶּהוּ בְּחִינַת רוֹצְחִים וְגַזְלָנִים שֶׁהֵם בִּבְחִינַת מֵצַח הַנָּחָשׁ וְהַכּוֹפֵר בְּשֹׁרֶשׁ הָרָצוֹן לְגַמְרֵי. כִּי אִלּוּ הָרְגוּ אֶת יוֹסֵף לֹא הָיָה אֶפְשָׁר עוֹד לְהִתְתַּקֵּן מֵאַחַר שֶׁהָיוּ מְסַלְּקִים חַס וְשָׁלוֹם, שֹׁרֶשׁ הָרָצוֹן מִן הָעוֹלָם, אֲבָל מֵאַחַר שֶׁנִּשְׁאַר יוֹסֵף חַי רַק שֶׁמְּכָרוּהוּ לְמִצְרַיִם, שֶׁזֶּהוּ בְּחִינַת שֶׁלֹּא קִשְּׁרוּ עַצְמָן לְיוֹסֵף לְשֹׁרֶשׁ הָרָצוֹן רַק, אַדְּרַבָּא, מְכָרוּהוּ בְּכֶסֶף בִּגְנֵבָה שֶׁזֶּהוּ בְּחִינַת פְּגַם הַחַיּוֹת רָעוֹת שֶׁהֵם הָאֶפִּיקוֹרְסִים לַהֲנָאָתָם כַּנַּ"ל

אז מדויק מהליקו"ה, שהשורש לכך שהשבטים גנבו (חטפו) את יוסף זה שהם לא קישרו את עצמם אליו, וממילא המשיכו היזק של גניבה שהתלבש במקרה הזה בהם. ומביא הליקו"ה בהמשך שההתקשרות הזאת היא בחינה של קישור הרצונות לשורש הרצון:

## ליקו"ה חו"מ הל' שומר שכר ה"ב אות יג

וְזֶהוּ בְּחִינַת הַמַּחֲלֹקֶת שֶׁל הַשְּׁבָטִים עַל יוֹסֵף הַצַּדִּיק, כִּי יוֹסֵף הַצַּדִּיק הוּא בְּחִינַת הַצַּדִּיק הַזֶּה הַנַּ"ל בְּחִינַת רַעֲוָא דְּרַעֲוִין, בְּחִינַת מֹשֶׁה, כִּי יוֹסֵף מֹשֶׁה דָּוִד כֻּלָּם בְּחִינָה אַחַת שֶׁנִּסְתַּלְּקוּ שְׁלָשְׁתָּם בְּשַׁבָּת בְּמִנְחָה בְּרַעֲוָא דְּרַעֲוִין וְכָל הַשְּׁבָטִים הַקְּדוֹשִׁים אַף עַל פִּי שֶׁכֻּלָּם הָיוּ צַדִּיקִים גְּדוֹלִים וְנוֹרָאִים, אַף עַל פִּי כֵן הָיוּ צְרִיכִין כֻּלָּם לְהִתְקַשֵּׁר אֵלָיו, שֶׁזֶּהוּ בְּחִינַת הִתְקַשְּׁרוּת כָּל הָרְצוֹנוֹת לְשֹׁרֶשׁ הָרָצוֹן שֶׁהוּא בְּחִינַת הַזָּקֵן שֶׁבַּקְּדֻשָּׁה שֶׁהוּא בְּחִינַת יוֹסֵף הַצַּדִּיק, כְּמוֹ שֶׁכָּתוּב, כִּי בֶן זְקֻנִים הוּא לוֹ, בֶן 'זְקֻנִים' דַּיְקָא שֶׁשָּׁרְשׁוֹ בִּבְחִינַת זָקֵן שֶׁבַּקְּדֻשָּׁה שֶׁהוּא בְּחִינַת מֵצַח הָרָצוֹן, בְּחִינַת רַעֲוָא דְּרַעֲוִין כַּנַּ"ל.

יוצא מדברים אלו שאם לא מתקשרים לצדיק, שזהו בחינת התקשרות כל הרצונות לשורש הרצון, ממשיכים היזק של גניבה בעולם, וכך נגרמה מכירת יוסף הצדיק, הצדיק באותו דור. לכן כדי להכניע את האפיקורסים מהם נמשך היזק זה של גניבה בעולם, אנחנו צריכים להתקשר לצדיק, בחינה של לקשר את כל הרצונות לשורש הרצון.

# עיקר ההתקשרות לצדיק

יש כל מיני דרכים להתקשר לצדיק, אך מה העיקר של ההתקשרות לצדיק? כותב הליקו"ה:

## ליקו"ה חו"מ הל' העושה שליח לגבות חובו ה"ג אות יז

וְזֶה בְּחִינַת וּמַתָּנוֹת לָאֶבְיוֹנִים, שֶׁצְּרִיכִין לְהַרְבּוֹת בִּצְדָקָה מְאֹד בְּפוּרִים, כִּי עִקַּר הִתְקַשְּׁרוּת לְהַצַּדִּיק הוּא עַל יְדֵי צְדָקָה שֶׁבָּזֶה אוֹחֵז בְּמִדָּתוֹ שֶׁל צַדִּיק שֶׁהוּא נִקְרָא צַדִּיק עַל שֵׁם הַצְּדָקָה שֶׁמַּשְׁפִּיעַ לַכֹּל בִּבְחִינַת צַדִּיק חוֹנֵן וְנוֹתֵן. וְכָל אֶחָד כְּפִי מַה שֶּׁמַּרְבֶּה בִּצְדָקָה כֵּן הוּא מִתְקַשֵּׁר בִּבְחִינַת צַדִּיק שֶׁמִּמֶּנּוּ הָרוּחַ חַיִּים. ...

# עיקר השלום ע"י הצדיק

ובאמת עיקר השלום, בו עוסקת התפילה של מורנו הרב שליט"א, הוא ע"י הצדיק שיכול לקשר כל הרצונות לשורש הרצון, כפי שכותב הליקו"ה:

## ליקו"ה חו"מ הל' שומר שכר ה"ב אות י'

וְהַכְּלָל, שֶׁעִקַּר הַמַּחֲלֹקֶת מֵחֲמַת שִׁנּוּי הָרְצוֹנוֹת וְכָל זֶה מֵחֲמַת שֶׁאֵין מְקַשְּׁרִין כָּל הָרְצוֹנוֹת לְשֹׁרֶשׁ הָרָצוֹן שֶׁהוּא רָצוֹן שֶׁבָּרְצוֹנוֹת, כִּי בֶּאֱמֶת רָאוּי לָדַעַת שֶׁכָּל הָרְצוֹנוֹת שֶׁל כָּל בְּנֵי הָאָדָם אַף עַל פִּי שֶׁיֵּשׁ בֵּינֵיהֶם שִׁנּוּיִים גְּדוֹלִים, אַף עַל פִּי כֵן כֻּלָּם נִמְשָׁכִין מֵרָצוֹן הָעֶלְיוֹן שֶׁאֵין בּוֹ שׁוּם שִׁנּוּי שֶׁהוּא בְּחִינַת רָצוֹן שֶׁבָּרְצוֹנוֹת, רַעֲוָא דְּרַעֲוִין. וּמִי שֶׁיּוֹדֵעַ וּמַאֲמִין בָּזֶה בִּשְׁלֵמוּת בְּוַדַּאי לֹא יַחֲזִיק בְּמַחֲלֹקֶת כִּי יָדִין אֶת חֲבֵרוֹ לְכַף זְכוּת תָּמִיד, כִּי יֹאמַר בְּלִבּוֹ אַף עַל פִּי שֶׁאֲנִי יוֹדֵעַ בְּעַצְמִי שֶׁכַּוָּנָתִי לְשֵׁם שָׁמַיִם לִגְמֹר מִצְוָה הַזֹּאת וְחֶפְצֵי שָׁמַיִם בֶּאֱמֶת וַחֲבֵרִי מְעַכֵּב מִזֶּה, אַף עַל פִּי כֵן גַּם כַּוָּנָתוֹ לְשֵׁם שָׁמַיִם, כִּי יֵשׁ לוֹ דֵּעָה אַחֶרֶת וְנִדְמֶה לוֹ שֶׁאֵין זֶה מִצְוָה כְּלָל אוֹ צְרִיכִין לַעֲסֹק בְּמִצְוָה אַחֶרֶת הַגְּדוֹלָה מִזֹּאת לְפִי דַּעְתּוֹ, וְאַף עַל פִּי שֶׁאֲנִי חָזָק בְּדַעְתִּי מְאֹד וְיֵשׁ לִי רְאָיוֹת וְהוֹכָחוֹת אֲמִתִּיּוֹת שֶׁצְּרִיכִין לְהַחֲזִיק בְּיוֹתֵר בְּדֶרֶךְ זֶה וּבַמִּצְוָה הַזֹּאת אַף עַל פִּי כֵן אֲפִלּוּ אִם הָאֱמֶת הִוא כְּדַעְתִּי. מִי יוֹדֵעַ מֵאֵיזֶה שֹׁרֶשׁ נִמְשָׁךְ חֲבֵרִי אֶפְשָׁר לְפִי שֹׁרֶשׁ נִשְׁמָתוֹ הוּא צָרִיךְ לְהִתְנַהֵג בְּדֶרֶךְ אַחֵר כִּי בֶּאֱמֶת כָּל הַדֵּעוֹת וְהָרְצוֹנוֹת נִמְשָׁכִין מֵרָצוֹן הָעֶלְיוֹן רַק שֶׁאִי אֶפְשָׁר לְהָבִין זֹאת. וְאִם כָּל אֶחָד יֹאמַר בְּלִבּוֹ כָּךְ בְּוַדַּאי יִתְבַּטֵּל הַמַּחֲלֹקֶת, כִּי עִקַּר הַמַּחֲלֹקֶת הוּא מֵחֲמַת שֶׁאֵין מְקַשְּׁרִין כָּל הָרְצוֹנוֹת לָרָצוֹן הָעֶלְיוֹן שֶׁשָּׁם כֻּלּוֹ חַד וְשָׁם כֻּלּוֹ שָׁלוֹם כִּי הַכֹּל נִכְלָל בְּאֶחָד. וְעַל כֵּן עִקַּר הַשָּׁלוֹם הוּא עַל יְדֵי הַצַּדִּיק הַדּוֹר הַגָּדוֹל בְּמַעֲלָה שֶׁהוּא בִּבְחִינַת מֹשֶׁה שֶׁנִּשְׁמָתוֹ מִבְּחִינַת רַעֲוָא דְּרַעֲוִין, רָצוֹן שֶׁבָּרְצוֹנוֹת שֶׁהוּא יוֹדֵעַ רְצוֹן כָּל אֶחָד וְאֶחָד מֵהֵיכָן הוּא נִמְשָׁךְ וְעַל כֵּן הוּא אִישׁ אֲשֶׁר רוּחַ בּוֹ שֶׁיּוֹדֵעַ לַהֲלֹךְ נֶגֶד רוּחוֹ שֶׁל כָּל אֶחָד וְאֶחָד כִּי יוֹדֵעַ רוּחוֹ וּרְצוֹנוֹ שֶׁל כָּל אֶחָד וְאֶחָד וְיוֹדֵעַ לְקַשֵּׁר וּלְיַחֵד רוּחוֹ וּרְצוֹנוֹ שֶׁל כָּל אֶחָד וְאֶחָד בְּשֹׁרֶשׁ הָרָצוֹן וְשָׁם מְיַחֵד וּמְקַשֵּׁר כֻּלָּם בְּרָצוֹן אֶחָד הָעֶלְיוֹן. וְאָז נִמְשָׁךְ אַהֲבָה וְשָׁלוֹם גָּדוֹל וְנִפְלָא בֵּינֵיהֶם מֵאַחַר שֶׁכֻּלָּם בְּדֵעוֹתֵיהֶם וּרְצוֹנוֹתֵיהֶם הַמְשֻׁנּוֹת מְקֻשָּׁרִים וּמְיֻחָדִים בְּרָצוֹן אֶחָד הָעֶלְיוֹן שֶׁאֵין בּוֹ שׁוּם שִׁנּוּי שֶׁהוּא בְּחִינַת רַעֲוָא דְּרַעֲוִין כַּנַּ"ל

# עיקר הצדקה הוא בשביל שלום

בנוסף לכך שעיקר השלום ע"י התקשרות לצדיק, ועיקר ההתקשרות לצדיק ע"י צדקה, השלום בעצמו נמשך ע"י צדקה, ועיקר מעלת הצדקה הוא שלום:

## ליקו"ה יו"ד הל' ריבית ראשי פרקים ה"ה

עַל פִּי מַה שֶּׁמְּבֹאָר בְּסִימָן קעט (לִקּוּטֵי חֵלֶק א). שֶׁלְּבַטֵּל כָּל הַמַּחֲלֹקֶת וְכוּ', הוּא עַל יְדֵי תַּעֲנִיּוֹת בְּחִינַת צְדָקָה וְכוּ', כִּי עִקַּר הַבְּחִירָה עַל יְדֵי הַשִּׁנּוּיִים, עַל יְדֵי שֶׁנִּתְהַוֶּה אֲחִיזַת הַשֶּׁקֶר אַחַר הַבְּרִיאָה וְכוּ', אַךְ עִקַּר אֲחִיזָתוֹ נִתְהַוֶּה רַק עַל יְדֵי רִבּוּי בְּנֵי אָדָם שֶׁיֵּשׁ שִׁנּוּי רְצוֹנוֹת הַרְבֵּה, כִּי כָּל אָדָם מְשֻׁנֶּה דַּעְתּוֹ מֵחֲבֵרוֹ, כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, וְעַל כֵּן עִקַּר אֲחִיזַת הַיֵּצֶר הָרָע נִתְהַוָּה כְּשֶׁנִּבְרְאָה חַוָּה וְנִתְהַוּוּ שְׁנַיִם, כְּמוֹ שֶׁפֵּרֵשׁ רַשִׁ"י עַל פָּסוּק, "הֵמָּה בִּקְּשׁוּ חִשְׁבּוֹנוֹת רַבִּים וְכוּ'", כִּי הַיֵּצֶר הָרָע שֶׁאֲחִיזָתוֹ מֵהַשֶּׁקֶר, שֶׁכָּל שֹׁרֶשׁ שָׁרְשׁוֹ וַאֲחִיזָתוֹ הוּא מִבְּחִינַת רִבּוּי, אֵין לוֹ כֹּחַ כְּנֶגֶד הָאָדָם שֶׁנִּבְרָא בְּצֶלֶם דִּיּוֹקְנוֹ יִתְבָּרַךְ כִּי אִם עַל יְדֵי שֶׁיֵּשׁ עוֹד אָדָם שֶׁיּוּכַל לְהִתְלַבֵּשׁ בּוֹ. וּכְמוֹ שֶׁאָמַר אֲדוֹנֵנוּ, מוֹרֵנוּ וְרַבֵּנוּ זַ"ל; שֶׁהַיֵּצֶר הָרַע אֵין לוֹ כֹּחַ לִמְנֹעַ כְּמוֹ הָאָדָם וְכוּ'...

וְעִקַּר הָעֲנִיּוּת נִמְשָׁךְ מֵאֲחִיזַת הָרַע שֶׁנִּמְשָׁךְ מֵהַשִּׁנּוּיִים כַּנַּ"ל, שֶׁמִּשָּׁם עִכּוּב הַשֶּׁפַע, חַס וְשָׁלוֹם, שֶׁמִּשָּׁם הָעֲנִיּוּת, חַס וְשָׁלוֹם, אַךְ מֵחֲמַת לַהַט הַחֶרֶב הַמִּתְהַפֶּכֶת וְכוּ' שֶׁנִּתְהַוָּה עַל יְדֵי חֵטְא אָדָם הָרִאשׁוֹן וְכוּ'. נִתְהַפֵּךְ הַגַּלְגַּל בְּחִינַת הַטִּיקְלָא עַד שֶׁעַל פִּי רֹב, בִּפְרָט בִּימֵי הַגָּלוּת נִמְשָׁךְ הַשֶּׁפַע אֶל הַחוּץ, חַס וְשָׁלוֹם, וּכְשֵׁרֵי יִשְֹרָאֵל וְהַצַּדִּיקִים סוֹבְלִים עֲנִיּוּת, רַחֲמָנָא לִצְלָן, כִּי הוּא מְחֻלָּל מִפְּשָׁעֵינוּ וְכוּ', אֲבָל עִקַּר שֹׁרֶשׁ הָעֲנִיּוּת נִמְשָׁךְ מֵהַשִּׁנּוּיִים הַנַּ"ל, שֶׁמִּשָּׁם אֲחִיזַת הָרַע וְהַשֶּׁקֶר, שֶׁמִּשָּׁם כָּל הַבְּחִירָה. וְזֶהוּ בְּחִינַת גֹּדֶל מַעֲלַת מִצְוַת צְדָקָה שֶׁשְּׁקוּלָה כְּכָל הַתּוֹרָה כֻּלָּהּ. וְכָל הַתּוֹרָה נִקְרֵאת עַל שֵׁם צְדָקָה, כְּמוֹ שֶׁנֶּאֱמַר, "וּצְדָקָה תִּהְיֶה לָּנוּ וְכוּ'", כִּי עַל יְדֵי צְדָקָה שֶׁמַּשְׁפִּיעַ הֶעָשִׁיר לְהֶעָנִי מְבַטֵּל בְּחִינַת עֲנִיּוּת שֶׁנִּמְשָׁךְ מֵהָרַע הַנֶּאֱחָז בְּהַשִּׁנּוּיִים. וְעַל כֵּן מַרְבֶּה צְדָקָה מַרְבֶּה שָׁלוֹם. וְעַל כֵּן הָעִקָּר הוּא שָׁלוֹם, כִּי לֹא מָצָא הַקָּדוֹשׁ בָּרוּךְ הוּא כְּלִי מַחֲזִיק בְּרָכָה וְכוּ'. וְכָל עִקַּר הִתְגַּבְּרוּת הַיֵּצֶר הָרָע עַל יְדֵי מַחֲלֹקֶת שֶׁמִּשָּׁם כָּל הַכְּפִירוֹת וְכוּ'. וּלְהֵפֶךְ עַל יְדֵי הַשָּׁלוֹם יְכוֹלִים לְהַמְשִׁיךְ כָּל הָעוֹלָם לַעֲבוֹדָתוֹ יִתְבָּרַךְ וְכוּ' וְזֶה נִמְשָׁךְ עַל יְדֵי צְדָקָה, כִּי הָעִקָּר שֶׁהָאָדָם יִתֵּן צְדָקָה, כִּי בְּוַדַּאי ה' יִתְבָּרַךְ הָיָה יָכֹל לְהַשְׁפִּיעַ לְהֶעָנִי בְּעַצְמוֹ, אַךְ כָּל הַבְּרִיאָה הָיְתָה בִּשְׁבִיל הַבְּחִירָה וְעִקַּר הַבְּחִירָה עַל יְדֵי שֶׁנָּתַן לְהָאָדָם כֹּחַ לְעַכֵּב הַשֶּׁפַע, אוֹ לְהַמְשִׁיכָהּ, כִּי כָּל תַּעֲנוּגָיו יִתְבָּרַךְ שֶׁיִּהְיֶה נִמְשָׁךְ הַשֶּׁפַע. וְזֶה בְּחִינַת תִּקּוּן כָּל הַמִּצְווֹת. וּלְהֵפֶךְ פְּגַם כָּל הָעֲבֵרוֹת. וְעַל כֵּן שְֹכַר הַמִּצְווֹת בְּלִי שִׁעוּר, כִּי כָּל מַה שֶּׁמַּשְׁפִּיעִים לוֹ יוֹתֵר מִתְעַנֵּג ה' יִתְבָּרַךְ יוֹתֵר וּמַשְׁפִּיעַ לוֹ עוֹד יוֹתֵר וְיוֹתֵר וְכוּ' וְעַל כֵּן עַיִן לֹא רָאֲתָה וְכוּ'. וְעַל כֵּן בֶּאֱמֶת הַכֹּל מֵאִתּוֹ יִתְבָּרַךְ, כְּמוֹ שֶׁנֶּאֱמַר, "מִי הִקְדִּימַנִי וַאֲשַׁלֵּם", וּכְתִיב, "כִּי מִמְּךָ הַכֹּל וּמִיָּדְךָ נָתַנּוּ לָךְ". רַק הָעִקָּר הוּא הָרָצוֹן, שֶׁיַּכְנִיעַ וִיבַטֵּל שִׁנּוּי רְצוֹנוֹ לְגַבֵּי הֶעָנִי שֶׁנִּמְשָׁךְ עָלָיו הָעֲנִיּוּת עַל יְדֵי שִׁנּוּי רְצוֹנוֹת כַּנַּ"ל. וְזֶהוּ תִּקּוּן כָּל הַתּוֹרָה שֶׁעִקָּרָהּ לַעֲשוֹת רְצוֹנוֹ יִתְבָּרַךְ, כְּמוֹ שֶׁנֶּאֱמַר, "נַחַת רוּחַ לְפָנַי", שֶׁאָמַרְתִּי וְנַעֲשָֹה רְצוֹנִי, כִּי בֶּאֱמֶת אֵין הָאָדָם עוֹשֶֹה כְּלָל, כִּי הַבָּא לִטָּהֵר מְסַיְּעִין לוֹ, רַק הָעִקָּר הוּא הָרָצוֹן, כְּמוֹ שֶׁמְּבֹאָר בְּמָקוֹם אַחֵר בִּעִנְיַן גֹּדֶל הַפְלָגַת מַעֲלַת הָרָצוֹן דִּקְדֻּשָּׁה. וְעִקַּר לְבַטֵּל רְצוֹנוֹ לְגַמְרֵי כְּנֶגֶד רְצוֹן ה' יִתְבָּרַךְ, שֶׁזֶּה זוֹכִין עַל יְדֵי צְדָקָה בְּחִינַת תַּעֲנִית, כִּי עִקַּר מַה שֶּׁיֵּשׁ בְּחִירָה לָאָדָם שֶׁיִּהְיֶה לוֹ אֵיזֶה שִׁנּוּי רָצוֹן, חַס וְשָׁלוֹם, נֶגֶד רְצוֹן ה' יִתְבָּרַךְ, שֶׁזֶּהוּ פְּלִיאָה גְּדוֹלָה אֵיךְ שַׁיָּךְ זֹאת, אַךְ זֶהוּ עִקַּר נִפְלְאוֹת נוֹרְאוֹת הַבְּחִירָה, שֶׁזֶּה הַפֶּלֶא הַגָּדוֹל מִכָּל יְצִירַת מַעֲשֵֹה בְּרֵאשִׁית שֶׁבָּרָא וְיָצַר וְעָשָֹה בָּעֶלְיוֹנִים וּבַתַּחְתּוֹנִים, כְּמוֹ שֶׁמְּבֹאָר בְּמָקוֹם אַחֵר. וְכָל כֹּחַ הַבְּחִירָה נִמְשָׁךְ מֵהַשִּׁנּוּיִים כַּנַּ"ל. שֶׁעַל יְדֵי זֶה נִמְשָׁךְ עַד שֶׁיֵּשׁ לְהָאָדָם שִׁנּוּי רָצוֹן נֶגְדּוֹ יִתְבָּרַךְ, שֶׁמִּשָּׁם נִמְשָׁךְ הָעֲנִיּוּת. וְעַל כֵּן כְּשֶׁנּוֹתֵן צְדָקָה וּמְבַטֵּל הָעֲנִיּוּת, וּכְמוֹ שֶׁאָמְרוּ, עַשֵֹּר בִּשְׁבִיל שֶׁתִּתְעַשֵּׁר, וּכְתִיב, "וּבְחָנוּנִי נָא וְכוּ' אִם לֹא אֶפְתַּח וְכוּ'", כִּי הַצְּדָקָה בְּחִינַת רָצוֹן טוֹב, בְּחִינַת נְדִיבוּת לֵב וְעַל כֵּן הֻזְהַרְנוּ בִּצְדָקָה, "וְלֹא יֵרַע לְבָבְךָ בְּתִתְּךָ לוֹ", כִּי צָרִיךְ לִתֵּן בְּרָצוֹן טוֹב וּבְשִֹמְחָה. וְעַל כֵּן עַל יְדֵי זֶה מְבַטֵּל הַשִּׁנּוּיִים, שֶׁזֶּהוּ עִקַּר קִיּוּם כָּל הַתּוֹרָה וְכוּ' כַּנַּ"ל:

אז יוצא שלזכות להתקשרות לצדיק ולשלום, שזה בחינת לקשר את כל הרצונות לשורש הרצון, וע"י זה להכניע את האפיקורסים להנאתם שמשם נמשכים היזקי גניבה ואבידה לעולם, וביניהם מכירת יוסף צדיק הדור, זה בעיקר ע"י צדקה. ומה נורא להתבונן, שמכל העלילות שיכלו להעליל על מורנו הרב שליט"א גלגלה ההשגחה שמורנו הרב שליט"א נעצר דווקא על קבלת פדיונות. כל בעל שכל רואה שההאשמות על הרב תמוהות, ובלתי נמנע לשאול על מה ולמה בחרו בקו תביעה כל כך מוזר. אך כך דרכה של ההשגחה לרמז לנו במה חטאנו. בדיוק כמו שיוסף ניסה לרמוז לאחים כל הזמן שהפגם הוא גניבת הצדיק ע"י שהאשים אותם בגניבת הגביע, כך ההשגחה מנסה לרמוז לנו שהפגם שגרם לגניבתו של הצדיק שלנו, הוא צדקה. לא מספיק עזרנו אחד לשני. לא מספיק עזרנו לצדיק. היינו שקועים בתאות ממון, שמרנו הכל לעצמנו..

# צדקה בימים הנוראים

לא בכדי הרב התקשר דווקא לפני התקופה הזאת וביקש ללמוד תפילה זו בעיון. ידועה ומפורסמת מעלת נתינת הפדיון לפני ראש השנה, אך חיזקו אותנו בנוסף דברי הליקו"ה הבאים על צדקה בתקופה זו של לפני הימים הנוראים ובמהלכם:

## ליקו"ה חו"מ הל' ערב ה"ג אות לה

... וְעַל כֵּן אֵלּוּ הַיָּמִים נִקְרָאִים "יָמִים נוֹרָאִים" שֶׁהֵם אֱלוּל וַעֲשֶֹרֶת יְמֵי תְּשׁוּבָה עַד יוֹם כִּפּוּר, כִּי אָז עִקַּר הִתְגַּלּוּת עֹצֶם הַיִּרְאָה דִּקְדֻשָּׁה שֶׁנִּמְשֶׁכֶת עַל יְדֵי הִתְגַּלּוּת הָרָצוֹן שֶׁמִּתְגַּלֶּה אָז בְּיוֹתֵר כַּנַּ"ל:

וְעַל כֵּן נוֹהֲגִין כָּל יִשְֹרָאֵל לְהַרְבּוֹת בִּצְדָקָה מְאֹד בְּאֵלּוּ הַיָּמִים, כִּי עִקַּר הַתִּקּוּן עַל יְדֵי הַצְּדָקָה, כִּי עִקַּר הַקִּלְקוּל הוּא עַל יְדֵי מֵצַח הַנָּחָשׁ כוּ' כַּנַּ"ל וּצְרִיכִין עַכְשָׁו לְהַכְנִיעוֹ וּלְבַטְּלוֹ וְעִקַּר הַכְנָעָתוֹ הוּא עַל יְדֵי צְדָקָה כַּנַּ"ל, עַל כֵּן צְרִיכִין אָז בַּיָּמִים הַנּוֹרָאִים לְהַרְבּוֹת מְאֹד בִּצְדָקָה לַעֲנִיִּים הֲגוּנִים כְּדֵי לְהַכְנִיעַ מֵצַח הַנָּחָשׁ כְּדֵי לְגַלּוֹת הֶאָרַת הָרָצוֹן כְּדֵי לִזְכּוֹת לְיִרְאָה שֶׁהִוא עִקַּר בְּחִינַת רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים, בְּחִינַת וּבְכֵן תֵּן פַּחְדְּךָ וְכוּ':

# עיקר הצדקה

אנחנו מחפשים את הדרך הכי טובה לעזור למורנו הרב שליט"א, לכן חשוב לדעת מהו עיקר מצוות הצדקה. אספנו כמה מקורות בנושא מהליקו"ה:

## ליקו"ה או"ח הל' בית הכנסת ה"ו אות כד

וְזֶה יָדוּעַ שֶׁעִקַּר הַצְּדָקָה הוּא כְּשֶׁזּוֹכֶה לִתֵּן לַעֲנִיִּים הֲגוּנִים בֶּאֱמֶת הָעוֹסְקִים בַּתּוֹרָה וּתְפִלָּה בֶּאֱמֶת וְהָעִקָּר שֶׁיִּזְכֶּה לִתֵּן לְצַדִּיקֵי אֱמֶת וְהַנִּלְוִים אֲלֵיהֶם שֶׁהֵם כְּלוּלִים מִכַּמָּה נַפְשׁוֹת יִשְרָאֵל שֶׁאָז הוּא זוֹכֶה לְהִתְחַבֵּר וּלְהִתְקַשֵּׁר לְהַקִּבּוּץ הַקָּדוֹשׁ שֶׁל הַרְבֵּה נַפְשׁוֹת יִשְרָאֵל שֶׁהֵם עִקַּר קִיּוּם הַמִּשְׁכָּן וּבֵית הַמִּקְדָּשׁ...

## ליקו"ה יו"ד הל' צדקה ה"ב אות ו'

וְעַל כֵּן פִּדְיוֹן שְׁבוּיִים הִוא מִצְוָה גְּדוֹלָה. וּצְדָקָה זוֹ שֶׁל פִּדְיוֹן שְׁבוּיִים גָּדוֹל מִכָּל הַצְּדָקוֹת, כַּמְבֹאָר בַּשֻּׁלְחָן עָרוּךְ, כִּי בֶּאֱמֶת כָּל הַצְּדָקוֹת הֵם בִּבְחִינַת פִּדְיוֹן שְׁבוּיִים, כִּי עַל יְדֵי הַצְּדָקוֹת מַעֲלֶה כַּמָּה נִיצוֹצוֹת הַקְּדוֹשִׁים מִן הַשִּׁבְיָה וְהַגָּלוּת וְכִבְיָכוֹל, מַעֲלֶה הַשְּׁכִינָה מֵהַגָּלוּת וְכַנַּ"ל. וְעַל כֵּן הוּא בְּחִינַת פִּדְיוֹן שְׁבוּיִים. וְעַל כֵּן כְּשֶׁיֵּשׁ פִּדְיוֹן שְׁבוּיִים מַמָּשׁ, דְּהַיְנוּ שֶׁאֶחָד מִיִּשְרָאֵל שָׁבוּי חַס וְשָׁלוֹם, זֶהוּ עִקַּר מִצְוַת הַצְּדָקָה, כִּי כְּשֶׁיִּשְרָאֵל שָׁבוּי הוּא גָּלוּת הַשְּׁכִינָה מַמָּשׁ. וְעַל כֵּן עַל יְדֵי צְדָקָה זוֹ שֶׁל פִּדְיוֹן שְׁבוּיִים הוּא מַעֲלֶה בְּיוֹתֵר הַשְּׁכִינָה מֵהַגָּלוּת וְהַשִּׁבְיָה כִּבְיָכוֹל, שֶׁזֶּהוּ עִקַּר בְּחִינַת צְדָקָה כַּנַּ"ל. וְעַל כֵּן צְדָקָה זוֹ גְּדוֹלָה מִן הַכֹּל:

## ליקו"ה או"ח הל' ברכות השחר ה"ה אות לה

וְעִקַּר הַצְּדָקָה לִתֵּן לַעֲנִיִּים הֲגוּנִים וְכוּ' הַנִּכְלָלִים בְּהַזְּקֵנִים דִּקְדֻשָּׁה הַמּוֹסִיפִים קְדֻשָּׁה וְכוּ' וּמִתְחַדְּשִׁים עַצְמָן בְּכָל פַּעַם, וְזֶה מַעֲלַת הַמְסַיֵּעַ בְּצִדְקָתוֹ לְהַדְפִּיס חִדּוּשֵׁי תּוֹרָה הֲגוּנִים עַל יְסוֹדוֹת הַצַּדִּיקִים הָאֲמִתִּיִּים הַנַּ"ל, כִּי עַל יְדֵי זֶה זוֹכֶה בְּיוֹתֵר וְיוֹתֵר לְתַקֵּן הַנַּעֲשֶה עַל יְדֵי הַצְּדָקָה שֶׁהוּא לְהַעֲלוֹת הַדַּעַת שֶׁנָּפַל מֵהַזְּקֵנִים שֶׁאֵינָם כָּרָאוּי וְכוּ', כְּמוֹ שֶׁמְּבֹאָר שָׁם, כִּי עַל יְדֵי צִדְקָתוֹ שֶׁנּוֹתֵן בִּשְׁבִיל חִדּוּשֵׁי הַתּוֹרָה הַנַּ"ל שֶׁיָּאִירוּ בָּעוֹלָם, עַל יְדֵי זֶה מַמְשִׁיךְ הֶאָרַת הַזְּקֵנִים דִּקְדֻשָּׁה וְכוּ' וְכַנַּ"ל:

אז יוצא שעיקר הצדקה הוא לתת לצדיק ולאנשיו, כשזאת מעלת המסייע להדפיס חידושי תורה על יסודות הצדיקים האמיתיים, והצדקה הגדולה מכולם היא פדיון שבויים. לאור הליקו"ה האלה, ראינו לנכון להציע שעיקר הצדקה שניתן בכוונה להציל את מורינו הרב שליט"א תלך לשלושה מקומות: לאנ"ש, לפדיון שבויים לרב ולהפצת התפילות, הספרים והתורות של הרב.

מי יתן ובזכות זה נזכה להציל את מורנו הרב שליט"א, ולראותו שוב איתנו, בריא, איתן ובן חורין.
